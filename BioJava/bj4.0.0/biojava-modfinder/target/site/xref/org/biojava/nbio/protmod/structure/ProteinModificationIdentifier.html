246 	/**
247 	 * Identify a set of modifications in a a list of chains.
248 	 * @param chains query {@link Chain}s.
249 	 * @param potentialModifications query {@link ProteinModification}s.
250 	 */
251 	public void identify(final List<Chain> chains,
252 			final Set<ProteinModification> potentialModifications) {
253 		
254 		if (chains==null) {
255 			throw new IllegalArgumentException("Null structure.");
256 		}
257 		
258 		if (potentialModifications==null) {
259 			throw new IllegalArgumentException("Null potentialModifications.");
260 		}
261 		
262 			
263 		reset();
264 		
265 		if (potentialModifications.isEmpty()) {
266 			return;
267 		}
268 		
269 		Map<String, Chain> mapChainIdChain = new HashMap<String, Chain>(chains.size());
270 		residues = new ArrayList<Group>();
271 		List<Group> ligands = new ArrayList<Group>();
272 		Map<Component, Set<Group>> mapCompGroups = new HashMap<Component, Set<Group>>();
273 		
274 		for (Chain chain : chains) {
275 			mapChainIdChain.put(chain.getChainID(), chain);
276 					
277 			List<Group> ress = StructureUtil.getAminoAcids(chain);
278 			
279 			
280 			//List<Group> ligs = chain.getAtomLigands();
281 			List<Group> ligs = StructureTools.filterLigands(chain.getAtomGroups());
282 			residues.addAll(ress);
283                         residues.removeAll(ligs);
284 			ligands.addAll(ligs);
285 			addModificationGroups(potentialModifications, ress, ligs, mapCompGroups);
286 		}
287 		
288 		if (residues.isEmpty()) {
289 			String pdbId = "?";
290 			if ( chains.size() > 0) {
291 				Structure struc = chains.get(0).getParent();
292 				if ( struc != null)
293 					pdbId = struc.getPDBCode(); 
294 			}
295 			logger.warn("No amino acids found for {}. Either you did not parse the PDB file with alignSEQRES records, or this record does not contain any amino acids.", pdbId);
296 		}
297 		List<ModifiedCompound> modComps = new ArrayList<ModifiedCompound>();
298 		
299 		for (ProteinModification mod : potentialModifications) {
300 			ModificationCondition condition = mod.getCondition();
301 			List<Component> components = condition.getComponents();
302 			if (!mapCompGroups.keySet().containsAll(components)) {
303 				// not all components exist for this mod.
304 				continue;
305 			}
306 			
307 			int sizeComps = components.size();
308 			if (sizeComps==1) {
309 				
310 				processCrosslink1(mapCompGroups, modComps, mod, components);
311 			
312 			} else {
313 				
314 				processMultiCrosslink(mapCompGroups, modComps, mod, condition);
315 			}
316 		}
317 
318 		if (recordAdditionalAttachments) {
319 			// identify additional groups that are not directly attached to amino acids.
320 			for (ModifiedCompound mc : modComps) {
321 				identifyAdditionalAttachments(mc, ligands, mapChainIdChain);
322 			}
323 		}
324 		
325 		mergeModComps(modComps);
326 		
327 		identifiedModifiedCompounds.addAll(modComps);
328 		
329 		
330 		// record unidentifiable linkage
331 		if (recordUnidentifiableModifiedCompounds) {
332 			recordUnidentifiableAtomLinkages(modComps, ligands);
333 			recordUnidentifiableModifiedResidues(modComps);
334 		}
335 	}

337 	private void reset() {
338 		identifiedModifiedCompounds = new LinkedHashSet<ModifiedCompound>();
339 		if (recordUnidentifiableModifiedCompounds) {
340 			unidentifiableAtomLinkages = new LinkedHashSet<StructureAtomLinkage>();
341 			unidentifiableModifiedResidues = new LinkedHashSet<StructureGroup>();
342 		}
343 		
344 	}

346 	private void processMultiCrosslink(
347 			Map<Component, Set<Group>> mapCompGroups,
348 			List<ModifiedCompound> modComps, ProteinModification mod,
349 			ModificationCondition condition) {
350 		// for multiple components
351 		
352 		// find linkages first
353 		List<List<Atom[]>> matchedAtomsOfLinkages =
354 				getMatchedAtomsOfLinkages(condition, mapCompGroups);
355 		
356 		if (matchedAtomsOfLinkages.size() != condition.getLinkages().size()) {
357 			return;
358 		} 
359 		
360 		assembleLinkages(matchedAtomsOfLinkages, mod, modComps);
361 		
362 	}

364 	private void processCrosslink1(Map<Component, Set<Group>> mapCompGroups,
365 			List<ModifiedCompound> modComps, ProteinModification mod,
366 			List<Component> components) {
367 		// modified residue
368 		// TODO: is this the correct logic for CROSS_LINK_1?
369 		Set<Group> modifiedResidues = mapCompGroups.get(components.get(0));
370 		if (modifiedResidues != null) {
371 			for (Group residue : modifiedResidues) {
372 				StructureGroup strucGroup = StructureUtil.getStructureGroup(residue, true);
373 				ModifiedCompound modRes = new ModifiedCompoundImpl(mod, strucGroup);
374 				modComps.add(modRes);
375 			}
376 		}
377 	}

379 	/**
380 	 * identify additional groups that are not directly attached to amino acids.
381 	 * @param mc {@link ModifiedCompound}.
382 	 * @param chain a {@link Chain}.
383 	 * @return a list of added groups.
384 	 */
385 	private void identifyAdditionalAttachments(ModifiedCompound mc, 
386 			List<Group> ligands, Map<String, Chain> mapChainIdChain) {
387 		if (ligands.isEmpty()) {
388 			return;
389 		}
390 		
391 		// TODO: should the additional groups only be allowed to the identified 
392 		// ligands or both amino acids and ligands? Currently only on ligands
393 		// ligands to amino acid bonds for same modification of unknown category
394 		// will be combined in mergeModComps()
395 		// TODO: how about chain-chain links?
396 		List<Group> identifiedGroups = new ArrayList<Group>();
397 		for (StructureGroup num : mc.getGroups(false)) {
398 			Group group;
399 			try {
400 				//String numIns = "" + num.getResidueNumber();
401 				//if (num.getInsCode() != null) {
402 				//	numIns += num.getInsCode();
403 				//}
404 				ResidueNumber resNum = new ResidueNumber();
405 				resNum.setChainId(num.getChainId());
406 				resNum.setSeqNum(num.getResidueNumber());
407 				resNum.setInsCode(num.getInsCode());
408 				//group = chain.getGroupByPDB(numIns);
409 				group = mapChainIdChain.get(num.getChainId()).getGroupByPDB(resNum);
410 			} catch (StructureException e) {
411 				logger.error("Exception: ", e);
412 				// should not happen
413 				continue;
414 			}
415 			identifiedGroups.add(group);
416 		}
417 		
418 		int start = 0;
419 		
420 		int n = identifiedGroups.size();
421 		while (n > start) {
422 			for (Group group1 : ligands) {
423 				for (int i=start; i<n; i++) {
424 					Group group2 = identifiedGroups.get(i);
425 					if (!identifiedGroups.contains(group1)) {
426 						List<Atom[]> linkedAtoms = StructureUtil.findAtomLinkages(
427 								group1, group2, false, bondLengthTolerance);
428 						if (!linkedAtoms.isEmpty()) {
429 							for (Atom[] atoms : linkedAtoms) {
430 								mc.addAtomLinkage(StructureUtil.getStructureAtomLinkage(atoms[0], 
431 										false, atoms[1], false));
432 							}
433 							identifiedGroups.add(group1);
434 							break;
435 						}
436 					}
437 				}
438 			}
439 			
440 			start = n;
441 			n = identifiedGroups.size();
442 		}
443 	}

445 	/**
446 	 * Merge identified modified compounds if linked.
447 	 */
448 	private void mergeModComps(List<ModifiedCompound> modComps) {
449 		TreeSet<Integer> remove = new TreeSet<Integer>();
450 		int n = modComps.size();
451 		for (int icurr=1; icurr<n; icurr++) {
452 			ModifiedCompound curr = modComps.get(icurr);
453 			
454 			String id = curr.getModification().getId();
455 			if (ProteinModificationRegistry.getById(id).getCategory()
456 					!=ModificationCategory.UNDEFINED)
457 				continue;
458 			
459 			// find linked compounds that before curr
460 			//List<Integer> merging = new ArrayList<Integer>();
461 			int ipre = 0;
462 			for (; ipre<icurr; ipre++) {
463 				if (remove.contains(ipre))	continue;
464 				ModifiedCompound pre = modComps.get(ipre);
465 				if (!Collections.disjoint(pre.getGroups(false),
466 						curr.getGroups(false))) {
467 					break;
468 				}
469 			}
470 			
471 			if (ipre<icurr) {				
472 				ModifiedCompound mcKeep = modComps.get(ipre);
473 				
474 				// merge modifications of the same type
475 				if (mcKeep.getModification().getId().equals(id)) {
476 					// merging the current one to the previous one
477 					mcKeep.addAtomLinkages(curr.getAtomLinkages());
478 					remove.add(icurr);
479 				}
480 			}
481 		}
482 		
483 		Iterator<Integer> it = remove.descendingIterator();
484 		while (it.hasNext()) {
485 			modComps.remove(it.next().intValue());
486 		}
487 	}

489 	/**
490 	 * Record unidentifiable atom linkages in a chain. Only linkages between two
491 	 * residues or one residue and one ligand will be recorded.
492 	 */
493 	private void recordUnidentifiableAtomLinkages(List<ModifiedCompound> modComps,
494 			List<Group> ligands) {
495 		
496 		// first put identified linkages in a map for fast query
497 		Set<StructureAtomLinkage> identifiedLinkages = new HashSet<StructureAtomLinkage>();
498 		for (ModifiedCompound mc : modComps) {
499 			identifiedLinkages.addAll(mc.getAtomLinkages());
500 		}
501 		
502 		// record
503 		// cross link
504 		int nRes = residues.size();
505 		for (int i=0; i<nRes-1; i++) {
506 			Group group1 = residues.get(i);
507 			for (int j=i+1; j<nRes; j++) {
508 				Group group2 = residues.get(j);
509 				List<Atom[]> linkages = StructureUtil.findAtomLinkages(
510 						group1, group2, true, bondLengthTolerance);
511 				for (Atom[] atoms : linkages) {
512 					StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(atoms[0], 
513 							true, atoms[1], true);
514 					unidentifiableAtomLinkages.add(link);
515 				}
516 			}
517 		}
518 		
519 		// attachment
520 		int nLig = ligands.size();
521 		for (int i=0; i<nRes; i++) {
522 			Group group1 = residues.get(i);
523 			for (int j=0; j<nLig; j++) {
524 				Group group2 = ligands.get(j);
525 				if (group1.equals(group2)) { // overlap between residues and ligands
526 					continue;
527 				}
528 				List<Atom[]> linkages = StructureUtil.findAtomLinkages(
529 						group1, group2, false, bondLengthTolerance);
530 				for (Atom[] atoms : linkages) {
531 					StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(atoms[0], 
532 							true, atoms[1], false);
533 					unidentifiableAtomLinkages.add(link);
534 				}
535 			}
536 		}
537 	}

539 	private void recordUnidentifiableModifiedResidues(List<ModifiedCompound> modComps) {
540 		Set<StructureGroup> identifiedComps = new HashSet<StructureGroup>();
541 		for (ModifiedCompound mc : modComps) {
542 			identifiedComps.addAll(mc.getGroups(true));
543 		}
544 		
545 		// TODO: use the ModifiedAminoAcid after Andreas add that.
546 		for (Group group : residues) {
547 			if (group.getType().equals(GroupType.HETATM)) {
548 				StructureGroup strucGroup = StructureUtil.getStructureGroup(
549 						group, true);
550 				if (!identifiedComps.contains(strucGroup)) {
551 					unidentifiableModifiedResidues.add(strucGroup);
552 				}
553 			}
554 		}
555 	}

557 	/**
558 	 * 
559 	 * @param modifications a set of {@link ProteinModification}s.
560 	 * @param residues
561 	 * @param ligands 
562 	 * @param saveTo save result to
563 	 * @return map from component to list of corresponding residues
564 	 *  in the chain.
565 	 */
566 	private void addModificationGroups(
567 			final Set<ProteinModification> modifications,
568 			final List<Group> residues,
569 			final List<Group> ligands,
570 			final Map<Component, Set<Group>> saveTo) {
571 		if (residues==null || ligands==null || modifications==null) {
572 			throw new IllegalArgumentException("Null argument(s).");
573 		}
574 		
575 		Map<Component,Set<Component>> mapSingleMultiComps = new HashMap<Component,Set<Component>>();
576 		for (ProteinModification mod : modifications) {
577 			ModificationCondition condition = mod.getCondition();
578 			for (Component comp : condition.getComponents()) {
579 				for (String pdbccId : comp.getPdbccIds()) {
580 					Component single = Component.of(Collections.singleton(pdbccId), 
581 							comp.isNTerminal(), comp.isCTerminal());
582 					Set<Component> mult = mapSingleMultiComps.get(single);
583 					if (mult == null) {
584 						mult = new HashSet<Component>();
585 						mapSingleMultiComps.put(single, mult);
586 					}
587 					mult.add(comp);
588 				}
589 			}
590 		}
591 		
592 		{
593 			// ligands
594 			Set<Component> ligandsWildCard = mapSingleMultiComps.get(
595 					Component.of("*"));
596 			for (Group group : ligands) {
597 				String pdbccId = group.getPDBName().trim();
598 				Set<Component> comps = mapSingleMultiComps.get(
599 						Component.of(pdbccId));
600 				
601 				for (Component comp : unionComponentSet(ligandsWildCard, comps)) {
602 					Set<Group> gs = saveTo.get(comp);
603 					if (gs==null) {
604 						gs = new LinkedHashSet<Group>();
605 						saveTo.put(comp, gs);
606 					}
607 					gs.add(group);
608 				}
609 			}
610 		}
611 		
612 		{
613 			// residues
614 			if (residues.isEmpty()) {
615 				return;
616 			}
617 			
618 			Set<Component> residuesWildCard = mapSingleMultiComps.get(
619 					Component.of("*"));
620 			
621 			// for all residues
622 			for (Group group : residues) {
623 				String pdbccId = group.getPDBName().trim();
624 				Set<Component> comps = mapSingleMultiComps.get(
625 						Component.of(pdbccId));
626 				
627 				for (Component comp : unionComponentSet(residuesWildCard, comps)) {
628 					Set<Group> gs = saveTo.get(comp);
629 					if (gs==null) {
630 						gs = new LinkedHashSet<Group>();
631 						saveTo.put(comp, gs);
632 					}
633 					gs.add(group);
634 				}
635 			}
636 
637 			// for N-terminal
638 			int nRes = residues.size();
639 			int iRes = 0;
640 			Group res;
641 			do {
642 				// for all ligands on N terminal and the first residue
643 				res = residues.get(iRes++);
644 
645 				Set<Component> nTermWildCard = mapSingleMultiComps.get(
646 						Component.of("*", true, false));
647 
648 				Set<Component> comps = mapSingleMultiComps.get(
649 						Component.of(res.getPDBName(), true, false));
650 				
651 				for (Component comp : unionComponentSet(nTermWildCard, comps)) {
652 					Set<Group> gs = saveTo.get(comp);
653 					if (gs==null) {
654 						gs = new LinkedHashSet<Group>();
655 						saveTo.put(comp, gs);
656 					}
657 					gs.add(res);
658 				}
659 			} while (iRes<nRes && ligands.contains(res));
660 			
661 			// for C-terminal
662 			iRes = residues.size()-1;
663 			do {
664 				// for all ligands on C terminal and the last residue
665 				res = residues.get(iRes--);
666 
667 				Set<Component> cTermWildCard = mapSingleMultiComps.get(
668 						Component.of("*", false, true));
669 				
670 				Set<Component> comps = mapSingleMultiComps.get(
671 						Component.of(res.getPDBName(), false, true));
672 
673 				for (Component comp : unionComponentSet(cTermWildCard, comps)) {
674 					Set<Group> gs = saveTo.get(comp);
675 					if (gs==null) {
676 						gs = new LinkedHashSet<Group>();
677 						saveTo.put(comp, gs);
678 					}
679 					gs.add(res);
680 				}
681 			} while (iRes>=0 && ligands.contains(res));
682 		}
683 	}

685 	private Set<Component> unionComponentSet(Set<Component> set1, Set<Component> set2) {
686 		if (set1 == null && set2 == null)
687 			return Collections.emptySet();
688 		
689 		if (set1 == null)
690 			return set2;
691 		
692 		if (set2 == null)
693 			return set1;
694 		
695 		Set<Component> set = new HashSet<Component>(set1.size()+set2.size());
696 		set.addAll(set1);
697 		set.addAll(set2);
698 		
699 		return set;
700 	}

702 	/**
703 	 * Get matched atoms for all linkages.	
704 	 */
705 	private List<List<Atom[]>> getMatchedAtomsOfLinkages(
706 			ModificationCondition condition, Map<Component, Set<Group>> mapCompGroups) {
707 		List<ModificationLinkage> linkages = condition.getLinkages();
708 		int nLink = linkages.size();
709 
710 		List<List<Atom[]>> matchedAtomsOfLinkages = 
711 				new ArrayList<List<Atom[]>>(nLink);
712 		
713 		for (int iLink=0; iLink<nLink; iLink++) {
714 			ModificationLinkage linkage = linkages.get(iLink);
715 			Component comp1 = linkage.getComponent1();
716 			Component comp2 = linkage.getComponent2();
717 
718 //			boolean isAA1 = comp1.;
719 //			boolean isAA2 = comp2.getType()==true;
720 			
721 			Set<Group> groups1 = mapCompGroups.get(comp1);
722 			Set<Group> groups2 = mapCompGroups.get(comp2);
723 			
724 			List<Atom[]> list = new ArrayList<Atom[]>();
725 
726 			List<String> potentialNamesOfAtomOnGroup1 = linkage.getPDBNameOfPotentialAtomsOnComponent1();
727 			for (String name : potentialNamesOfAtomOnGroup1) {
728 				if (name.equals("*")) {
729 					// wildcard
730 					potentialNamesOfAtomOnGroup1 = null; // search all atoms
731 					break;
732 				}
733 			}
734 			
735 			List<String> potentialNamesOfAtomOnGroup2 = linkage.getPDBNameOfPotentialAtomsOnComponent2();
736 			for (String name : potentialNamesOfAtomOnGroup2) {
737 				if (name.equals("*")) {
738 					// wildcard
739 					potentialNamesOfAtomOnGroup2 = null; // search all atoms
740 					break;
741 				}
742 			}
743 
744 			for (Group g1 : groups1) {
745 				for (Group g2 : groups2) {
746 					if (g1.equals(g2)) {
747 						continue;
748 					}
749 
750                                         // only for wildcard match of two residues
751                                         boolean ignoreNCLinkage = 
752                                                 potentialNamesOfAtomOnGroup1 == null &&
753                                                 potentialNamesOfAtomOnGroup2 == null &&
754                                                 residues.contains(g1) &&
755                                                 residues.contains(g2);
756 		
757 					Atom[] atoms = StructureUtil.findNearestAtomLinkage(
758 							g1, g2, 
759 							potentialNamesOfAtomOnGroup1,
760 							potentialNamesOfAtomOnGroup2,
761                                                         ignoreNCLinkage,
762 							bondLengthTolerance);
763 					if (atoms!=null) {
764 						list.add(atoms);
765 					}
766 				}
767 			}
768 				
769 			if (list.isEmpty()) {
770 				// broken linkage
771 				break;
772 			}
773 	
774 			matchedAtomsOfLinkages.add(list);
775 		}
776 		
777 		return matchedAtomsOfLinkages;
778 	}

780 	/**
781 	 * Assembly the matched linkages.
782 	 * @param matchedAtomsOfLinkages
783 	 * @param mod
784 	 * @param condition
785 	 * @param ret ModifiedCompound will be stored here.
786 	 */
787 	private void assembleLinkages(List<List<Atom[]>> matchedAtomsOfLinkages,
788 			ProteinModification mod, List<ModifiedCompound> ret) {
789 		ModificationCondition condition = mod.getCondition();
790 		List<ModificationLinkage> modLinks = condition.getLinkages();
791 		
792 		int nLink = matchedAtomsOfLinkages.size();
793 		int[] indices = new int[nLink];
794 		Set<ModifiedCompound> identifiedCompounds = new HashSet<ModifiedCompound>();
795 		while (indices[0]<matchedAtomsOfLinkages.get(0).size()) {
796 			List<Atom[]> atomLinkages = new ArrayList<Atom[]>(nLink);
797 			for (int iLink=0; iLink<nLink; iLink++) {
798 				Atom[] atoms = matchedAtomsOfLinkages.get(iLink).get(indices[iLink]);
799 				atomLinkages.add(atoms);
800 			}
801 			if (matchLinkages(modLinks, atomLinkages)) {
802 				// matched
803 				
804 				int n = atomLinkages.size();
805 				List<StructureAtomLinkage> linkages = new ArrayList<StructureAtomLinkage>(n);
806 				for (int i=0; i<n; i++) {
807 					Atom[] linkage = atomLinkages.get(i);
808 					StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(
809 							linkage[0], residues.contains(linkage[0].getGroup()),
810 							linkage[1], residues.contains(linkage[1].getGroup()));
811 					linkages.add(link);
812 				}
813 				
814 				ModifiedCompound mc = new ModifiedCompoundImpl(mod, linkages);
815 				if (!identifiedCompounds.contains(mc)) {
816 					ret.add(mc);
817 					identifiedCompounds.add(mc);
818 				}
819 			}
820 			
821 			// indices++ (e.g. [0,0,1]=>[0,0,2]=>[1,2,0])
822 			int i = nLink-1;
823 			while (i>=0) {
824 				if (i==0 || indices[i]<matchedAtomsOfLinkages.get(i).size()-1) {
825 					indices[i]++;
826 					break;
827 				} else {
828 					indices[i] = 0;
829 					i--;
830 				}
831 			}
832 		}
833 	}

835 	/**
836 	 * 
837 	 * @param linkages
838 	 * @param atomLinkages
839 	 * @return true if atomLinkages satisfy the condition; false, otherwise.
840 	 */
841 	private boolean matchLinkages(List<ModificationLinkage> linkages, 
842 			List<Atom[]> atomLinkages) {
843 		int nLink = linkages.size();
844 		if (nLink != atomLinkages.size()) {
845 			return false;
846 		}
847 		for (int i=0; i<nLink-1; i++) {
848 			ModificationLinkage link1 = linkages.get(i);
849 			Atom[] atoms1 = atomLinkages.get(i);
850 			for (int j=i+1; j<nLink; j++) {
851 				ModificationLinkage link2 = linkages.get(j);
852 				Atom[] atoms2 = atomLinkages.get(j);
853 				
854 				// check components
855 				if (((link1.getIndexOfComponent1()==link2.getIndexOfComponent1())
856 							!= (atoms1[0].getGroup().equals(atoms2[0].getGroup())))
857 					|| ((link1.getIndexOfComponent1()==link2.getIndexOfComponent2())
858 							!= (atoms1[0].getGroup().equals(atoms2[1].getGroup())))
859 					|| ((link1.getIndexOfComponent2()==link2.getIndexOfComponent1())
860 							!= (atoms1[1].getGroup().equals(atoms2[0].getGroup())))
861 					|| ((link1.getIndexOfComponent2()==link2.getIndexOfComponent2())
862 							!= (atoms1[1].getGroup().equals(atoms2[1].getGroup())))) {
863 					return false;
864 				}
865 				
866 				// check atoms
867 				String label11 = link1.getLabelOfAtomOnComponent1();
868 				String label12 = link1.getLabelOfAtomOnComponent2();
869 				String label21 = link2.getLabelOfAtomOnComponent1();
870 				String label22 = link2.getLabelOfAtomOnComponent2();
871 				if ((label11!=null && label21!=null && label11.equals(label21))
872 							!= (atoms1[0].equals(atoms2[0]))
873 					 || (label11!=null && label22!=null && label11.equals(label22))
874 							!= (atoms1[0].equals(atoms2[1]))
875 					 || (label12!=null && label21!=null && label12.equals(label21))
876 							!= (atoms1[1].equals(atoms2[0]))
877 					 || (label12!=null && label22!=null && label12.equals(label22))
878 							!= (atoms1[1].equals(atoms2[1]))) {
879 					return false;
880 				}
881 			}
882 		}
883 		
884 		return true;
885 	}

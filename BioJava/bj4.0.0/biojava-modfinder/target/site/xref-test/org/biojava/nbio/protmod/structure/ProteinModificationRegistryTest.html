45  	 * Note: if you change this unit test, also change the cook book:
46  	 * http://www.biojava.org/wiki/BioJava:CookBook3:AddProtMod
47  	 */
48  	public void testRegisterModification() {
49  		// define the involved components, in this case two cystines (CYS)
50  		List<Component> components = new ArrayList<Component>(2);
51  		components.add(Component.of("CYS"));
52  		components.add(Component.of("CYS"));
53  
54  		// define the atom linkages between the components, in this case the SG atoms on both CYS groups
55  		ModificationLinkage linkage = new ModificationLinkage(components, 0, "SG", 1, "SG");
56  
57  		// define the modification condition, i.e. what components are involved and what atoms are linked between them
58  		ModificationCondition condition = new ModificationConditionImpl(components, Collections.singletonList(linkage));
59  
60  		// build a modification
61  		ProteinModification mod = 
62  			new ProteinModificationImpl.Builder("0018_test", 
63  				ModificationCategory.CROSS_LINK_2,
64  				ModificationOccurrenceType.NATURAL,
65  				condition)
66  			.setDescription("A protein modification that effectively cross-links two L-cysteine residues to form L-cystine.")
67  			.setFormula("C 6 H 8 N 2 O 2 S 2")
68  			.setResidId("AA0025")
69  			.setResidName("L-cystine")
70  			.setPsimodId("MOD:00034")
71  			.setPsimodName("L-cystine (cross-link)")
72  			.setSystematicName("(R,R)-3,3'-disulfane-1,2-diylbis(2-aminopropanoic acid)")
73  			.addKeyword("disulfide bond")
74  			.addKeyword("redox-active center")
75  			.build();
76  
77  		//register the modification
78  		ProteinModificationRegistry.register(mod);
79  		assertNotNull(ProteinModificationRegistry.getById("0018_test"));
80  	}

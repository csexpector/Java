137 	public List<ModifiedCompound> testXMLSerialization(String pdbId){
138 		String xml = null;
139 		ModifiedCompound currentMC = null;
140 		List<ModifiedCompound> all = new ArrayList<ModifiedCompound>();		
141 		try {
142 
143 			Structure struc = TmpAtomCache.cache.getStructure(pdbId);
144 
145 			ProteinModificationIdentifier parser = new ProteinModificationIdentifier();
146 
147 			for (Chain c : struc.getChains()) {
148 
149 				parser.identify(c, ProteinModificationRegistry.allModifications());
150 				Set<ModifiedCompound> mcs = parser.getIdentifiedModifiedCompound();
151 
152 				for (ModifiedCompound mc : mcs){
153 					currentMC = mc;
154 					xml =  doXMLSerialization(mc) ;
155 					//logger.info( pdbId + " got XML: " + String.format("%n") + xml);
156 					ModifiedCompound newMC = getModifiedCompoundFromXML(xml);
157 					String xml2 = doXMLSerialization(newMC);
158 					assertEquals(xml,xml2);
159 					//logger.info(xml2);
160 					//assertEquals("The two objects are not equal before and after XML serialization" , mc, newMC);
161 					//logger.info(mc.getDescription());
162 					//logger.info(newMC.getDescription());
163 					all.add(mc);
164 				}
165 			}
166 		} catch (Exception e){
167 			logger.error("Error when serializing {}", pdbId);
168 			logger.error(currentMC.getDescription());
169 			logger.error(xml, e);
170 			fail(e.getMessage());
171 		}
172 		xml = null;
173 		currentMC =null;
174 		return all;
175 	}

201 	public String convertStreamToString(InputStream is)
202 	throws IOException {
203 		/*
204 		 * To convert the InputStream to String we use the
205 		 * Reader.read(char[] buffer) method. We iterate until the
206 		 * Reader return -1 which means there's no more data to
207 		 * read. We use the StringWriter class to produce the string.
208 		 */
209 		if (is != null) {
210 			Writer writer = new StringWriter(); 
211 			char[] buffer = new char[1024];
212 			try {
213 				Reader reader = new BufferedReader(
214 						new InputStreamReader(is, "UTF-8"));
215 				int n;
216 				while ((n = reader.read(buffer)) != -1) {
217 					writer.write(buffer, 0, n);
218 				}
219 			} finally {
220 				is.close();
221 			}
222 			return writer.toString();
223 		} else {       
224 			return "";
225 		}

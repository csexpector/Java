306 	private void parserTest(String pdbId, String residId) throws IOException, StructureException {
307 		Set<ProteinModification> mods;
308 		if (residId==null) {
309 			mods = ProteinModificationRegistry.allModifications();
310 		} else {
311 			mods = ProteinModificationRegistry.getByResidId(residId);
312 		}
313 		
314 		parserTest(pdbId, mods);
315 	}

317 	private void parserTest(String pdbId, Set<ProteinModification> mods) throws IOException, StructureException {	
318 		Structure struc = TmpAtomCache.cache.getStructure(pdbId);
319 
320 		ProteinModificationIdentifier parser = new ProteinModificationIdentifier();
321 		boolean recordUnidentifiable = false;
322 		parser.setRecordUnidentifiableCompounds(recordUnidentifiable);
323 //		parser.setbondLengthTolerance(2);
324 		
325 		assertFalse(mods.isEmpty());
326 
327 		parser.identify(struc, mods);
328 
329 		if (! parser.getIdentifiedModifiedCompound().isEmpty() ){
330 			logger.warn("Did not identify any modified compounds for {}", pdbId);
331 		}
332 		
333 		assertFalse("Did not identify any modified compounds for " + pdbId , 
334 				parser.getIdentifiedModifiedCompound().isEmpty());
335 		
336 		boolean print = false;
337 		if (print)
338 			printResult(pdbId, parser, recordUnidentifiable);
339 	}

341 	private void printResult(String pdbId, ProteinModificationIdentifier parser, boolean recordUnidentifiable) {
342 		StringBuilder sb = new StringBuilder();
343 
344 		sb.append("===");
345 		sb.append(pdbId);
346 		sb.append("===\n");
347 		
348 		Set<ModifiedCompound> mcs = parser.getIdentifiedModifiedCompound();
349 		
350 		int i=0;
351 		for (ModifiedCompound mc : mcs) {
352 			sb.append("Modification #");
353 			sb.append(++i);
354 			sb.append(":\n");
355 			sb.append(mc);
356 			sb.append('\n');
357 		}
358 		
359 		if (recordUnidentifiable) {
360 			Set<StructureGroup> unidentifiedModifiedResidues = parser.getUnidentifiableModifiedResidues();
361 			i = 0;
362 			for (StructureGroup group : unidentifiedModifiedResidues) {
363 				sb.append("Unidenfied modified residue #");
364 				sb.append(++i);
365 				sb.append(":\n");
366 				sb.append(group);
367 				sb.append('\n');
368 			}
369 	
370 			Set<StructureAtomLinkage> unidentifiedLinkages = parser.getUnidentifiableAtomLinkages();
371 			i = 0;
372 			for (StructureAtomLinkage link : unidentifiedLinkages) {
373 				sb.append("Unidenfied linkage #");
374 				sb.append(++i);
375 				sb.append(":\n");
376 				sb.append(link);
377 				sb.append('\n');
378 			}
379 		}
380 		
381 		logger.info(sb.toString());
382 	}

393 		Set<ModifiedCompound> mcs = identifyAllModfications(struc);

395  
396 		// identify all phosphosites from PDB:3MVJ and print them
397 		pdbId = "3MVJ";
398 		struc = TmpAtomCache.cache.getStructure(pdbId);
399 		List<ResidueNumber> psites = identifyPhosphosites(struc);
400 		assertFalse(psites.isEmpty());
401 	}

407 	private Set<ModifiedCompound> identifyAllModfications(Structure struc) {
408 		ProteinModificationIdentifier parser = new ProteinModificationIdentifier();
409 		parser.identify(struc);
410 		Set<ModifiedCompound> mcs = parser.getIdentifiedModifiedCompound();
411 		return mcs;
412 	}

414 	/**
415 	 * Note: if you change this unit test, also change the cook book:
416 	 * http://www.biojava.org/wiki/BioJava:CookBook3:ProtMod
417 	 */
418 	private List<ResidueNumber> identifyPhosphosites(Structure struc) {
419 		List<ResidueNumber> phosphosites = new ArrayList<ResidueNumber>();
420 		ProteinModificationIdentifier parser = new ProteinModificationIdentifier();
421 		parser.identify(struc, ProteinModificationRegistry.getByKeyword("phosphoprotein"));
422 		Set<ModifiedCompound> mcs = parser.getIdentifiedModifiedCompound();
423 		for (ModifiedCompound mc : mcs) {
424 			Set<StructureGroup> groups = mc.getGroups(true);
425 			for (StructureGroup group : groups) {
426 				phosphosites.add(group.getPDBResidueNumber());
427 			}
428 		}
429 		return phosphosites;
430 	}

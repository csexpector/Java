61  	@BeforeClass
62  	public static void parseSequences() {
63  		InputStream cdsIs = new ClasspathResource(
64  				"org/biojava/nbio/core/sequence/BRCA2-cds.fasta").getInputStream();
65  		InputStream pepIs = new ClasspathResource(
66  				"org/biojava/nbio/core/sequence/BRCA2-peptide.fasta").getInputStream();
67  		InputStream volDnaIs = new ClasspathResource(
68  				"org/biojava/nbio/core/sequence/volvox-cds.fasta").getInputStream();
69  		InputStream volPepIs = new ClasspathResource(
70  				"org/biojava/nbio/core/sequence/volvox-peptide.fasta").getInputStream();
71  
72  		try {
73  			FastaReader<DNASequence, NucleotideCompound> dnaReader = new FastaReader<DNASequence, NucleotideCompound>(cdsIs,
74  					new GenericFastaHeaderParser<DNASequence, NucleotideCompound>(), new DNASequenceCreator(dnaCs));
75  			brca2Dna = dnaReader.process().values().iterator().next();
76  			FastaReader<ProteinSequence, AminoAcidCompound> pReader = new FastaReader<ProteinSequence, AminoAcidCompound>(
77  					pepIs, new GenericFastaHeaderParser<ProteinSequence, AminoAcidCompound>(), new ProteinSequenceCreator(
78  							aaCs));
79  			brca2Pep = pReader.process().values().iterator().next();
80  
81  			FastaReader<DNASequence, NucleotideCompound> volvoxDnaReader = new FastaReader<DNASequence, NucleotideCompound>(volDnaIs,
82  					new GenericFastaHeaderParser<DNASequence, NucleotideCompound>(), new DNASequenceCreator(dnaCs));
83  			volvoxDna = volvoxDnaReader.process().values().iterator().next();
84  			FastaReader<ProteinSequence, AminoAcidCompound> volvoxPepReader = new FastaReader<ProteinSequence, AminoAcidCompound>(
85  					volPepIs, new GenericFastaHeaderParser<ProteinSequence, AminoAcidCompound>(), new ProteinSequenceCreator(
86  							aaCs));
87  			volvoxPep = volvoxPepReader.process().values().iterator().next();
88  		}
89  		catch (IOException e) {
90  			logger.error("Exception: ", e);
91  			Assert.fail("Encountered exception");
92  		}
93  		finally {
94  			close(cdsIs);
95  			close(pepIs);
96  			close(volDnaIs);
97  			close(volPepIs);
98  		}
99  	}

133 	@Test
134 	public void multiFrameTranslation() throws CompoundNotFoundException {
135 		TranscriptionEngine e = TranscriptionEngine.getDefault();
136 		DNASequence dna = new DNASequence("ATGGCGTGA");
137 
138 		Map<Frame, String> expectedTranslations = new EnumMap<Frame, String>(Frame.class) {
139 
140 			{
141 				put(Frame.ONE, "MA");
142 				put(Frame.TWO, "WR");
143 				put(Frame.THREE, "GV");
144 				put(Frame.REVERSED_ONE, "SRH");
145 				put(Frame.REVERSED_TWO, "HA");
146 				put(Frame.REVERSED_THREE, "TP");
147 			}
148 		};
149 
150 		Map<Frame, Sequence<AminoAcidCompound>> translations =
151 				e.multipleFrameTranslation(dna, Frame.getAllFrames());
152 
153 		for (Entry<Frame, Sequence<AminoAcidCompound>> entry : translations.entrySet()) {
154 			String expected = expectedTranslations.get(entry.getKey());
155 			Sequence<AminoAcidCompound> protein = entry.getValue();
156 			assertThat("Checking 6 frame translation", protein.toString(), is(expected));
157 		}
158 	}

210 	@Test
211 	public void waitForStartCodon() throws CompoundNotFoundException{
212 		//Should not start translation until a start codon is encountered
213 		TranscriptionEngine e = new TranscriptionEngine.Builder().waitForStartCodon(true).build();
214 		RNASequence rna = new RNASequence("UCCAUGAGC");
215 		String pep = rna.getProteinSequence(e).getSequenceAsString();
216 		assertThat("Translation starts at Start Codon",pep, is("MS"));
217 		
218 		//And should start at start of sequence (NB this is implied by success of all other tests)
219 		e = new TranscriptionEngine.Builder().waitForStartCodon(false).build();
220 		pep = rna.getProteinSequence(e).getSequenceAsString();
221 		assertThat("Translation starts at start of sequence", pep, is("SMS"));
222 	}

234 	/** test for https://github.com/biojava/biojava/issues/53  */
235 	@Test
236 	public void testHashCollision() throws CompoundNotFoundException{
237 		Builder builder = new TranscriptionEngine.Builder();
238 		builder.initMet(false);
239 		builder.translateNCodons(true);
240 		builder.trimStop(false);
241 		TranscriptionEngine engine = builder.build();
242 		Sequence<AminoAcidCompound> seq=engine.translate(new
243 				DNASequence("GTNTGTTAGTGT"));
244 		assertThat("XC*C", is(seq.toString()));
245 		Sequence<AminoAcidCompound> seq2=engine.translate(new
246 				DNASequence("ANAANG"));
247 		assertEquals("XX",seq2.toString());
248 		assertNotSame("HR",seq2.toString());
249 	}

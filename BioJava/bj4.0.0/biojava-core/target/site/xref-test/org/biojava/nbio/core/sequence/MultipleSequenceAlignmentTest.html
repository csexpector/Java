24  package org.biojava.nbio.core.sequence;

26  import org.biojava.nbio.core.exceptions.CompoundNotFoundException;
27  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
28  import org.biojava.nbio.core.sequence.compound.AminoAcidCompoundSet;
29  import org.biojava.nbio.core.sequence.compound.DNACompoundSet;
30  import org.biojava.nbio.core.sequence.compound.NucleotideCompound;
31  import org.junit.Before;
32  import org.junit.Test;
33  
34  import java.util.ArrayList;
35  import java.util.List;
36  
37  import static org.junit.Assert.assertEquals;
38  
39  public class MultipleSequenceAlignmentTest {
40  
41      private MultipleSequenceAlignment<ProteinSequence, AminoAcidCompound> msaProteins;
42      private MultipleSequenceAlignment<DNASequence,NucleotideCompound> msaDNA;
43  
44      @Before
45      public void setup() throws CompoundNotFoundException {
46          msaProteins = new MultipleSequenceAlignment<ProteinSequence, AminoAcidCompound>();
47          for (int i = 0; i < 8; i++) {
48              msaProteins.addAlignedSequence(new ProteinSequence("ARNDCEQGHILKMFPSTWYVBZJX"));
49          }
50          msaDNA = new MultipleSequenceAlignment<DNASequence, NucleotideCompound>();
51          for (int i = 0; i < 7; i++) {
52              msaDNA.addAlignedSequence(new DNASequence("ATCGATCGATCGATCG")); 
53          }
54      }
55  
56      @Test
57      public void testGetCompoundsAt() {
58          AminoAcidCompound aminoAcid = AminoAcidCompoundSet.getAminoAcidCompoundSet().getCompoundForString("N");
59          List<AminoAcidCompound> colProteins = new ArrayList<AminoAcidCompound>();
60          for (int i = 0; i < 8; i++) {
61              colProteins.add(aminoAcid);
62          }
63          assertEquals(msaProteins.getCompoundsAt(3), colProteins);
64          NucleotideCompound nucleotide = DNACompoundSet.getDNACompoundSet().getCompoundForString("C");
65          List<NucleotideCompound> colDNA = new ArrayList<NucleotideCompound>();
66          for (int i = 0; i < 7; i++) {
67              colDNA.add(nucleotide);
68          }
69          assertEquals(msaDNA.getCompoundsAt(3), colDNA);
70      }

72  }

36  public class WindowViewTests {
37  
38    @Test
39    public void basicWindow() throws CompoundNotFoundException { 
40      RNASequence rna = new RNASequence("AUGCCU");
41      WindowedSequence<NucleotideCompound> window = new WindowedSequence<NucleotideCompound>(rna, 3);
42  
43      Iterator<SequenceView<NucleotideCompound>> iter = window.iterator();
44      assertTrue("hasNext() returns true", iter.hasNext());
45  
46      int count = 0;
47      for(SequenceView<NucleotideCompound> c: window) {
48        count++;
49        if(count == 0) {
50          String extracted = c.getCompoundAt(0).toString() + c.getCompoundAt(1).toString() + c.getCompoundAt(2).toString();
51          assertEquals("Checking codon string", "AUG", extracted);
52        }
53      }
54      assertThat("Windowed iterator broken", count, is(2));
55    }
56  
57    @Test
58    public void reaminderWindow() throws CompoundNotFoundException {
59      RNASequence rna = new RNASequence("AUGCC");
60      WindowedSequence<NucleotideCompound> window = new WindowedSequence<NucleotideCompound>(rna, 3);
61      List<SequenceView<NucleotideCompound>> list = new ArrayList<SequenceView<NucleotideCompound>>();
62      for(SequenceView<NucleotideCompound> c: window) {
63        list.add(c);
64      }
65      assertThat("First window is size 3", list.get(0).getLength(), is(3));
66      assertThat("Only 1 window", list.size(), is(1));
67    }

135     private String parse(BufferedReader bufferedReader) {
136         String sectionKey = null;
137         List<String[]> section;
138         // Get an ordered list of key->value pairs in array-tuples
139         do {
140             section = this.readSection(bufferedReader);
141             sectionKey = ((String[]) section.get(0))[0];
142             if (sectionKey == null) {
143                 throw new ParserException("Section key was null");
144             }
145             // process section-by-section
146             if (sectionKey.equals(LOCUS_TAG)) {
147                 String loc = ((String[]) section.get(0))[1];
148                 header = loc;
149                 Matcher m = lp.matcher(loc);
150                 if (m.matches()) {
151                     headerParser.setName(m.group(1));
152                     headerParser.setAccession(m.group(1)); // default if no accession found
153 
154                     String lengthUnits = m.group(2);
155                     String type = m.group(5);
156 
157                     if (lengthUnits.equals("aa")) {
158                         compoundType = AminoAcidCompoundSet.getAminoAcidCompoundSet();
159                     } else if (lengthUnits.equals("bp")) {
160                         if (type != null) {
161                             if (type.contains("RNA")) {
162                                 compoundType = RNACompoundSet.getRNACompoundSet();
163                             } else {
164                                 compoundType = DNACompoundSet.getDNACompoundSet();
165                             }
166                         } else {
167                             compoundType = DNACompoundSet.getDNACompoundSet();
168                         }
169                     }
170 
171                     log.debug("compound type: {}", compoundType.getClass().getSimpleName());
172 
173                 } else {
174                     throw new ParserException("Bad locus line");
175                 }
176             } else if (sectionKey.equals(DEFINITION_TAG)) {
177                 headerParser.setDescription(((String[]) section.get(0))[1]);
178             } else if (sectionKey.equals(ACCESSION_TAG)) {
179                 // if multiple accessions, store only first as accession,
180                 // and store rest in annotation
181                 String[] accs = ((String[]) section.get(0))[1].split("\\s+");
182                 accession = accs[0].trim();
183                 headerParser.setAccession(accession);
184             } else if (sectionKey.equals(VERSION_TAG)) {
185                 String ver = ((String[]) section.get(0))[1];
186                 Matcher m = vp.matcher(ver);
187                 if (m.matches()) {
188                     String verAcc = m.group(1);
189                     if (!accession.equals(verAcc)) {
190                         // the version refers to a different accession!
191                         // believe the version line, and store the original
192                         // accession away in the additional accession set
193                         accession = verAcc;
194                     }
195                     if (m.group(3) != null) {
196                         headerParser.setVersion(Integer.parseInt(m.group(3)));
197                     }
198                     if (m.group(5) != null) {
199                         headerParser.setIdentifier(m.group(5));
200                     }
201                 } else {
202                     throw new ParserException("Bad version line");
203                 }
204             } else if (sectionKey.equals(KEYWORDS_TAG)) {
205             } else if (sectionKey.equals(SOURCE_TAG)) {
206                 // ignore - can get all this from the first feature
207             } else if (sectionKey.equals(REFERENCE_TAG)) {
208             } else if (sectionKey.equals(COMMENT_TAG)) {
209                 // Set up some comments
210                 headerParser.setComment(((String[]) section.get(0))[1]);
211             } else if (sectionKey.equals(FEATURE_TAG)) {
212                 // starting from second line of input, start a new feature whenever we come across
213                 // a key that does not start with /
214                 AbstractFeature gbFeature = null;
215                 for (int i = 1; i < section.size(); i++) {
216                     String key = ((String[]) section.get(i))[0];
217                     String val = ((String[]) section.get(i))[1];
218                     if (key.startsWith("/")) {
219                         if (gbFeature == null) {
220                             throw new ParserException("Malformed GenBank file: found a qualifier without feature.");
221                         }
222                         key = key.substring(1); // strip leading slash
223                         val = val.replaceAll("\\s*[\\n\\r]+\\s*", " ").trim();
224                         if (val.endsWith("\"")) {
225                             val = val.substring(1, val.length() - 1); // strip quotes
226                         }
227                         // parameter on old feature
228                         if (key.equals("db_xref")) {
229                             Matcher m = dbxp.matcher(val);
230                             if (m.matches()) {
231                                 String dbname = m.group(1);
232                                 String raccession = m.group(2);
233                                 Qualifier xref = new DBReferenceInfo(dbname, raccession);
234                                 gbFeature.addQualifier(key, xref);
235 
236                                 ArrayList<DBReferenceInfo> listDBEntry = new ArrayList<DBReferenceInfo>();
237                                 listDBEntry.add((DBReferenceInfo) xref);
238                                 mapDB.put(key, listDBEntry);
239                             } else {
240                                 throw new ParserException("Bad dbxref");
241                             }
242                         } else if (key.equalsIgnoreCase("organism")) {
243                             Qualifier q = new Qualifier(key, val.replace('\n', ' '));
244                             gbFeature.addQualifier(key, q);
245                         } else {
246                             if (key.equalsIgnoreCase("translation")) {
247                                 // strip spaces from sequence
248                                 val = val.replaceAll("\\s+", "");
249                                 Qualifier q = new Qualifier(key, val);
250                                 gbFeature.addQualifier(key, q);
251                             } else {
252                                 Qualifier q = new Qualifier(key, val);
253                                 gbFeature.addQualifier(key, q);
254                             }
255                         }
256                     } else {
257                         // new feature!
258                         gbFeature = new TextFeature(key, val, key, key);
259                         Location l = 
260                                 locationParser.parse(val);
261                         gbFeature.setLocation((AbstractLocation)l);
262 
263                         if (!featureCollection.containsKey(key)) {
264                             featureCollection.put(key, new ArrayList());
265                         }
266                         featureCollection.get(key).add(gbFeature);
267                     }
268                 }
269             } else if (sectionKey.equals(BASE_COUNT_TAG)) {
270                 // ignore - can calculate from sequence content later if needed
271             } else if (sectionKey.equals(START_SEQUENCE_TAG)) {
272                 // our first line is ignorable as it is the ORIGIN tag
273                 // the second line onwards conveniently have the number as
274                 // the [0] tuple, and sequence string as [1] so all we have
275                 // to do is concat the [1] parts and then strip out spaces,
276                 // and replace '.' and '~' with '-' for our parser.
277                 StringBuffer seq = new StringBuffer();
278                 for (int i = 1; i < section.size(); i++) {
279                     seq.append(((String[]) section.get(i))[1]);
280                 }
281                 seqData = seq.toString().replaceAll("\\s+", "").replaceAll("[\\.|~]", "-").toUpperCase();
282             }
283         } while (!sectionKey.equals(END_SEQUENCE_TAG));
284         return seqData;
285     }

291     // reads an indented section, combining split lines and creating a list of
292     // key->value tuples
293     // reads an indented section, combining split lines and creating a list of
294     // key->value tuples
295     private List<String[]> readSection(BufferedReader bufferedReader) {
296         List<String[]> section = new ArrayList<String[]>();
297         String line = "";
298         String currKey = null;
299         StringBuffer currVal = new StringBuffer();
300         boolean done = false;
301         int linecount = 0;
302 
303         try {
304             while (!done) {
305                 bufferedReader.mark(320);
306                 line = bufferedReader.readLine();
307                 String firstSecKey = section.isEmpty() ? ""
308                         : ((String[]) section.get(0))[0];
309                 if (line != null && line.matches("\\p{Space}*")) {
310 					// regular expression \p{Space}* will match line
311                     // having only white space characters
312                     continue;
313                 }
314                 if (line == null
315                         || (!line.startsWith(" ") && linecount++ > 0 && (!firstSecKey
316                         .equals(START_SEQUENCE_TAG) || line
317                         .startsWith(END_SEQUENCE_TAG)))) {
318                     // dump out last part of section
319                     section.add(new String[]{currKey, currVal.toString()});
320                     bufferedReader.reset();
321                     done = true;
322                 } else {
323                     Matcher m = sectp.matcher(line);
324                     if (m.matches()) {
325                         // new key
326                         if (currKey != null) {
327                             section.add(new String[]{currKey,
328                                 currVal.toString()});
329                         }
330                         // key = group(2) or group(4) or group(6) - whichever is
331                         // not null
332                         currKey = m.group(2) == null ? (m.group(4) == null ? m
333                                 .group(6) : m.group(4)) : m.group(2);
334                         currVal = new StringBuffer();
335 			// val = group(3) if group(2) not null, group(5) if
336                         // group(4) not null, "" otherwise, trimmed
337                         currVal.append((m.group(2) == null ? (m.group(4) == null ? ""
338                                 : m.group(5))
339                                 : m.group(3)).trim());
340                     } else {
341                         // concatted line or SEQ START/END line?
342                         if (line.startsWith(START_SEQUENCE_TAG)
343                                 || line.startsWith(END_SEQUENCE_TAG)) {
344                             currKey = line;
345                         } else {
346                             currVal.append("\n"); // newline in between lines -
347                             // can be removed later
348                             currVal.append(currKey.charAt(0) == '/' ? line
349                                     .substring(21) : line.substring(12));
350                         }
351                     }
352                 }
353             }
354         } catch (IOException e) {
355             throw new ParserException(e.getMessage());
356         } catch (RuntimeException e) {
357             throw new ParserException(e.getMessage());
358         }
359         return section;
360     }

362     @Override
363     public String getSequence(BufferedReader bufferedReader, int sequenceLength) throws IOException {
364         featureCollection = new HashMap<String, ArrayList<AbstractFeature>>();
365         mapDB = new LinkedHashMap<String, ArrayList<DBReferenceInfo>>();
366         headerParser = new GenericGenbankHeaderParser<S, C>();
367 
368         parse(bufferedReader);
369 
370         return seqData;
371     }

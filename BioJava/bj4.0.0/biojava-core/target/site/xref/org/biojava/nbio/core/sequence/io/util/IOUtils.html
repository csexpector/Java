44  	/**
45  	 * Closes any Object which implements the interface {@link Closeable} and
46  	 * sending any error to the logger but not forcing any explicit catching of
47  	 * stream errors.
48  	 *
49  	 * @param c The stream to close
50  	 */
51  	public static void close(Closeable c) {
52  		try {
53  			if (c != null) {
54  				c.close();
55  			}
56  		} catch (IOException e) {
57  			Logger log = Logger.getLogger(IOUtils.class.getName());
58  			log.log(Level.WARNING, "Cannot close down the given Closeable object", e);
59  		}
60  	}

62  	/**
63  	 * Moves the bytes from input to output using a 4KB byte array.
64  	 *
65  	 * @param input Input stream of bytes
66  	 * @param output Output stream of bytes
67  	 * @throws IOException If anything occurs in the case of the reads and writes
68  	 */
69  	public static void copy(InputStream input, OutputStream output)
70  			throws IOException {
71  		byte[] buffer = new byte[BUFFER];
72  		int n = 0;
73  		while (-1 != (n = input.read(buffer))) {
74  			output.write(buffer, 0, n);
75  		}
76  	}

78  	/**
79  	 * Takes in a reader and a processor, reads every line from the given
80  	 * file and then invokes the processor. What you do with the lines is
81  	 * dependent on your processor.
82  	 *
83  	 * The code will automatically close the given BufferedReader.
84  	 *
85  	 * @param br The reader to process
86  	 * @param processor The processor to invoke on all lines
87  	 * @throws ParserException Can throw this if we cannot parse the given reader
88  	 */
89  	public static void processReader(BufferedReader br, ReaderProcessor processor) throws ParserException {
90  		String line;
91  		try {
92  			while( (line = br.readLine()) != null ) {
93  				processor.process(line);
94  			}
95  		}
96  		catch(IOException e) {
97  			throw new ParserException("Could not read from the given BufferedReader");
98  		}
99  		finally {
100 			close(br);
101 		}
102 	}

104 	/**
105 	 * Returns the contents of a buffered reader as a list of strings
106 	 *
107 	 * @param br BufferedReader to read from; <strong>will be closed</strong>
108 	 * @return List of Strings
109 	 * @throws ParserException Can throw this if we cannot parse the given reader
110 	 */
111 	public static List<String> getList(BufferedReader br) throws ParserException {
112 		final List<String> list = new ArrayList<String>();
113 		processReader(br, new ReaderProcessor() {
114 			@Override
115 			public void process(String line) {
116 				list.add(line);
117 			}
118 		});
119 		return list;
120 	}

123 	 * Delegates to {@link #getList(BufferedReader)} by wrapping the InputStream
124 	 * in a valid reader. No encoding is mentioned so if you need anything
125 	 * more advanced then use the other version of this method.
126 	 *
127 	 * @param is InputStream which is a text file
128 	 * @return List of Strings representing the lines of the files
129 	 * @throws ParserException Can throw this if the file is not a file or we

131 	 */
132 	public static List<String> getList(InputStream is) throws ParserException {
133 		return getList(new BufferedReader(new InputStreamReader(is)));

136 	/**
137 	 * Delegates to {@link #getList(InputStream)} by wrapping the File
138 	 * in a valid stream. No encoding is mentioned so if you need anything
139 	 * more advanced then use the other version of this method. Since this
140 	 * uses {@link #openFile(File)} this code can support GZipped and plain
141 	 * files.
142 	 *
143 	 * @param file File which is a text file
144 	 * @return List of Strings representing the lines of the files
145 	 * @throws ParserException Can throw this if the file is not a file or we
146 	 * cannot parse it
147 	 */
148 	public static List<String> getList(File file) throws ParserException {
149 		return getList(openFile(file));
150 	}

152 	/**
153 	 * For a filename this code will check the extension of the file for a
154 	 * .gz extension. If it finds one then the InputStream given back
155 	 * is a {@link GZIPInputStream}. Otherwise we return a normal
156 	 * {@link FileInputStream}.
157 	 *
158 	 * @param file File which may or may not be GZipped
159 	 * @return The final stream
160 	 * @throws ParserException Can throw this if the file is not a file or we
161 	 * cannot open it for processing
162 	 */
163 	public static InputStream openFile(File file) throws ParserException {
164 		final InputStream is;
165 		if(!file.isFile()) {
166 			throw new ParserException("The file "+file+" is not a file.");
167 		}
168 		String name = file.getName();
169 		try {
170 			if(name.endsWith(".gz")) {
171 				is = new GZIPInputStream(new FileInputStream(file));
172 			}
173 			else {
174 				is = new FileInputStream(file);
175 			}
176 		}
177 		catch(IOException e) {
178 			throw new ParserException("Cannot open "+file+" for processing", e);
179 		}
180 		return is;
181 	}

194 	/**
195 	 * Calculates GCG checksum for entire list of sequences
196 	 *
197 	 * @param sequences list of sequences
198 	 * @return GCG checksum
199 	 */
200 	public static <S extends Sequence<C>, C extends Compound> int getGCGChecksum(List<S> sequences) {
201 		int check = 0;
202 		for (S as : sequences) {
203 			check += getGCGChecksum(as);
204 		}
205 		return check % 10000;
206 	}
207 
208 	/**
209 	 * Calculates GCG checksum for a given sequence
210 	 *
211 	 * @param sequence given sequence
212 	 * @return GCG checksum
213 	 */
214 	public static <S extends Sequence<C>, C extends Compound> int getGCGChecksum(S sequence) {
215 		String s = sequence.toString().toUpperCase();
216 		int count = 0, check = 0;
217 		for (int i = 0; i < s.length(); i++) {
218 			count++;
219 			check += count * s.charAt(i);
220 			if (count == 57) {
221 				count = 0;
222 			}
223 		}
224 		return check % 10000;
225 	}

227 	/**
228 	 * Assembles a GCG file header
229 	 *
230 	 * @param sequences list of sequences
231 	 * @return GCG header
232 	 */
233 	public static <S extends Sequence<C>, C extends Compound> String getGCGHeader(List<S> sequences) {
234 		StringBuilder header = new StringBuilder();
235 		S s1 = sequences.get(0);
236 		header.append(String.format("MSA from BioJava%n%n MSF: %d  Type: %s  Check: %d ..%n%n",
237 				s1.getLength(), getGCGType(s1.getCompoundSet()), getGCGChecksum(sequences)));
238 		String format = " Name: " + getIDFormat(sequences) + " Len: " + s1.getLength() + "  Check: %4d  Weight: 1.0%n";
239 		for (S as : sequences) {
240 			header.append(String.format(format, as.getAccession(), getGCGChecksum(as)));
241 			// TODO show weights in MSF header
242 		}
243 		header.append(String.format("%n//%n%n"));
244 		// TODO? convert gap characters to '.'
245 		return header.toString();
246 	}

249 	 * Determines GCG type
250 	 * @param cs compound set of sequences
251 	 * @return GCG type
252 	 */
253 	public static <C extends Compound> String getGCGType(CompoundSet<C> cs) {
254 		return (cs == DNACompoundSet.getDNACompoundSet() || cs == AmbiguityDNACompoundSet.getDNACompoundSet()) ? "D" :
255 			(cs == RNACompoundSet.getRNACompoundSet() || cs == AmbiguityRNACompoundSet.getRNACompoundSet()) ? "R" : "P";
256 	}

258 	/**
259 	 * Creates format String for accession IDs
260 	 *
261 	 * @param sequences list of sequences
262 	 * @return format String for accession IDs
263 	 */
264 	public static <S extends Sequence<C>, C extends Compound> String getIDFormat(List<S> sequences) {
265 		int length = 0;
266 		for (S as : sequences) {
267 			length = Math.max(length, (as.getAccession() == null) ? 0 : as.getAccession().toString().length());
268 		}
269 		return (length == 0) ? null : "%-" + (length + 1) + "s";
270 	}

289 	 *
290 	 * @param web true for HTML display
291 	 * @param c1 character in first sequence
292 	 * @param c2 character in second sequence
293 	 * @param similar true if c1 and c2 are considered similar compounds
294 	 * @return formatted String
295 	 */
296 	public static String getPDBConservation(boolean web, char c1, char c2, boolean similar) {
297 		return getPDBString(web, c1, c2, similar, "|", ".", " ", web ? "&nbsp;" : " ");
298 	}

300 	// helper method for getPDBCharacter and getPDBConservation
301 	private static String getPDBString(boolean web, char c1, char c2, boolean similar, String m, String sm, String dm,
302 			String qg) {
303 		if (c1 == c2)
304 			return web ? "<span class=\"m\">" + m + "</span>" : m;                             
305 		else if (similar)
306 			return web ? "<span class=\"sm\">" + sm + "</span>" : sm;
307 		else if (c1 == '-' || c2 == '-')
308 			return web ? "<span class=\"dm\">" + dm + "</span>" : dm;
309 		else
310 			return web ? "<span class=\"qg\">" + qg + "</span>" : qg;
311 	}

322 		s.append("          <b>Legend:</b>");
323 		s.append("          <span class=\"m\">Green</span> - identical residues |"); 
324 		s.append("          <span class=\"sm\">Pink</span> - similar residues | ");
325 		s.append("          <span class=\"qg\">Blue</span> - sequence mismatch |");
326 		s.append("          <span class=\"dm\">Brown</span> - insertion/deletion |");                  
327 		s.append("      </div>");
328 		s.append(String.format("%n"));
329 		return s.toString();
330 	}

332 	/**
333 	 * Prints {@code string} to {@code file}.
334 	 * @throws IOException If any I/O exception occurs while printing; this method does not catch any exceptions
335 	 */
336 	public static void print(String string, File file) throws IOException {
337 		PrintWriter out = null;
338 		try {
339 			out = new PrintWriter(new BufferedWriter(new FileWriter(file)));
340 			out.print(string);
341 			out.flush();
342 			out.close();
343 		} finally {
344 			if (out != null) out.close();
345 		}
346 	}

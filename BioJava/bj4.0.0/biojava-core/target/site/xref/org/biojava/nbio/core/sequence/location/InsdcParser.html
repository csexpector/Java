54          /**
55       * parse a location. if group(1) is null than the feature is on the positive
56       * strand, group(2) start position, group(3) end position.
57       */
58      protected static final Pattern singleLocationPattern = Pattern.compile("(?:[A-Z]([A-Za-z\\.0-9_]*?):)?(<?)(\\d+)(\\.{2}|\\^)?(>?)(\\d+)?(>?)?");
59      /**
60       * Decodes a split pattern. Split patterns are a composition of multiple
61       * locationsString qualified by actions: join(location,location, ...
62       * location): The indicated elements should be joined (placed end-to-end) to
63       * form one contiguous sequence. order(location,location, ... location): The
64       * elements can be found in the specified order (5' to 3' direction),
65       * nothing is implied about their reasonableness
66       * bond(location,location...location): Found in protein files. These
67       * generally are used to describe disulfide bonds.
68       * complement(location,location...location): consider locations in their
69       * complement versus
70       *
71       * takes in input a comma splitted location string. The split must be done
72       * for outer level commas group(1) is the qualifier group(2) is the location
73       * string to getFeatures. In case of complex splits it will contain the
74       * nested expression
75       *
76       * Not really sure that they are not declared obsolete but they are still in
77       * several files.
78       */
79      //protected static final Pattern genbankSplitPattern = Pattern.compile("^\\s?(join|order|bond|complement|)\\(?([^\\)]+)\\)?");
80      protected static final Pattern genbankSplitPattern = Pattern.compile("^\\s?(join|order|bond|complement|)\\(?(.+)\\)?");
81      /**
82       * designed to recursively split a location string in tokens. Valid tokens
83       * are those divided by coma that are not inside a bracket. I. e. split on
84       * the comma only if that comma has zero, or an even number of quotes ahead
85       * of it.
86       */
87      protected static final String locationSplitPattern = ",(?=([^\\(|\\)]+\\([^\\(|\\)]+\\))[^\\(|\\)]+)";
88      /**
89       * these variables are used to compute the global start and end of complex
90       * features
91       */
92      protected Integer featureGlobalStart, featureGlobalEnd;
93

107     private complexFeaturesAppendEnum complexFeaturesAppendMode = complexFeaturesAppendEnum.HIERARCHICAL;

127     /**
128      * Main method for parsing a location from a String instance
129      *
130      * @param locationString Represents a logical location
131      * @return The parsed location
132      * @throws ParserException thrown in the event of any error during parsing
133      */
134     public Location parse(String locationString) throws ParserException {
135         featureGlobalStart = Integer.MAX_VALUE;
136         featureGlobalEnd = 1;
137         
138         Location l;
139         List<Location> ll = parseLocationString(locationString, 1);
140         
141         if (ll.size() == 1) {
142             l = ll.get(0);
143         } else {
144             l = new SimpleLocation(
145                     featureGlobalStart, 
146                     featureGlobalEnd,
147                     Strand.UNDEFINED,
148                     ll);
149         }  
150         return l;
151     }

167     private List<Location> parseLocationString(String string, int versus) throws ParserException {
168         Matcher m;
169         List<Location> boundedLocationsCollection = new ArrayList<Location>();
170 
171         //String[] tokens = string.split(locationSplitPattern);
172         List<String> tokens = splitString(string);
173         for (String t : tokens) {
174             m = genbankSplitPattern.matcher(t);
175             if (!m.find()) {
176                 throw new ParserException("Cannot interpret split pattern " + t
177                         + "\nin location string:" + string);
178             }
179             String splitQualifier = m.group(1);
180             String splitString = m.group(2);
181 
182             if (!splitQualifier.isEmpty()) {
183                 //recursive case
184                 int localVersus = splitQualifier.equalsIgnoreCase("complement") ? -1 : 1;
185                 List<Location> subLocations = parseLocationString(splitString, versus * localVersus);
186 
187                 switch (complexFeaturesAppendMode) {
188                     case FLATTEN:
189                         boundedLocationsCollection.addAll(subLocations);
190                         break;
191                     case HIERARCHICAL:
192                         if (subLocations.size() == 1) {
193                             boundedLocationsCollection.addAll(subLocations);
194                         } else {
195                             Point min = Location.Tools.getMin(subLocations).getStart();
196                             Point max = Location.Tools.getMax(subLocations).getEnd();
197                             AbstractLocation motherLocation
198                                     = new SimpleLocation(
199                                             min, 
200                                             max
201                                     );
202 
203                             if (splitQualifier.equalsIgnoreCase("join")) {
204                                 motherLocation = new InsdcLocations.GroupLocation(subLocations);
205                             }
206                             if (splitQualifier.equalsIgnoreCase("order")) {
207                                 motherLocation = new InsdcLocations.OrderLocation(subLocations);
208                             }
209                             if (splitQualifier.equalsIgnoreCase("bond")) {
210                                 motherLocation = new InsdcLocations.BondLocation(subLocations);
211                             }
212                             motherLocation.setStrand(getGroupLocationStrand(subLocations));
213                             boundedLocationsCollection.add(motherLocation);
214                         }
215                     break;
216                 }
217             } else {
218                 //base case
219                 m = singleLocationPattern.matcher(splitString);
220                 if (!m.find()) {
221                     throw new ParserException("Cannot interpret location pattern " + splitString
222                             + "\nin location string:" + string);
223                 }
224 
225                 String accession = m.group(1);
226                 Strand s = versus == 1 ? Strand.POSITIVE : Strand.NEGATIVE;
227                 int start = new Integer(m.group(3));
228                 int end = m.group(6) == null ? start : new Integer(m.group(6));
229 
230                 if (featureGlobalStart > start) {
231                     featureGlobalStart = start;
232                 }
233                 if (featureGlobalEnd < end) {
234                     featureGlobalEnd = end;
235                 }
236 
237                 AbstractLocation l = new SimpleLocation(
238                         start,
239                         end,
240                         s
241                 );
242                 
243                 if(m.group(4) != null && m.group(4).equals("^")) l.setBetweenCompounds(true);
244 
245                 if (m.group(2).equals("<")) {
246                     l.setPartialOn5prime(true);
247                 }
248                 if (m.group(5) != null && (m.group(5).equals(">") || m.group(7).equals(">"))) {
249                     l.setPartialOn3prime(true);
250                 }
251                 
252                 if (!(accession == null || "".equals(accession))) l.setAccession(new AccessionID(accession));
253 
254                 boundedLocationsCollection.add(l);
255 
256             }
257         }
258 
259         return boundedLocationsCollection;
260     }

263     private List<String> splitString(String input) {
264         List<String> result = new ArrayList<String>();
265         int start = 0;
266         int openedParenthesis = 0;
267         for (int current = 0; current < input.length(); current++) {
268             if (input.charAt(current) == '(') {
269                 openedParenthesis++;
270             }
271             if (input.charAt(current) == ')') {
272                 openedParenthesis--;
273             }
274             boolean atLastChar = (current == input.length() - 1);
275             if (atLastChar) {
276                 result.add(input.substring(start));
277             } else if (input.charAt(current) == ',' && openedParenthesis == 0) {
278                 result.add(input.substring(start, current));
279                 start = current + 1;
280             }
281         }
282         return result;
283     }

285     private Strand getGroupLocationStrand(List<Location> ll){
286         Strand returnStrand = null;
287         
288         for (Location l: ll) {
289             if (returnStrand == null) returnStrand = l.getStrand();
290             if (returnStrand != l.getStrand()) return Strand.UNDEFINED;
291         }
292         return returnStrand;
293     }

304             "complement(join(complement(2000..4000),complement(70..80)),bond(34,35),join(56,80),complement(45..56))",
305             
306         };
307         InsdcParser p = new InsdcParser();
308         p.setComplexFeaturesAppendMode(complexFeaturesAppendEnum.HIERARCHICAL);
309         
310         for (String s: testStrings){
311             Location l = p.parse(s);
312             System.out.println(l.toString());
313         }

44  	protected static final int MAX_WIDTH = 80;
45  	protected static final int QUALIFIER_INDENT = 21;
46  	protected static final String QUALIFIER_INDENT_STR = "                     ";
47  	protected static final String QUALIFIER_INDENT_TMP = "     %s                ";
48  	private static final String lineSep = "%n";

50  	/**
51  	 * Format a feature qualifier using the MAX_WIDTH (default 80)
52  	 * @param key
53  	 * @param value
54  	 * @param quote
55  	 */
56  	private String _write_feature_qualifier(String key, String value, boolean quote) {
57  		String line = "";
58  		if(null == value) {
59  			line = QUALIFIER_INDENT_STR + "/" + key + lineSep;
60  			return line;
61  		}
62  		if(quote) {  // quote should be true for numerics
63  			line = QUALIFIER_INDENT_STR + "/" + key + "=\"" + value + "\"";
64  		} else {
65  			line = QUALIFIER_INDENT_STR + "/" + key + "=" + value;			
66  		}
67  		if(line.length() <= MAX_WIDTH) {
68  			return line + lineSep;
69  		}
70  		String goodlines = "";
71  		while(!"".equals(line.replaceAll("^\\s+", ""))) {
72  			if(line.length() <= MAX_WIDTH) {
73  				goodlines += line + lineSep;
74  				break;
75  			}
76  			//Insert line break...
77  			int index;
78  			for(index = Math.min(line.length()-1, MAX_WIDTH); index > QUALIFIER_INDENT ; index--) {
79  				if(' ' == line.charAt(index)) {
80  					break;
81  				}
82  			}
83  			if(' ' != line.charAt(index)) {
84  				//no nice place to break...
85  				index = MAX_WIDTH;
86  			}
87  			assert index <= MAX_WIDTH;
88  			goodlines += line.substring(0,index) + lineSep;
89  			line = QUALIFIER_INDENT_STR + line.substring(index).replaceAll("^\\s+", "");
90  		}
91  		return goodlines;
92  	}
93  	/**
94  	 * Split a feature location into lines (break at commas).
95  	 * @param location
96  	 */
97  	private String _wrap_location(String location) {
98  		int length = MAX_WIDTH - QUALIFIER_INDENT;
99  		if(location.length() <= length) {
100 			return location;
101 		}
102 		int index = location.substring(length).lastIndexOf(",");
103 		if(-1 == index) {
104 			//No good place to split (!)
105 			return location;
106 		}
107 		return location.substring(0,index+1) + lineSep + QUALIFIER_INDENT_STR + _wrap_location(location.substring(index+1));
108 	}
109 	/**
110 	 * Write a single SeqFeature object to features table.
111 	 * @param feature
112 	 * @param record_length
113 	 */
114 	protected String _write_feature(FeatureInterface<AbstractSequence<C>, C> feature, int record_length) {
115 		String location = _insdc_feature_location_string(feature, record_length);
116 		String f_type = feature.getType().replace(" ", "_");
117 		StringBuilder sb = new StringBuilder();
118 		Formatter formatter = new Formatter(sb,Locale.US);
119 		formatter.format(QUALIFIER_INDENT_TMP, f_type);
120 		String line = formatter.toString().substring(0, QUALIFIER_INDENT) + _wrap_location(location) + lineSep;
121 		formatter.close();
122 		
123 		//Now the qualifiers...
124 		for(Qualifier  q : feature.getQualifiers().values()) {
125 			line += _write_feature_qualifier(q.getName(), q.getValue(), q.needsQuotes());
126 		}
127 		return line;
128 		/*
129         self.handle.write(line)
130         #Now the qualifiers...
131         for key, values in feature.qualifiers.items():
132             if isinstance(values, list) or isinstance(values, tuple):
133                 for value in values:
134                     self._write_feature_qualifier(key, value)
135             elif values:
136                 #String, int, etc
137                 self._write_feature_qualifier(key, values)
138             else:
139                 #e.g. a /psuedo entry
140                 self._write_feature_qualifier(key)
141 		 */
142 	}
143 	/**
144 	 * Build a GenBank/EMBL location string from a SeqFeature (PRIVATE).
145 
146     There is a choice of how to show joins on the reverse complement strand,
147     GenBank used "complement(join(1,10),(20,100))" while EMBL used to use
148     "join(complement(20,100),complement(1,10))" instead (but appears to have
149     now adopted the GenBank convention). Notice that the order of the entries
150     is reversed! This function therefore uses the first form. In this situation
151     we expect the parent feature and the two children to all be marked as
152     strand == -1, and in the order 0:10 then 19:100.
153 
154     Also need to consider dual-strand examples like these from the Arabidopsis
155     thaliana chloroplast NC_000932: join(complement(69611..69724),139856..140650)
156     gene ArthCp047, GeneID:844801 or its CDS (protein NP_051038.1 GI:7525057)
157     which is further complicated by a splice:
158     join(complement(69611..69724),139856..140087,140625..140650)
159 
160     For mixed this mixed strand feature, the parent SeqFeature should have
161     no strand (either 0 or None) while the child features should have either
162     strand +1 or -1 as appropriate, and be listed in the order given here.
163 	 * @param feature
164 	 * @param record_length
165 	 */
166 	private String _insdc_feature_location_string(FeatureInterface<AbstractSequence<C>, C> feature, int record_length) {
167 		if(feature.getChildrenFeatures().isEmpty()) {
168 	        //Non-recursive.
169 			String location = _insdc_location_string_ignoring_strand_and_subfeatures(feature.getLocations(), record_length);
170 			if(feature.getLocations().getStrand() == Strand.NEGATIVE) {
171 				StringBuilder sb = new StringBuilder();
172 				Formatter formatter = new Formatter(sb,Locale.US);
173 				formatter.format("complement(%s)", location);
174 				String output = formatter.toString();
175 				formatter.close();
176 				location = output;
177 			}
178 			return location;
179 		}
180 	    // As noted above, treat reverse complement strand features carefully:
181 		if(feature.getLocations().getStrand() == Strand.NEGATIVE) {
182 			for(FeatureInterface<?, ?> f  : feature.getChildrenFeatures()) {
183 				if(f.getLocations().getStrand() != Strand.NEGATIVE) {
184 					StringBuilder sb = new StringBuilder();
185 					Formatter formatter = new Formatter(sb,Locale.US);
186 					formatter.format("Inconsistent strands: %r for parent, %r for child", feature.getLocations().getStrand(), f.getLocations().getStrand());
187 					String output = formatter.toString();
188 					formatter.close();
189 					throw new RuntimeException(output);
190 				}
191 			}
192 			StringBuilder sb = new StringBuilder();
193 			Formatter formatter = new Formatter(sb,Locale.US);
194 			ArrayList<String> locations = new ArrayList<String>();
195 			for(FeatureInterface<AbstractSequence<C>, C> f  : feature.getChildrenFeatures()) {
196 				locations.add(_insdc_location_string_ignoring_strand_and_subfeatures(f.getLocations(), record_length));
197 			}
198 			String location = StringManipulationHelper.join(locations, ",");
199 			formatter.format("complement(%s(%s))", /*feature.location_operator*/ "join", location);
200 			String output = formatter.toString();
201 			formatter.close();
202 			return output;
203 		}
204 	    //This covers typical forward strand features, and also an evil mixed strand:
205 		StringBuilder sb = new StringBuilder();
206 		Formatter formatter = new Formatter(sb,Locale.US);
207 		ArrayList<String> locations = new ArrayList<String>();
208 		for(FeatureInterface<AbstractSequence<C>, C> f  : feature.getChildrenFeatures()) {
209 			locations.add(_insdc_location_string_ignoring_strand_and_subfeatures(f.getLocations(), record_length));
210 		}
211 		String location =  StringManipulationHelper.join(locations, ",");
212 		formatter.format("%s(%s)", /*feature.location_operator*/ "join", location);
213 		String output = formatter.toString();
214 		formatter.close();
215 		return output;
216 	}

218 	private String _insdc_location_string_ignoring_strand_and_subfeatures(
219 			//SequenceLocation<AbstractSequence<C>, C> sequenceLocation,
220                         org.biojava.nbio.core.sequence.location.template.AbstractLocation sequenceLocation,
221 			int record_length) {
222 	/*
223 	if location.ref:
224         ref = "%s:" % location.ref
225     else:
226         ref = ""
227     assert not location.ref_db
228     */
229 		String ref = "";
230 		if(!sequenceLocation.getStart().isUncertain() && !sequenceLocation.getEnd().isUncertain() && sequenceLocation.getStart() == sequenceLocation.getEnd()) {
231 	        //Special case, for 12:12 return 12^13
232 	        //(a zero length slice, meaning the point between two letters)
233 			if(sequenceLocation.getEnd().getPosition() == record_length) {
234 	            //Very special case, for a between position at the end of a
235 	            //sequence (used on some circular genomes, Bug 3098) we have
236 	            //N:N so return N^1
237 				StringBuilder sb = new StringBuilder();
238 				Formatter formatter = new Formatter(sb,Locale.US);
239 				formatter.format("%s%d^1", ref, record_length);
240 				String output = formatter.toString();
241 				formatter.close();
242 				return output;
243 			} else {
244 				StringBuilder sb = new StringBuilder();
245 				Formatter formatter = new Formatter(sb,Locale.US);
246 				formatter.format("%s%d^%d", ref, sequenceLocation.getStart().getPosition(), sequenceLocation.getEnd().getPosition());
247 				String output = formatter.toString();
248 				formatter.close();
249 				return output;				
250 			}
251 		}
252 		if(!sequenceLocation.getStart().isUncertain() && !sequenceLocation.getEnd().isUncertain() && sequenceLocation.getStart().getPosition() + 1 == sequenceLocation.getEnd().getPosition()) {
253 	        //Special case, for 11:12 return 12 rather than 12..12
254 	        //(a length one slice, meaning a single letter)
255 			StringBuilder sb = new StringBuilder();
256 			Formatter formatter = new Formatter(sb,Locale.US);
257 			formatter.format("%s%d", ref, sequenceLocation.getEnd().getPosition());
258 			String output = formatter.toString();
259 			formatter.close();
260 			return output;							
261 		} else if(sequenceLocation.getStart().isUnknown() || sequenceLocation.getEnd().isUnknown()) {
262 			//Special case for features from SwissProt/UniProt files
263 			if(sequenceLocation.getStart().isUnknown() && sequenceLocation.getEnd().isUnknown()) {
264 				throw new RuntimeException("Feature with unknown location");
265 			} else if(sequenceLocation.getStart().isUnknown()) {
266 				//Treat the unknown start position as a BeforePosition
267 				StringBuilder sb = new StringBuilder();
268 				Formatter formatter = new Formatter(sb,Locale.US);
269 				formatter.format("%s<%d..%s", ref, sequenceLocation.getEnd().getPosition(), _insdc_feature_position_string(sequenceLocation.getEnd()));
270 				String output = formatter.toString();
271 				formatter.close();
272 				return output;				
273 			} else {
274 				//Treat the unknown start position as an AfterPosition
275 				StringBuilder sb = new StringBuilder();
276 				Formatter formatter = new Formatter(sb,Locale.US);
277 				formatter.format("%s%s..>%d", ref, _insdc_feature_position_string(sequenceLocation.getStart()), sequenceLocation.getStart().getPosition());
278 				String output = formatter.toString();
279 				formatter.close();
280 				return output;								
281 			}
282 		} else {
283 			//Typical case, e.g. 12..15 gets mapped to 11:15
284 			return ref + _insdc_feature_position_string(sequenceLocation.getStart(), 1) + ".." + _insdc_feature_position_string(sequenceLocation.getEnd());
285 		}
286 	}

329 	/**
330 	 * Returns a list of strings.
331      *   
332      *   Any single words which are too long get returned as a whole line
333      *   (e.g. URLs) without an exception or warning.
334 	 * @param text
335 	 * @param max_len
336 	 */
337 	protected ArrayList<String> _split_multi_line(String text, int max_len) {
338 		// TODO Auto-generated method stub
339 		ArrayList<String> output = new ArrayList<String>();
340 		text = text.trim();
341 		if(text.length() <= max_len) {
342 			output.add(text);
343 			return output;
344 		}
345 		
346 		ArrayList<String> words = new ArrayList<String>();
347 		Collections.addAll(words, text.split("\\s+"));
348 		while(!words.isEmpty()) {
349 			text = words.remove(0);
350 			while(!words.isEmpty() && (text.length() + 1 + words.get(0).length()) <= max_len) {
351 				text += " " + words.remove(0);
352 				text = text.trim();
353 			}
354 			output.add(text);
355 		}
356 		assert words.isEmpty();
357 		return output;
358 	}

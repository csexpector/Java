24  package org.biojava.nbio.genome;
25  
26  import junit.framework.TestCase;
27  import org.biojava.nbio.genome.parsers.gff.Feature;
28  import org.biojava.nbio.genome.parsers.gff.FeatureList;
29  import org.biojava.nbio.genome.parsers.gff.Location;
30  import org.junit.Test;
31  
32  /**
33   * @author mckeee1
34   *
35   */
36  public class FeatureListTest extends TestCase {
37      @Test
38      public void testAddIndex() throws Exception
39      {
40          FeatureList fl = new FeatureList();
41          fl.add(new Feature("seqname", "source", "type", new Location(1, 2), (double)0, 0, "gene_id \"gene_id_1\"; transcript_id \"transcript_id_1\";"));
42          fl.addIndex("transcript_id");
43          assertEquals(1, fl.selectByAttribute("transcript_id").size());
44  
45      
46          FeatureList f2 = new FeatureList();
47          f2.addIndex("transcript_id");
48          f2.add(new Feature("seqname", "source", "type", new Location(1, 2), (double)0, 0, "gene_id \"gene_id_1\"; transcript_id \"transcript_id_1\";"));
49          assertEquals(1, f2.selectByAttribute("transcript_id").size());
50      }
51  }

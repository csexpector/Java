24  package org.biojava.nbio.genome.parsers.genename;
25  
26  import java.io.Serializable;
27  
28  /** A simple bean that contains gene name information as available from www.genenames.org
29   * 
30   * @author Andreas Prlic
31   *
32   */
33  public class GeneName implements Serializable, Comparable<GeneName>{
34  	//[HGNC ID, Approved Symbol, Approved Name, Status, Previous Symbols, Previous Names, Synonyms, 
35  	// Chromosome, Accession Numbers, RefSeq IDs,Uniprot]
36  	
37  	/**
38  	 * 
39  	 */
40  	private static final long serialVersionUID = -7163977639324764020L;
41  	String hgncId;
42  	String approvedSymbol;
43  	String approvedName;
44  	String status;
45  	String previousSymbols;
46  	String previousNames;
47  	String synonyms;
48  	String chromosome;
49  	String accessionNr;
50  	String refseqIds;
51  	String uniprot;
52  	String omimId;
53  	String ensemblGeneId;
54  	
55  	public String getHgncId() {
56  		return hgncId;
57  	}
58  	public void setHgncId(String hgncId) {
59  		this.hgncId = hgncId;
60  	}
61  	public String getApprovedSymbol() {
62  		return approvedSymbol;
63  	}
64  	public void setApprovedSymbol(String approvedSymbol) {
65  		this.approvedSymbol = approvedSymbol;
66  	}
67  	public String getApprovedName() {
68  		return approvedName;
69  	}
70  	public void setApprovedName(String approvedName) {
71  		this.approvedName = approvedName;
72  	}
73  	public String getStatus() {
74  		return status;
75  	}
76  	public void setStatus(String status) {
77  		this.status = status;
78  	}
79  	public String getPreviousSymbols() {
80  		return previousSymbols;
81  	}
82  	public void setPreviousSymbols(String previousSymbols) {
83  		this.previousSymbols = previousSymbols;
84  	}
85  	public String getPreviousNames() {
86  		return previousNames;
87  	}
88  	public void setPreviousNames(String previousNames) {
89  		this.previousNames = previousNames;
90  	}
91  	public String getSynonyms() {
92  		return synonyms;
93  	}
94  	public void setSynonyms(String synonyms) {
95  		this.synonyms = synonyms;
96  	}
97  	public String getChromosome() {
98  		return chromosome;
99  	}
100 	public void setChromosome(String chromosome) {
101 		this.chromosome = chromosome;
102 	}
103 	public String getAccessionNr() {
104 		return accessionNr;
105 	}
106 	public void setAccessionNr(String accessionNr) {
107 		this.accessionNr = accessionNr;
108 	}
109 	public String getRefseqIds() {
110 		return refseqIds;
111 	}
112 	public void setRefseqIds(String refseqIds) {
113 		this.refseqIds = refseqIds;
114 	}
115 	public String getUniprot() {
116 		return uniprot;
117 	}
118 	public void setUniprot(String uniprot) {
119 		this.uniprot = uniprot;
120 	}
121 	
122 	public String getEnsemblGeneId() {
123 		return ensemblGeneId;
124 	}
125 	public void setEnsemblGeneId(String ensemblGeneId) {
126 		this.ensemblGeneId = ensemblGeneId;
127 	}
128 	public String getOmimId() {
129 		return omimId;
130 	}
131 	public void setOmimId(String omimId) {
132 		this.omimId = omimId;
133 	}
134 	@Override
135 	public int compareTo(GeneName o) {
136 		return hgncId.compareTo(o.getHgncId());
137 	}
138 	
139 	public boolean equals(GeneName o){
140 		return hgncId.equals(o.getHgncId());
141 	}
142 	@Override
143 	public String toString() {
144 		return "GeneName [hgncId=" + hgncId + ", approvedSymbol="
145 				+ approvedSymbol + ", approvedName=" + approvedName
146 				+ ", status=" + status + ", previousSymbols=" + previousSymbols
147 				+ ", previousNames=" + previousNames + ", synonyms=" + synonyms
148 				+ ", chromosome=" + chromosome + ", accessionNr=" + accessionNr
149 				+ ", refseqIds=" + refseqIds + ", uniprot=" + uniprot
150 				+ ", omimId=" + omimId + ", ensemblGeneId=" + ensemblGeneId
151 				+ "]";
152 	}
153 	
154 }

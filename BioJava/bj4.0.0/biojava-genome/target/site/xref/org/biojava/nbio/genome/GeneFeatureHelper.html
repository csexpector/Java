48      static public LinkedHashMap<String, ChromosomeSequence> loadFastaAddGeneFeaturesFromUpperCaseExonFastaFile(File fastaSequenceFile, File uppercaseFastaFile, boolean throwExceptionGeneNotFound) throws Exception {
49          LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList = new LinkedHashMap<String, ChromosomeSequence>();
50          LinkedHashMap<String, DNASequence> dnaSequenceList = FastaReaderHelper.readFastaDNASequence(fastaSequenceFile);
51          for (String accession : dnaSequenceList.keySet()) {
52              DNASequence contigSequence = dnaSequenceList.get(accession);
53              ChromosomeSequence chromsomeSequence = new ChromosomeSequence(contigSequence.getSequenceAsString());
54              chromsomeSequence.setAccession(contigSequence.getAccession());
55              chromosomeSequenceList.put(accession, chromsomeSequence);
56          }
57  
58  
59          LinkedHashMap<String, DNASequence> geneSequenceList = FastaReaderHelper.readFastaDNASequence(uppercaseFastaFile);
60          for (DNASequence dnaSequence : geneSequenceList.values()) {
61              String geneSequence = dnaSequence.getSequenceAsString();
62              String lcGeneSequence = geneSequence.toLowerCase();
63              String reverseGeneSequence = dnaSequence.getReverse().getSequenceAsString();
64              String lcReverseGeneSequence = reverseGeneSequence.toLowerCase();
65              Integer bioStart = null;
66              Integer bioEnd = null;
67              Strand strand = Strand.POSITIVE;
68              boolean geneFound = false;
69              String accession = "";
70              DNASequence contigDNASequence = null;
71              for (String id : dnaSequenceList.keySet()) {
72                  accession = id;
73                  contigDNASequence = dnaSequenceList.get(id);
74                  String contigSequence = contigDNASequence.getSequenceAsString().toLowerCase();
75                  bioStart = contigSequence.indexOf(lcGeneSequence);
76                  if (bioStart != -1) {
77                      bioStart = bioStart + 1;
78                      bioEnd = bioStart + geneSequence.length() - 1;
79                      geneFound = true;
80                      break;
81                  } else {
82                      bioStart = contigSequence.indexOf(lcReverseGeneSequence);
83                      if (bioStart != -1) {
84                          bioStart = bioStart + 1;
85                          bioEnd = bioStart - geneSequence.length() - 1;
86                          strand = Strand.NEGATIVE;
87                          geneFound = true;
88                          break;
89                      }
90                  }
91              }
92  
93              if (geneFound) {
94                  logger.info("Gene {} found at {} {} {} {}",
95                  		dnaSequence.getAccession().toString(), contigDNASequence.getAccession().toString(), bioStart, bioEnd, strand);
96                  ChromosomeSequence chromosomeSequence = chromosomeSequenceList.get(accession);
97  
98                  ArrayList<Integer> exonBoundries = new ArrayList<Integer>();
99  
100                 //look for transitions from lowercase to upper case
101                 for (int i = 0; i < geneSequence.length(); i++) {
102                     if (i == 0 && Character.isUpperCase(geneSequence.charAt(i))) {
103                         exonBoundries.add(i);
104                     } else if (i == geneSequence.length() - 1) {
105                         exonBoundries.add(i);
106                     } else if (Character.isUpperCase(geneSequence.charAt(i)) && Character.isLowerCase(geneSequence.charAt(i - 1))) {
107                         exonBoundries.add(i);
108                     } else if (Character.isUpperCase(geneSequence.charAt(i)) && Character.isLowerCase(geneSequence.charAt(i + 1))) {
109                         exonBoundries.add(i);
110                     }
111                 }
112                 if (strand == Strand.NEGATIVE) {
113                     Collections.reverse(exonBoundries);
114                 }
115 
116 
117                 String geneaccession = dnaSequence.getAccession().getID();
118                 String note = geneaccession;
119                 String[] values = geneaccession.split(" ");
120                 geneaccession = values[0];
121 
122 
123 
124                 GeneSequence geneSeq = chromosomeSequence.addGene(new AccessionID(geneaccession), bioStart, bioEnd, strand);
125                 geneSeq.addNote(note);
126                 geneSeq.setSource(uppercaseFastaFile.getName());
127                 //String transcriptName = geneaccession + "-transcript";
128                 //TranscriptSequence transcriptSequence = geneSeq.addTranscript(new AccessionID(transcriptName), bioStart, bioEnd);
129 
130                 int runningFrameLength = 0;
131                 for (int i = 0; i < exonBoundries.size() - 1; i = i + 2) {
132                     int cdsBioStart = exonBoundries.get(i) + bioStart;
133                     int cdsBioEnd = exonBoundries.get(i + 1) + bioStart;
134                     runningFrameLength = runningFrameLength + Math.abs(cdsBioEnd - cdsBioStart) + 1;
135                     //String cdsName = transcriptName + "-cds-" + cdsBioStart + "-" + cdsBioEnd;
136 
137                     //AccessionID cdsAccessionID = new AccessionID(cdsName);
138                     //ExonSequence exonSequence = geneSeq.addExon(cdsAccessionID, cdsBioStart, cdsBioEnd);
139                     int remainder = runningFrameLength % 3;
140                     //int frame = 0;
141                     if (remainder == 1) {
142                         //frame = 2; // borrow 2 from next CDS region
143                     } else if (remainder == 2) {
144                         //frame = 1;
145                     }
146                     //CDSSequence cdsSequence = transcriptSequence.addCDS(cdsAccessionID, cdsBioStart, cdsBioEnd, frame);
147                 }
148 
149 
150             } else {
151                 if (throwExceptionGeneNotFound) {
152                     throw new Exception(dnaSequence.getAccession().toString() + " not found");
153                 }
154                 logger.info("Gene not found {}", dnaSequence.getAccession().toString());
155             }
156 
157         }
158         return chromosomeSequenceList;
159     }

161     /**
162      * Output a gff3 feature file that will give the length of each scaffold/chromosome in the fasta file.
163      * Used for gbrowse so it knows length.
164      * @param fastaSequenceFile
165      * @param gffFile
166      * @throws Exception
167      */
168     static public void outputFastaSequenceLengthGFF3(File fastaSequenceFile, File gffFile) throws Exception {
169         LinkedHashMap<String, DNASequence> dnaSequenceList = FastaReaderHelper.readFastaDNASequence(fastaSequenceFile);
170         String fileName = fastaSequenceFile.getName();
171         FileWriter fw = new FileWriter(gffFile);
172         String newLine = System.getProperty("line.separator");
173         fw.write("##gff-version 3" + newLine);
174         for (DNASequence dnaSequence : dnaSequenceList.values()) {
175             String gff3line = dnaSequence.getAccession().getID() + "\t" + fileName + "\t" + "contig" + "\t" + "1" + "\t" + dnaSequence.getBioEnd() + "\t.\t.\t.\tName=" + dnaSequence.getAccession().getID() + newLine;
176             fw.write(gff3line);
177         }
178         fw.close();
179     }

198      * Load GFF2 feature file generated from the geneid prediction algorithm and map features onto the chromosome sequences
199      *
200      * @param chromosomeSequenceList
201      * @param listGenes
202      * @throws Exception
203      */
204     static public void addGeneIDGFF2GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
205         Collection<String> geneIds = listGenes.attributeValues("gene_id");
206         for (String geneid : geneIds) {
207             FeatureList gene = listGenes.selectByAttribute("gene_id", geneid);
208             FeatureI geneFeature = gene.get(0);
209             ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
210             geneid = geneid.replaceAll("_", ".G");
211             AccessionID geneAccessionID = new AccessionID(geneid);
212             GeneSequence geneSequence = null;
213             Collection<String> transcriptids = gene.attributeValues("gene_id");
214             for (String transcriptid : transcriptids) {
215                 // get all the individual features (exons, CDS regions, etc.) of this gene
216                 FeatureList transcriptFeature = listGenes.selectByAttribute("gene_id", transcriptid);
217                 transcriptid = transcriptid.replaceAll("_", ".G");
218 
219 
220 
221 
222                 //      String seqName = feature.seqname();
223                 //FeatureI startCodon = null;
224                 //FeatureI stopCodon = null;
225                 Integer startCodonBegin = null;
226                 Integer stopCodonEnd = null;
227                 //String startCodonName = "";
228                 //String stopCodonName = "";
229 
230 
231                 // now select only the coding regions of this gene
232                 FeatureList firstFeatures = transcriptFeature.selectByType("First");
233                 FeatureList terminalFeatures = transcriptFeature.selectByType("Terminal");
234                 FeatureList internalFeatures = transcriptFeature.selectByType("Internal");
235                 FeatureList singleFeatures = transcriptFeature.selectByType("Single");
236                 FeatureList cdsFeatures = new FeatureList();
237                 cdsFeatures.add(firstFeatures);
238                 cdsFeatures.add(terminalFeatures);
239                 cdsFeatures.add(internalFeatures);
240                 cdsFeatures.add(singleFeatures);
241                 // sort them
242                 cdsFeatures = cdsFeatures.sortByStart();
243                 Strand strand = Strand.POSITIVE;
244                 FeatureI feature = cdsFeatures.get(0);
245                 if (feature.location().isNegative()) {
246                     strand = Strand.NEGATIVE;
247                 }
248                 if (startCodonBegin == null) {
249                     FeatureI firstFeature = cdsFeatures.get(0);
250                     if (strand == Strand.NEGATIVE) {
251                         startCodonBegin = firstFeature.location().bioEnd();
252                     } else {
253                         startCodonBegin = firstFeature.location().bioStart();
254                     }
255                 }
256 
257                 if (stopCodonEnd == null) {
258 
259                     FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
260                     if (strand == Strand.NEGATIVE) {
261                         stopCodonEnd = lastFeature.location().bioStart();
262                     } else {
263                         stopCodonEnd = lastFeature.location().bioEnd();
264                     }
265                 }
266                 //for gtf ordering can be strand based so first is last and last is first
267                 if (startCodonBegin > stopCodonEnd) {
268                     int temp = startCodonBegin;
269                     startCodonBegin = stopCodonEnd;
270                     stopCodonEnd = temp;
271                 }
272 
273                 AccessionID transcriptAccessionID = new AccessionID(transcriptid);
274                 if (geneSequence == null) {
275                     geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
276                     geneSequence.setSource(((Feature) feature).source());
277                 } else {
278                     //if multiple transcripts for one gene make sure the gene is defined as the min and max start/end
279 
280                     if (startCodonBegin < geneSequence.getBioBegin()) {
281                         geneSequence.setBioBegin(startCodonBegin);
282                     }
283                     if (stopCodonEnd > geneSequence.getBioBegin()) {
284                         geneSequence.setBioEnd(stopCodonEnd);
285                     }
286 
287                 }
288                 TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
289                 /*
290                 if (startCodon != null) {
291                     if (startCodonName == null || startCodonName.length() == 0) {
292                         startCodonName = transcriptid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
293                     }
294                     transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
295                 }
296                 if (stopCodon != null) {
297                     if (stopCodonName == null || stopCodonName.length() == 0) {
298                         stopCodonName = transcriptid + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
299                     }
300                     transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
301                 }
302                 */
303 
304                 for (FeatureI cdsFeature : cdsFeatures) {
305                     Feature cds = (Feature) cdsFeature;
306                     String cdsName = cds.getAttribute("transcript_name");
307                     if (cdsName == null || cdsName.length() == 0) {
308                         cdsName = transcriptid + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
309                     }
310                     AccessionID cdsAccessionID = new AccessionID(cdsName);
311                     //ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
312                     CDSSequence cdsSequence = transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), cds.frame());
313                     cdsSequence.setSequenceScore(cds.score());
314                 }
315             }
316         }

320     static public LinkedHashMap<String, ChromosomeSequence> getChromosomeSequenceFromDNASequence(LinkedHashMap<String, DNASequence> dnaSequenceList) {
321         LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList = new LinkedHashMap<String, ChromosomeSequence>();
322         for (String key : dnaSequenceList.keySet()) {
323             DNASequence dnaSequence = dnaSequenceList.get(key);
324             ChromosomeSequence chromosomeSequence = new ChromosomeSequence(dnaSequence.getProxySequenceReader()); //we want the underlying sequence but don't need storage
325             chromosomeSequence.setAccession(dnaSequence.getAccession());
326             chromosomeSequenceList.put(key, chromosomeSequence);
327         }
328         return chromosomeSequenceList;
329     }

353      * @throws Exception
354      */
355     static public void addGmodGFF3GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
356 
357 
358         // key off mRNA as being a known feature that may or may not have a parent gene
359 
360 
361         FeatureList mRNAFeatures = listGenes.selectByType("mRNA");
362         LinkedHashMap<String,FeatureList> featureIDHashMap = FeatureHelper.buildFeatureAtrributeIndex("ID", listGenes);
363         LinkedHashMap<String,FeatureList> featureParentHashMap = FeatureHelper.buildFeatureAtrributeIndex("Parent", listGenes);
364 
365         for (FeatureI f : mRNAFeatures) {
366             String geneID;
367             String geneNote = null;
368             String geneSource = null;
369             String sequenceName = null;
370             ChromosomeSequence seq = null;
371             GeneSequence geneSequence = null;
372 
373             Feature mRNAFeature = (Feature) f;
374             String mRNAID = mRNAFeature.getAttribute("ID");
375             String mRNAsource = mRNAFeature.source();
376             String mRNANote = mRNAFeature.getAttribute("Note");
377             String mRNAParent = mRNAFeature.getAttribute("Parent");
378             if (mRNAParent != null && mRNAParent.length() > 0) {
379                // FeatureList geneFeatureList = listGenes.selectByAttribute("ID", mRNAParent);
380                 FeatureList geneFeatureList = featureIDHashMap.get(mRNAParent);
381                 Feature geneFeature = (Feature) geneFeatureList.get(0);
382                 geneID = geneFeature.getAttribute("ID");
383                 geneNote = geneFeature.getAttribute("Note");
384                 geneSource = ((Feature) geneFeature).source();
385                 sequenceName = geneFeature.seqname();
386 
387                 //
388             } else {
389                 //deal with cases where no parent gene is given
390                 geneID = mRNAID;
391                 geneSource = mRNAsource;
392                 sequenceName = mRNAFeature.seqname();
393             }
394 
395             seq = chromosomeSequenceList.get(sequenceName);
396 
397             AccessionID geneAccessionID = new AccessionID(geneID);
398 
399           //  FeatureList mRNAChildren = listGenes.selectByAttribute("Parent", mRNAID);
400             FeatureList mRNAChildren = featureParentHashMap.get(mRNAID);
401             FeatureList cdsFeatures = mRNAChildren.selectByType("CDS");
402             FeatureI feature = cdsFeatures.get(0);
403             Strand strand = Strand.POSITIVE;
404 
405             if (feature.location().isNegative()) {
406                 strand = Strand.NEGATIVE;
407             }
408             cdsFeatures = cdsFeatures.sortByStart();
409 
410 
411 
412 
413 
414 
415 
416             //String seqName = feature.seqname();
417             FeatureI startCodon = null;
418             FeatureI stopCodon = null;
419             Integer startCodonBegin = null;
420             Integer stopCodonEnd = null;
421             String startCodonName = "";
422             String stopCodonName = "";
423             FeatureList startCodonList = mRNAChildren.selectByType("five_prime_UTR");
424             if (startCodonList != null && startCodonList.size() > 0) {
425                 startCodon = startCodonList.get(0);
426                 if (strand == Strand.NEGATIVE) {
427                     startCodonBegin = startCodon.location().bioEnd();
428                 } else {
429                     startCodonBegin = startCodon.location().bioStart();
430                 }
431                 startCodonName = startCodon.getAttribute("ID");
432             }
433 
434             FeatureList stopCodonList = mRNAChildren.selectByType("three_prime_UTR");
435 
436             if (stopCodonList != null && stopCodonList.size() > 0) {
437                 stopCodon = stopCodonList.get(0);
438                 if (strand == Strand.NEGATIVE) {
439                     stopCodonEnd = stopCodon.location().bioStart();
440                 } else {
441                     stopCodonEnd = stopCodon.location().bioEnd();
442                 }
443                 stopCodonName = stopCodon.getAttribute("ID");
444 
445             }
446 
447 
448 
449 
450             if (startCodonBegin == null) {
451                 if (strand == Strand.NEGATIVE) {
452                     FeatureI firstFeature = cdsFeatures.get(0);
453 
454                     startCodonBegin = firstFeature.location().bioEnd();
455                 } else {
456                     FeatureI firstFeature = cdsFeatures.get(0);
457 
458                     startCodonBegin = firstFeature.location().bioStart();
459                 }
460             }
461 
462             if (stopCodonEnd == null) {
463                 if (strand == Strand.NEGATIVE) {
464                     FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
465                     stopCodonEnd = lastFeature.location().bioStart();
466                 } else {
467                     FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
468                     stopCodonEnd = lastFeature.location().bioEnd();
469                 }
470             }
471             //for gtf ordering can be strand based so first is last and last is first
472             if (startCodonBegin > stopCodonEnd) {
473                 int temp = startCodonBegin;
474                 startCodonBegin = stopCodonEnd;
475                 stopCodonEnd = temp;
476             }
477 
478 
479 
480             AccessionID transcriptAccessionID = new AccessionID(mRNAID);
481             geneSequence = seq.getGene(geneID);
482             if (geneSequence == null) {
483                 geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
484                 geneSequence.setSource(geneSource);
485                 if (geneNote != null && geneNote.length() > 0) {
486                     geneSequence.addNote(geneNote);
487                 }
488             } else {
489 
490                 if (startCodonBegin < geneSequence.getBioBegin()) {
491                     geneSequence.setBioBegin(startCodonBegin);
492                 }
493                 if (stopCodonEnd > geneSequence.getBioBegin()) {
494                     geneSequence.setBioEnd(stopCodonEnd);
495                 }
496 
497             }
498             TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
499             transcriptSequence.setSource(mRNAsource);
500             if (mRNANote != null && mRNANote.length() > 0) {
501                 transcriptSequence.addNote(mRNANote);
502 
503             }
504             if (startCodon != null) {
505                 if (startCodonName == null || startCodonName.length() == 0) {
506                     startCodonName = mRNAID + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
507                 }
508                 transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
509             }
510             if (stopCodon != null) {
511                 if (stopCodonName == null || stopCodonName.length() == 0) {
512                     stopCodonName = mRNAID + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
513                 }
514                 transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
515             }
516 
517             for (FeatureI cdsFeature : cdsFeatures) {
518                 Feature cds = (Feature) cdsFeature;
519                 String cdsNote = cdsFeature.getAttribute("Note");
520                 String cdsSource = cds.source();
521                 String cdsName = cds.getAttribute("ID");
522                 if (cdsName == null || cdsName.length() == 0) {
523                     cdsName = mRNAID + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
524                 }
525                 AccessionID cdsAccessionID = new AccessionID(cdsName);
526                 ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
527                 exonSequence.setSource(cdsSource);
528                 if (cdsNote != null && cdsNote.length() > 0) {
529                     exonSequence.addNote(cdsNote);
530                 }
531                 transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), cds.frame());
532             }
533             geneSequence.addIntronsUsingExons();
534 
535         }

547     static public void addGlimmerGFF3GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {

549         for (FeatureI f : mRNAFeatures) {
550             Feature mRNAFeature = (Feature) f;
551             String geneid = mRNAFeature.getAttribute("ID");
552             String source = mRNAFeature.source();
553 
554             FeatureList gene = listGenes.selectByAttribute("Parent", geneid);
555             FeatureI geneFeature = gene.get(0);
556             ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
557             AccessionID geneAccessionID = new AccessionID(geneid);
558             GeneSequence geneSequence = null;
559 
560             FeatureList cdsFeatures = gene.selectByType("CDS");
561             FeatureI feature = cdsFeatures.get(0);
562             Strand strand = Strand.POSITIVE;
563 
564             if (feature.location().isNegative()) {
565                 strand = Strand.NEGATIVE;
566             }
567             cdsFeatures = cdsFeatures.sortByStart();
568 
569 
570 
571 
572 
573 
574 
575             //String seqName = feature.seqname();
576             FeatureI startCodon = null;
577             FeatureI stopCodon = null;
578             Integer startCodonBegin = null;
579             Integer stopCodonEnd = null;
580             String startCodonName = "";

622                 if (strand == Strand.NEGATIVE) {
623                     FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);

627                     stopCodonEnd = lastFeature.location().bioEnd();

629             }

634                 stopCodonEnd = temp;

657                 if (startCodonName == null || startCodonName.length() == 0) {
658                     startCodonName = geneid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();

666                 transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());

678             }

692     static public void addGeneMarkGTFGeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
693         Collection<String> geneIds = listGenes.attributeValues("gene_id");
694         for (String geneid : geneIds) {
695             //       if(geneid.equals("45_g")){
696             //           int dummy =1;
697             //       }
698             FeatureList gene = listGenes.selectByAttribute("gene_id", geneid);
699             FeatureI geneFeature = gene.get(0);
700             ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
701             AccessionID geneAccessionID = new AccessionID(geneid);
702             GeneSequence geneSequence = null;
703             Collection<String> transcriptids = gene.attributeValues("transcript_id");
704             for (String transcriptid : transcriptids) {
705                 // get all the individual features (exons, CDS regions, etc.) of this gene
706 
707 
708                 FeatureList transcriptFeature = listGenes.selectByAttribute("transcript_id", transcriptid);
709                 // now select only the coding regions of this gene
710                 FeatureList cdsFeatures = transcriptFeature.selectByType("CDS");
711                 // sort them
712                 cdsFeatures = cdsFeatures.sortByStart();
713 
714                 FeatureI feature = cdsFeatures.get(0);
715                 Strand strand = Strand.POSITIVE;
716 
717                 if (feature.location().isNegative()) {
718                     strand = Strand.NEGATIVE;
719                 }
720 
721                 //String seqName = feature.seqname();
722                 FeatureI startCodon = null;
723                 FeatureI stopCodon = null;
724                 Integer startCodonBegin = null;
725                 Integer stopCodonEnd = null;
726                 String startCodonName = "";
727                 String stopCodonName = "";
728                 FeatureList startCodonList = transcriptFeature.selectByType("start_codon");
729                 if (startCodonList != null && startCodonList.size() > 0) {
730                     startCodon = startCodonList.get(0);
731                     if (strand == Strand.POSITIVE) {
732                         startCodonBegin = startCodon.location().bioStart();
733                     } else {
734                         startCodonBegin = startCodon.location().bioEnd();
735                     }
736                     startCodonName = startCodon.getAttribute("transcript_name");
737                 }
738 
739                 FeatureList stopCodonList = transcriptFeature.selectByType("stop_codon");
740 
741                 if (stopCodonList != null && stopCodonList.size() > 0) {
742                     stopCodon = stopCodonList.get(0);
743                     if (strand == Strand.POSITIVE) {
744                         stopCodonEnd = stopCodon.location().bioEnd();
745                     } else {
746                         stopCodonEnd = stopCodon.location().bioStart();
747                     }
748 
749                     stopCodonName = stopCodon.getAttribute("transcript_name");
750 
751                 }
752 
753 
754 
755 
756                 if (startCodonBegin == null) {
757                     if (strand == Strand.NEGATIVE) {
758                         FeatureI firstFeature = cdsFeatures.get(0);
759 
760                         startCodonBegin = firstFeature.location().bioEnd();
761                     } else {
762                         FeatureI firstFeature = cdsFeatures.get(0);
763 
764                         startCodonBegin = firstFeature.location().bioStart();
765                     }
766                 }
767 
768                 if (stopCodonEnd == null) {
769                     if (strand == Strand.NEGATIVE) {
770                         FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
771                         stopCodonEnd = lastFeature.location().bioStart();
772                     } else {
773                         FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
774                         stopCodonEnd = lastFeature.location().bioEnd();
775                     }
776                 }
777                 //for gtf ordering can be strand based so first is last and last is first
778                 if (startCodonBegin > stopCodonEnd) {
779                     int temp = startCodonBegin;
780                     startCodonBegin = stopCodonEnd;
781                     stopCodonEnd = temp;
782                 }
783 
784                 AccessionID transcriptAccessionID = new AccessionID(transcriptid);
785                 if (geneSequence == null) {
786                     geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
787                     geneSequence.setSource(((Feature) feature).source());
788                 } else {
789                     //if multiple transcripts for one gene make sure the gene is defined as the min and max start/end
790 
791                     if (startCodonBegin < geneSequence.getBioBegin()) {
792                         geneSequence.setBioBegin(startCodonBegin);
793                     }
794                     if (stopCodonEnd > geneSequence.getBioBegin()) {
795                         geneSequence.setBioEnd(stopCodonEnd);
796                     }
797 
798                 }
799                 TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
800                 if (startCodon != null) {
801                     if (startCodonName == null || startCodonName.length() == 0) {
802                         startCodonName = transcriptid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
803                     }
804                     transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
805                 }
806                 if (stopCodon != null) {
807                     if (stopCodonName == null || stopCodonName.length() == 0) {
808                         stopCodonName = transcriptid + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
809                     }
810                     transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
811                 }
812 
813                 for (FeatureI cdsFeature : cdsFeatures) {
814                     Feature cds = (Feature) cdsFeature;
815                     // for genemark it appears frame of 2 =1 and frame of 1 = 2
816                     // doesn't matter when you string cds regions together as one block
817                     // but does make a difference when you try to make a protein sequence for each CDS region where
818                     // you give up or borrow based on the frame value
819                     // compared with gff like files and docs for geneid and glimmer where geneid and glimmer both do it the same
820                     // way that appears to match the gff3 docs.
821                     int frame = cds.frame();
822                     if (frame == 1) {
823                         frame = 2;
824                     } else if (frame == 2) {
825                         frame = 1;
826                     } else {
827                         frame = 0;
828                     }
829                     String cdsName = cds.getAttribute("transcript_name");
830                     if (cdsName == null || cdsName.length() == 0) {
831                         cdsName = transcriptid + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
832                     }
833                     AccessionID cdsAccessionID = new AccessionID(cdsName);
834                     //ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
835                     transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), frame);
836                 }
837             }
838         }

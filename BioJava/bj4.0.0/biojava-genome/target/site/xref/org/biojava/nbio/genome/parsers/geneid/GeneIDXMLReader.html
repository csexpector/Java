41  /**
42   *
43   * @author Scooter Willis <willishf at gmail dot com>
44   */
45  public class GeneIDXMLReader {
46  
47  	private static final Logger logger = LoggerFactory.getLogger(GeneIDXMLReader.class);
48  
49      Document geneidDoc = null;
50  
51      public GeneIDXMLReader(String geneidXMLFile) throws Exception {
52          logger.info("Start read of {}", geneidXMLFile);
53          geneidDoc = XMLHelper.loadXML(geneidXMLFile);
54          logger.info("Read finished");
55      }
56  
57      public LinkedHashMap<String, ProteinSequence> getProteinSequences() throws Exception {
58          LinkedHashMap<String, ProteinSequence> proteinSequenceList = new LinkedHashMap<String, ProteinSequence>();
59          ArrayList<Element> elementList = XMLHelper.selectElements(geneidDoc.getDocumentElement(), "prediction/gene/protein");
60          logger.info("{} hits", elementList.size());
61  
62          for (Element proteinElement : elementList) {
63              Element geneElement = (Element) proteinElement.getParentNode();
64              String sequence = proteinElement.getTextContent().replaceAll("\\W","");
65              ProteinSequence proteinSequence = new ProteinSequence(sequence);
66              String idGene = geneElement.getAttribute("idGene");
67              proteinSequence.setAccession(new AccessionID(idGene));
68              proteinSequenceList.put(idGene, proteinSequence);
69          }
70  
71          return proteinSequenceList;
72      }
73  
74      public LinkedHashMap<String, DNASequence> getDNACodingSequences() throws Exception {
75          LinkedHashMap<String, DNASequence> dnaSequenceList = new LinkedHashMap<String, DNASequence>();
76          ArrayList<Element> elementList = XMLHelper.selectElements(geneidDoc.getDocumentElement(), "prediction/gene/cDNA");
77          logger.info("{} hits", elementList.size());
78  
79          for (Element dnaElement : elementList) {
80              Element geneElement = (Element) dnaElement.getParentNode();
81              String sequence = dnaElement.getTextContent().replaceAll("\\W","");
82              DNASequence dnaSequence = new DNASequence(sequence);
83              String idGene = geneElement.getAttribute("idGene");
84              dnaSequence.setAccession(new AccessionID(idGene));
85              dnaSequenceList.put(idGene, dnaSequence);
86          }
87  
88          return dnaSequenceList;
89      }
90  
91      public static void main(String[] args) {
92          try {
93              GeneIDXMLReader geneIDXMLReader = new GeneIDXMLReader("/Users/Scooter/scripps/dyadic/geneid/geneid/c1_geneid.xml");
94              LinkedHashMap<String, ProteinSequence> proteinSequenceHashMap = geneIDXMLReader.getProteinSequences();
95              FastaWriterHelper.writeProteinSequence(new File("/Users/Scooter/scripps/dyadic/geneid/geneid/c1_geneid.faa"), proteinSequenceHashMap.values());
96  
97              LinkedHashMap<String, DNASequence> dnaSequenceHashMap = geneIDXMLReader.getDNACodingSequences();
98              FastaWriterHelper.writeNucleotideSequence(new File("/Users/Scooter/scripps/dyadic/geneid/geneid/c1_geneid.fna"), dnaSequenceHashMap.values());
99  
100         } catch (Exception e) {
101             logger.error("Exception: ", e);
102         }
103     }
104 }

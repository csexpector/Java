26  package org.biojava.nbio.genome.parsers.gff;
27  
28  import java.util.LinkedHashMap;
29  
30  /**
31   *
32   * @author Scooter Willis <willishf at gmail dot com>
33   */
34  public class FeatureHelper {
35  
36      /**
37       * Build a list of individual features to allow easy indexing and to avoid iterating through large genome gff3 files
38       * The index for the returned HashMap is the value of the attribute used to build the index
39       * @param attribute
40       * @param list
41       * @return
42       */
43      static public LinkedHashMap<String,FeatureList> buildFeatureAtrributeIndex(String attribute,FeatureList list){
44  
45          LinkedHashMap<String,FeatureList> featureHashMap = new LinkedHashMap<String,FeatureList>();
46          FeatureList featureList = list.selectByAttribute(attribute);
47          for(FeatureI feature : featureList){
48              String value = feature.getAttribute(attribute);
49              FeatureList features = featureHashMap.get(value);
50              if(features == null){
51                  features = new FeatureList();
52                  featureHashMap.put(value, features);
53              }
54              features.add(feature);
55          }
56  
57          return featureHashMap;
58      }
59  
60  }

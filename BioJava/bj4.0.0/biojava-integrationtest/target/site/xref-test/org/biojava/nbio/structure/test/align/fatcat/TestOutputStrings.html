30     public void printFirstMismatch(String s1, String s2){
31        String[] spl1 = s1.split(newline);
32        String[] spl2 = s2.split(newline);
33  
34        for (int i = 0 ; i < spl1.length ; i++){
35  
36           String line1 = spl1[i];
37  
38           if ( i >= spl2.length){
39              System.err.println("s2 does not contain line " + (i+1));
40              return;
41           }
42           String line2 = spl2[i];
43  
44           if ( line1.equals(line2)){
45              continue;
46           }
47  
48           System.err.println("mismatch in line: " + (i+1));
49  
50           for ( int j = 0 ; j < line1.length();j++){
51              char c1 = line1.charAt(j);
52  
53              if ( j >= line2.length()){
54                 System.err.println("s2 is shorter than s1. length s1:" + line1.length() + " length2:" + line2.length() );
55                 return;
56              }
57  
58              char c2 = line2.charAt(j);
59              if ( c1 != c2){
60  
61                 System.err.println("line1: " + line1.substring(0,j+1));
62                 System.err.println("line2: " + line2.substring(0,j+1));
63  
64                 System.err.println("mismatch at position " + (j+1) + " c1: "+ c1 + " " + c2);
65               
66                 return;
67              }
68           }
69  
70  
71        }
72  
73     }

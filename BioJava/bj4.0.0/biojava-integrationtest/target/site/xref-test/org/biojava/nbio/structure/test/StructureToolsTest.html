103 	public void testGetSubRanges() throws StructureException {
104 		String range;
105 		Structure substr;
106 		Chain chain;
107 
108 		// normal substructures
109 		range = "A:3-7";
110 		substr = StructureTools.getSubRanges(structure2, range);
111 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
112 
113 		chain = substr.getChain(0);
114 
115 		assertEquals("Did not find the expected number of residues in "+range, 5, chain.getAtomLength() );
116 
117 		// full chains
118 		range = "A";
119 		substr = StructureTools.getSubRanges(structure2, range);
120 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
121 
122 		chain = substr.getChain(0);
123 
124 		assertEquals("Did not find the expected number of residues in "+range, 411, chain.getAtomLength() );
125 		//assertEquals("subrange doesn't equal original chain A.", structure2.getChainByPDB("A"), chain);
126 
127 		// full chains
128 		range = "A:";
129 		substr = StructureTools.getSubRanges(structure2, range);
130 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
131 
132 		chain = substr.getChain(0);
133 
134 		assertEquals("Did not find the expected number of residues in "+range, 411, chain.getAtomLength() );
135 		//assertEquals("subrange doesn't equal original chain A.", structure2.getChainByPDB("A"), chain);
136 
137 		// combined ranges
138 		range = "A:3-7,B:8-12";
139 		substr = StructureTools.getSubRanges(structure2, range);
140 		assertEquals("Wrong number of chains in "+range, 2, substr.size());
141 
142 		chain = substr.getChain(0);
143 		assertEquals("Did not find the expected number of residues in first chain of "+range, 5, chain.getAtomLength() );
144 
145 		chain = substr.getChain(1);
146 		assertEquals("Did not find the expected number of residues in second chain of "+range, 5, chain.getAtomLength() );
147 
148 		// combined ranges
149 		range = "A,B:8-12";
150 		substr = StructureTools.getSubRanges(structure2, range);
151 		assertEquals("Wrong number of chains in "+range, 2, substr.size());
152 
153 		chain = substr.getChain(0);
154 		assertEquals("Did not find the expected number of residues in first chain of "+range, 411, chain.getAtomLength() );
155 
156 		chain = substr.getChain(1);
157 		assertEquals("Did not find the expected number of residues in second chain of "+range, 5, chain.getAtomLength() );
158 
159 		// parentheses
160 		range = "(A:3-7)";
161 		substr = StructureTools.getSubRanges(structure2, range);
162 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
163 
164 		chain = substr.getChain(0);
165 		assertEquals("Did not find the expected number of residues in "+range, 5, chain.getAtomLength() );
166 
167 		// Special '-' case
168 		range = "-";
169 		substr = StructureTools.getSubRanges(structure2, range);
170 		assertEquals("Should have gotten whole structure",structure2, substr);
171 
172 		// Test single-chain syntax
173 		range = "_:";
174 		substr = StructureTools.getSubRanges(structure, range);
175 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
176 
177 		chain = substr.getChain(0);
178 		assertEquals("Did not find the expected number of residues in first chain of "+range, 123, chain.getAtomLength() );
179 
180 		// Test single-chain syntax in a multi-chain structure. Should give chain A.
181 		range = "_:";
182 		substr = StructureTools.getSubRanges(structure2, range);
183 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
184 		
185 		chain = substr.getChain(0);
186 		assertEquals("Chain _ not converted to chain A.","A",chain.getChainID());
187 		assertEquals("Did not find the expected number of residues in first chain of "+range, 411, chain.getAtomLength() );
188 		
189 		try {
190 			// Illegal chain name
191 			range = "X:";
192 			substr = StructureTools.getSubRanges(structure2, range);
193 			fail("Illegal chain name in '"+range+"'. Should throw StructureException");
194 		} catch(StructureException ex) {} //expected 
195 		
196 		// some negative tests
197 		try {
198 			range = "7-10";
199 			substr = StructureTools.getSubRanges(structure2, range);
200 			fail("Illegal range '"+range+"'. Should throw StructureException");
201 		} catch(StructureException ex) {} //expected 
202 		try {
203 			range = "A7-10";
204 			substr = StructureTools.getSubRanges(structure2, range);
205 			fail("Illegal range '"+range+"'. Should throw StructureException");
206 		} catch(StructureException ex) {} //expected 
207 	}

209 	public void testRevisedConvention() throws IOException, StructureException{
210 
211 		AtomCache cache = new AtomCache();
212 
213 		
214 		String name11 = "4hhb.A";
215 		Structure s = cache.getStructure(name11);
216 		assertTrue(s.getChains().size() == 1);
217 
218 
219 		String name12 = "4hhb.A:";
220 		s = cache.getStructure(name12);
221 		assertTrue(s.getChains().size() == 1);
222 
223 		String name13 = "4hhb.A_";
224 		s = cache.getStructure(name13);
225 		assertTrue(s.getChains().size() == 1);
226 		
227 		String name9 = "4hhb.C_1-83";
228 		String chainId = "C";
229 		s = cache.getStructure(name9);
230 
231 		assertTrue(s.getChains().size() == 1);
232 		Chain c = s.getChainByPDB(chainId);
233 		assertEquals(c.getChainID(),chainId);
234 		Atom[] ca = StructureTools.getAtomCAArray(s);
235 		assertEquals(83,ca.length);
236 
237 		String name10 = "4hhb.C_1-83,A_1-10";
238 		s = cache.getStructure(name10);
239 		assertTrue(s.getChains().size() == 2);
240 		ca = StructureTools.getAtomCAArray(s);
241 		assertEquals(93, ca.length);
242 
243 		
244 	}

289 	public void testGetSubRangesExtended() throws StructureException {
290 		String range;
291 		Structure substr;
292 		Chain chain;
293 
294 		// negative indices
295 		range = "A:-3-7";
296 		substr = StructureTools.getSubRanges(structure2, range);
297 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
298 
299 		chain = substr.getChain(0);
300 
301 		// Note residue 0 is missing from 1lnl
302 		assertEquals("Did not find the expected number of residues in "+range, 10, chain.getAtomLength() );
303 
304 		// double negative indices
305 		range = "A:-3--1";
306 		substr = StructureTools.getSubRanges(structure2, range);
307 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
308 
309 		chain = substr.getChain(0);
310 
311 		assertEquals("Did not find the expected number of residues in "+range, 3, chain.getAtomLength() );
312 
313 		// mixed indices
314 		range = "A:-3-+1";
315 		substr = StructureTools.getSubRanges(structure2, range);
316 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
317 
318 		chain = substr.getChain(0);
319 
320 		assertEquals("Did not find the expected number of residues in "+range, 4, chain.getAtomLength() );
321 
322 		// positive indices
323 		range = "A:+1-6";
324 		substr = StructureTools.getSubRanges(structure2, range);
325 		assertEquals("Wrong number of chains in "+range, 1, substr.size());
326 
327 		chain = substr.getChain(0);
328 
329 		assertEquals("Did not find the expected number of residues in "+range, 6, chain.getAtomLength() );
330 
331 
332 		// whitespace
333 		range = "A:3-7, B:8-12";
334 		substr = StructureTools.getSubRanges(structure2, range);
335 		assertEquals("Wrong number of chains in "+range, 2, substr.size());
336 
337 		chain = substr.getChain(0);
338 		assertEquals("Did not find the expected number of residues in first chain of "+range, 5, chain.getAtomLength() );
339 
340 		chain = substr.getChain(1);
341 		assertEquals("Did not find the expected number of residues in second chain of "+range, 5, chain.getAtomLength() );

405 	public void testCAmmCIF() throws StructureException {
406 		//mmCIF files left justify their atom names (eg "CA  "), so can have different behavior
407 		AtomCache pdbCache = new AtomCache();
408 		pdbCache.setUseMmCif(false);
409 		AtomCache mmcifCache = new AtomCache();
410 		mmcifCache.setUseMmCif(true);
411 		
412 		Structure pdb=null, mmcif=null;
413 		
414 		String name = "3PIU";
415 		try {
416 			pdb = pdbCache.getStructure(name);
417 			mmcif = mmcifCache.getStructure(name);
418 		} catch (IOException e) {
419 			assumeNoException(e);
420 		}
421 		
422 		Atom[] pdbCA = StructureTools.getAtomCAArray(pdb);
423 		Atom[] mmcifCA = StructureTools.getAtomCAArray(mmcif);
424 		
425 		assertEquals("PDB has wrong length",409,pdbCA.length);
426 		assertEquals("PDB has wrong length",409,mmcifCA.length);
427 	}

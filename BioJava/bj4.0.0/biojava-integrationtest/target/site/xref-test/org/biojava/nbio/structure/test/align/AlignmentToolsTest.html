46  	public void testIsSequential() throws StructureException, IOException {
47  		AtomCache cache = new AtomCache();
48  		
49  		String name1, name2;
50  		Atom[] ca1, ca2;
51  		AFPChain afpChain;
52  		StructureAlignment ce;
53  		
54  		
55  		// CP case
56  		name1="1QDM.A"; // swaposin
57  		name2="1NKL"; // saposin
58  		
59  		ca1=cache.getAtoms(name1);
60  		ca2=cache.getAtoms(name2);
61  		
62  		ce = StructureAlignmentFactory.getAlgorithm(CeCPMain.algorithmName);
63  		afpChain = ce.align(ca1,ca2);
64  		
65  		assertFalse("CeCPMain should give non-sequential alignments (between blocks).",AlignmentTools.isSequentialAlignment(afpChain,false));
66  		assertFalse("CeCPMain should give non-sequential alignments (within blocks).",AlignmentTools.isSequentialAlignment(afpChain,true));
67  
68  		// linear case		
69  		ce = StructureAlignmentFactory.getAlgorithm(CeMain.algorithmName);
70  		afpChain = ce.align(ca1,ca2);
71  		
72  		assertTrue("CeMain should give sequential alignments (within blocks).",AlignmentTools.isSequentialAlignment(afpChain,true));
73  		assertTrue("CeMain should give sequential alignments (between blocks).",AlignmentTools.isSequentialAlignment(afpChain,false));
74  
75  		// now change the block interior a bit
76  		
77  		int[][][] optAln = afpChain.getOptAln();
78  		int tmp;
79  		tmp = optAln[0][0][2];
80  		optAln[0][0][2] = optAln[0][0][1];
81  		optAln[0][0][1] = tmp;
82  		tmp = optAln[0][1][2];
83  		optAln[0][1][2] = optAln[0][1][1];
84  		optAln[0][1][1] = tmp;
85  		
86  		assertTrue("Modifying block interior shouldn't effect block sequence.",AlignmentTools.isSequentialAlignment(afpChain,false));
87  		assertFalse("Modifying block interior should be not sequential.",AlignmentTools.isSequentialAlignment(afpChain,true));
88  
89  	}

153 		// noisy C3 alignment
154 		Map<Integer,Integer> alignment1 = new HashMap<Integer,Integer>();
155 		alignment1.put(1, 5);
156 		alignment1.put(2, 6);
157 		alignment1.put(4, 7);
158 		alignment1.put(6, 9);
159 		alignment1.put(7, 11);
160 		alignment1.put(9, 2);
161 		alignment1.put(10, 3);
162 		alignment1.put(11, 4);
163 		
164 		// Sequential version of the alignment
165 		Map<Integer,Integer> sequentialForward = new HashMap<Integer,Integer>();
166 		sequentialForward.put(1, 2);
167 		sequentialForward.put(2, 3);
168 		sequentialForward.put(4, 4);
169 		sequentialForward.put(6, 5);
170 		sequentialForward.put(7, 6);
171 		sequentialForward.put(9, 7);
172 		sequentialForward.put(10, 9);
173 		sequentialForward.put(11, 11);
174 		
175 		// inverse of sequentialForward
176 		Map<Integer,Integer> sequentialBackward = new HashMap<Integer,Integer>();
177 		sequentialBackward.put(2, 1);
178 		sequentialBackward.put(3, 2);
179 		sequentialBackward.put(4, 4);
180 		sequentialBackward.put(5, 6);
181 		sequentialBackward.put(6, 7);
182 		sequentialBackward.put(7, 9);
183 		sequentialBackward.put(9, 10);
184 		sequentialBackward.put(11, 11);
185 		
186 	
187 		Map<Integer,Integer> result;
188 		
189 		result = AlignmentTools.guessSequentialAlignment(alignment1, false);
190 		assertEquals("Wrong forward alignment",sequentialForward,result);
191 	
192 		result = AlignmentTools.guessSequentialAlignment(alignment1, true);
193 		assertEquals("Wrong backward alignment",sequentialBackward,result);
194 	}

394 	 * Tests that {@link AlignmentTools#updateSuperposition(AFPChain, Atom[], Atom[])} calculates the correct RMSD and TM-score for an AFPChain of 1 block.
395 	 * TODO: Write a test with 2 blocks
396 	 */
397 	public void testUpdateSuperposition() throws IOException, StructureException {
398 		Structure s = StructureTools.getStructure("31BI");
399 		Atom[] ca1 = StructureTools.getAtomCAArray(s);
400 		Atom[] ca2 = StructureTools.getAtomCAArray(s);
401 		StringBuilder sb = new StringBuilder();
402 		BufferedReader br = new BufferedReader(new FileReader("src/test/resources/align/31BI_symm_align.xml"));
403 		String line = "";
404 		while ((line = br.readLine()) != null) {
405 			sb.append(line);
406 		}
407 		br.close();
408 		AFPChain afpChain = AFPChainXMLParser.fromXML(sb.toString(), ca1, ca2);
409 		afpChain.setTMScore(-1);
410 		afpChain.setTotalRmsdOpt(-1);
411 		AlignmentTools.updateSuperposition(afpChain, ca1, ca2);
412 		assertEquals("TM-score is wrong", 0.62779, afpChain.getTMScore(), 0.001);
413 		assertEquals("RMSD is wrong", 2.50569, afpChain.getTotalRmsdOpt(), 0.001);
414 	}

416 	/**
417 	 * Maps (Double d)->((int)(d+offset))
418 	 * @author blivens
419 	 *
420 	 */
421 	public static class OffsetMap extends AbstractMap<Double,Integer>
422 	{
423 		private int offset;
424 		public OffsetMap(int offset) {
425 			this.offset = offset;
426 		}
427 		@Override
428 		public Integer get(Object key) {
429 			if(key==null) return null;
430 			return (int)((Double)key+offset);
431 		}
432 		
433 		/**
434 		 * Always returns the empty set
435 		 */
436 		@Override
437 		public Set<java.util.Map.Entry<Double,Integer>> entrySet() {
438 			return Collections.emptySet();
439 		}
440 		
441 		@Override
442 		public boolean containsKey(Object key) {
443 			return true;
444 		}
445 	}

73  	private void flip(String name1, String name2, String algorithmName) throws StructureException, IOException{
74  
75  		AtomCache cache = new AtomCache();
76  
77  		Structure s1 = cache.getStructure(name1);
78  		Structure s2 = cache.getStructure(name2);
79  
80  		Atom[] ca1 = StructureTools.getAtomCAArray(s1);
81  		Atom[] ca2 = StructureTools.getAtomCAArray(s2);
82  
83  		StructureAlignment algorithm = StructureAlignmentFactory.getAlgorithm(algorithmName );
84  		AFPChain afpChain = algorithm.align(ca1,ca2);
85  		afpChain.setName1(name1);
86  		afpChain.setName2(name2);
87  
88  		afpChain.setCalculationTime(0);
89  
90  		String xml = AFPChainXMLConverter.toXML(afpChain, ca1, ca2);
91  
92  		AFPChain newC    = AFPChainXMLParser.fromXML(xml, ca1, ca2);			
93  		AFPChain flipped = AFPChainFlipper.flipChain(newC);
94  
95  		assertEquals(afpChain.getName1(), flipped.getName2());
96  		assertEquals(afpChain.getName2(),flipped.getName1());
97  		assertEquals(afpChain.getCa1Length(),flipped.getCa2Length());
98  		assertEquals(afpChain.getCa2Length(),flipped.getCa1Length());
99  		assertEquals(afpChain.getAlgorithmName(),flipped.getAlgorithmName());
100 		assertEquals(afpChain.getVersion(), flipped.getVersion());
101 
102 		
103 		String xmlNew = AFPChainXMLConverter.toXML(flipped, ca2, ca1);
104 
105 		AFPChain backChain = AFPChainXMLParser.fromXML(xmlNew, ca2, ca1);
106 
107 		AFPChain origFlip  = AFPChainFlipper.flipChain(backChain);
108 		//AFPChainXMLParser.rebuildAFPChain(origFlip, ca1, ca2);
109 		origFlip.setCalculationTime(0);
110 		
111 		String xmlBack = AFPChainXMLConverter.toXML(origFlip);
112 		if ( ! xmlBack.equals(xml)){
113 			printFirstMismatch(xmlBack, xml);
114 		}
115 
116 
117 		double rmsd1 = getRMSD(afpChain,ca1,ca2);
118 		double rmsd2 = getRMSD(flipped,ca2,ca1);
119 		//System.out.println("rmsd:" +rmsd1 + " " + rmsd2);
120 		assertTrue("The RMSD are vastly different!", Math.abs(rmsd1-rmsd2) < 0.01);
121 
122 		double rmsd3 = getRMSD(origFlip, ca1,ca2);
123 		assertTrue("The RMSD are vastly different!", Math.abs(rmsd1-rmsd3) < 0.01);
124 		//assertEquals(xmlBack, xml);
125 
126 
127 
128 	}

131 	/** get the RMSD between the aligned positions
132 	 * 
133 	 * @param afpChain
134 	 * @param ca1
135 	 * @param ca2
136 	 * @return
137 	 */
138 	private double getRMSD(AFPChain afpChain, Atom[] ca1, Atom[] ca2) 
139 	throws StructureException {
140 
141 		Atom[] ca2clone = StructureTools.cloneCAArray(ca2);
142 		rotateAtoms2(afpChain,ca2clone);
143 
144 		// get only the subset of Atoms that is on structurally equivalent positions
145 
146 		Atom[] catmp1 = AFPAlignmentDisplay.getAlignedAtoms1(afpChain, ca1);
147 		Atom[] catmp2 = AFPAlignmentDisplay.getAlignedAtoms2(afpChain, ca2clone);
148 
149 		assertTrue(catmp1.length == catmp2.length);
150 
151 		assertTrue(catmp1.length == afpChain.getNrEQR());
152 
153 		return SVDSuperimposer.getRMS(catmp1,catmp2);
154 	}

156 	public static void rotateAtoms2(AFPChain afpChain,Atom[] ca2){
157 
158 
159 
160 		int blockNum = afpChain.getBlockNum();
161 
162 		int[] optLen = afpChain.getOptLen();
163 		int[][][] optAln = afpChain.getOptAln();
164 
165 		for(int bk = 0; bk < blockNum; bk ++)       {
166 
167 			Matrix m= afpChain.getBlockRotationMatrix()[bk];
168 			Atom shift = afpChain.getBlockShiftVector()[bk];
169 			for ( int i=0;i< optLen[bk];i++){
170 				int pos = optAln[bk][1][i];
171 				Atom a = ca2[pos];
172 
173 				Calc.rotate(a, m);
174 				Calc.shift(a, shift);
175 
176 				//atoms.add(ca2[pos]);
177 			}
178 
179 		}
180 
181 	}

184 	static final String newline = System.getProperty("line.separator");
185 	public void printFirstMismatch(String s1, String s2){
186 		String[] spl1 = s1.split(newline);
187 		String[] spl2 = s2.split(newline);
188 
189 		for (int i = 0 ; i < spl1.length ; i++){
190 
191 			String line1 = spl1[i];
192 
193 			if ( i >= spl2.length){
194 				System.err.println("s2 does not contain line " + (i+1));
195 				return;
196 			}
197 			String line2 = spl2[i];
198 
199 			if ( line1.equals(line2)){
200 				continue;
201 			}
202 
203 			System.err.println("mismatch in line: " + (i+1));
204 
205 			for ( int j = 0 ; j < line1.length();j++){
206 				char c1 = line1.charAt(j);
207 
208 				if ( j >= line2.length()){
209 					System.err.println("s2 is shorter than s1. length s1:" + line1.length() + " length2:" + line2.length() );
210 					return;
211 				}
212 
213 				char c2 = line2.charAt(j);
214 				if ( c1 != c2){
215 
216 					System.err.println("line1: " + line1.substring(0,j+1));
217 					System.err.println("line2: " + line2.substring(0,j+1));
218 
219 					System.err.println("mismatch at position " + (j+1) + " c1: "+ c1 + " " + c2);
220 
221 					return;
222 				}
223 			}
224 
225 
226 		}

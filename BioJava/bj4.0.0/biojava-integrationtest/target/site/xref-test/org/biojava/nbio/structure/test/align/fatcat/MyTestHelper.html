40     public static final String pdbPath = System.getProperty("java.io.tmpdir");
41  
42     public static String compareAlignment(String pdb1, String chain1, String pdb2, String chain2, String originalOutput, boolean doRigid){
43  
44        AFPChain afpChain = null;
45  
46        PDBFileReader pdbpars = new PDBFileReader();
47        pdbpars.setPath(pdbPath);
48        pdbpars.setFetchBehavior(FetchBehavior.FETCH_FILES);
49        
50        FileParsingParameters params = new FileParsingParameters();
51        params.setAlignSeqRes(true);
52        params.setLoadChemCompInfo(false);
53        params.setParseCAOnly(true);
54        pdbpars.setFileParsingParameters(params);
55        
56        Structure structure1;
57        Structure structure2;
58  
59        //                   default:      new:
60        // 1buz - 1ali : time: 8.3s eqr 68 rmsd 3.1 score 161 | time 6.4 eqr 58 rmsd 3.0 scre 168 | rigid: identical, flexible: not significant alignment, 
61        // 5pti - 1tap : time: 6.2s eqr 48 rmsd 2.67 score 164 | time 5.2 eqr 49 rmsd 2.9 score 151 | rigid: 
62        // 1cdg - 8tim
63        // 1jbe - 1ord : identical with fatcat
64        // 1nbw.A - 1kid : rigid: identical, flexible: not identical, alignment not significant.
65        // 1t4y - 1rp5
66        // 1a64.A - 1hng.B
67        // 1zzw - 1bw6
68  
69        try {
70           structure1 = pdbpars.getStructureById(pdb1);
71           structure2 = pdbpars.getStructureById(pdb2);
72  
73           //structure1 = pdbpars.getStructureById("1cdg");
74           Chain c1 = structure1.getChainByPDB(chain1);
75  
76           //structure2 = pdbpars.getStructureById("2aaa");         
77           Chain c2 = structure2.getChainByPDB(chain2);
78  
79           Structure s3 = new StructureImpl();
80           s3.addChain(c1);
81  
82           Structure s4 = new StructureImpl();
83           s4.addChain(c2);
84  
85           Atom[] ca1 = StructureTools.getAtomCAArray(s3);
86           Atom[] ca2 = StructureTools.getAtomCAArray(s4);
87  
88           // keep an independent copy of them for tests further down..
89           Atom[] ca3 = new Atom[ca2.length];
90           for (int i = 0 ; i < ca2.length; i++){
91              Group g = (Group)ca2[i].getGroup().clone();
92              g.setChain(ca2[i].getGroup().getChain());
93              ca3[i] = g.getAtom(StructureTools.CA_ATOM_NAME);
94           }
95  
96           StructureAlignment fatCat;
97  
98           FatCatParameters fparams = new FatCatParameters();
99  
100          if ( doRigid)
101             fatCat = new FatCatRigid();            
102          else 
103             fatCat = new FatCatFlexible();
104 
105          afpChain = fatCat.align(ca1, ca2, fparams);
106 
107          afpChain.setName1(pdb1+chain1);
108          afpChain.setName2(pdb2+chain2);
109 
110 
111          // TEST THE XML SERIALIZATION AND DE_SERIALIZATION!
112          String result = afpChain.toFatcat(ca1, ca2);
113 
114          String xml = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
115 
116          //System.out.println(xml);
117 
118          AFPChain newChain = AFPChainXMLParser.fromXML (xml, ca1, ca3);
119 
120          String resultSerialized = newChain.toFatcat(ca1, ca3);
121          //System.out.println("*** RESULT2 "+result2);
122 
123          if ( ! result.equals(resultSerialized)) {
124             System.out.println("not identical toFatCat()!!!");
125             System.out.println(xml);
126             System.out.println(result);
127             System.out.println("***");
128             System.out.println(resultSerialized);
129             throw new StructureException("the JFatCat alignment image does not look identical after XML serialization.");        	 
130          }
131 
132          if ( ! afpChain.toString().equals(newChain.toString())){
133             System.err.println("not identical toStrings!!!");
134             System.err.println(afpChain.toString());
135             System.err.println(newChain.toString());
136             throw new StructureException("The AFPChain.toString() does not look identical after XML serialization.");
137          }
138 
139 
140 
141          if ( result.equals(originalOutput)){
142             return "";
143          } else {
144             return result;
145          }
146 
147       } catch ( Exception e){
148          e.printStackTrace();
149          return e.getMessage();
150       }
151 
152    }

210 	public void testMutation() throws Exception {
211 
212 		Group g1 = (Group)structure.getChain(0).getAtomGroup(21).clone();
213 		assertTrue(g1 != null);
214 
215 
216 		Group g2 = (Group)structure.getChain(0).getAtomGroup(53).clone();
217 		assertTrue(g2 != null);
218 
219 
220 		assertEquals("The group at position 22 is not a PHE","PHE", g1.getPDBName());
221 		assertEquals("The group position is  not number 22","22", g1.getResidueNumber().toString());
222 
223 		assertEquals("The group at position 54 is not a THR","THR", g2.getPDBName());
224 		assertEquals("The group position is not number 54","54", g2.getResidueNumber().toString());
225 
226 		Atom[] atoms1 = new Atom[3];
227 		Atom[] atoms2 = new Atom[3];
228 
229 		atoms1[0] = g1.getAtom("N");
230 		atoms1[1] = g1.getAtom("CA");
231 		atoms1[2] = g1.getAtom("CB");
232 
233 
234 		atoms2[0] = g2.getAtom("N");
235 		atoms2[1] = g2.getAtom("CA");
236 		atoms2[2] = g2.getAtom("CB");
237 
238 
239 		SVDSuperimposer svds = new SVDSuperimposer(atoms1,atoms2);
240 
241 
242 		Matrix rotMatrix = svds.getRotation();
243 		Atom   tran      = svds.getTranslation();
244 
245 		Group newGroup = (Group)g2.clone();
246 
247 		Calc.rotate(newGroup,rotMatrix);
248 
249 		Calc.shift(newGroup,tran);
250 
251 		Atom ca1    =       g1.getAtom("CA");
252 		Atom oldca2 =       g2.getAtom("CA");
253 		Atom newca2 = newGroup.getAtom("CA");
254 		Element e1 = ca1.getElement();
255 		
256 		assertEquals(Element.C, e1);
257 
258 		// this also tests the cloning ...
259 		double olddistance = Calc.getDistance(ca1,oldca2);
260 		assertTrue( olddistance > 10 );
261 
262 		// final test check that the distance between the CA atoms is small ;
263 
264 		double newdistance = Calc.getDistance(ca1,newca2);
265 		assertTrue( newdistance < 0.1);
266 
267 
268 	}

42  /**
43   * A test to make sure both PDB and mmCIF parsers can parse 
44   * properly large samples of the PDB. 
45   * 
46   * Will take very long to run, thus they are excluded by default in the pom.
47   * To run them use, for the 1000 entries one:
48   * <pre> 
49   * mvn -Dtest=TestLongPdbVsMmCifParsing#testLongPdbVsMmCif test
50   * </pre>
51   * or for the 10000 entries:
52   * <pre>
53   * mvn -Dtest=TestLongPdbVsMmCifParsing#testVeryLongPdbVsMmCif test
54   * </pre>
55   * 
56   * 
57   * @author duarte_j
58   *
59   */
60  public class TestLongPdbVsMmCifParsing {

139 		
140 		long start = System.currentTimeMillis();

142 		System.out.println("##### Total of "+pdbIds.size()+" PDB entries to test");	
143 		
144 		for (int i = 0; i<pdbIds.size(); i++) {
145 			pdbId = pdbIds.get(i);

147 			System.out.print(".");

226 	private void testHeader(Structure sPdb, Structure sCif) {
227 		
228 		PDBHeader hPdb = sPdb.getPDBHeader();
229 		PDBHeader hCif = sCif.getPDBHeader();
230 		
231 		boolean isNmr = sPdb.isNmr();
232 		boolean isCrystallographic = sPdb.isCrystallographic();
233 		
234 		assertNotNull(hPdb);
235 		assertNotNull(hCif);
236 		
237 		assertEquals("failed for PDB id (getIdCode)",hPdb.getIdCode(),hCif.getIdCode());
238 
239 		assertNotNull("pdb authors null",hPdb.getAuthors());
240 		assertNotNull("cif authors null",hCif.getAuthors());
241 		// I suppose 2 is a safe bet for authors length...
242 		assertTrue("authors length should be at least 2",hPdb.getAuthors().length()>=2);
243 		// for authors we strip spaces in case of ambiguities with names
244 		// there's too much variability in authors, commenting out, e.g. for 1zjo they don't coincide
245 		//assertEquals("failed getAuthors:",
246 		//		hPdb.getAuthors().toLowerCase().replaceAll(" ", ""),
247 		//		hCif.getAuthors().toLowerCase().replaceAll(" ", ""));
248 		
249 		assertNotNull("pdb classification null in pdb",hPdb.getClassification());
250 		assertNotNull("cif classification null in cif",hCif.getClassification());
251 		// there's too much variability in classification between pdb and mmcif, e.g. in 3ofb they don't coincide
252 		//assertEquals("failed getClassification:",hPdb.getClassification().toLowerCase(), hCif.getClassification().toLowerCase());
253 		
254 		// description is set in CIF parser to same as classification (_struct_keywords.pdbx_keywords field)
255 		// while in PDB parser it is simply not set
256 		//assertNotNull("pdb description null",hPdb.getDescription());
257 		assertNotNull("cif description null",hCif.getDescription());
258 		//assertEquals("failed getDescription:",hPdb.getDescription().toLowerCase(), hCif.getDescription().toLowerCase());
259 		
260 		assertEquals("failed getDepDate:",hPdb.getDepDate(), hCif.getDepDate());
261 		assertEquals("failed getModDate:",hPdb.getModDate(), hCif.getModDate());
262 		
263 		assertNotNull(hPdb.getExperimentalTechniques());
264 		assertNotNull(hCif.getExperimentalTechniques());
265 		assertTrue(hPdb.getExperimentalTechniques().size()>0);
266 		assertEquals("failed for getExperimentalTechniques",hPdb.getExperimentalTechniques(),hCif.getExperimentalTechniques());
267 
268 		// for some Electron Microscopy/Crystallography entries (e.g. 3iz2) the resolution in mmCIF is not present in the usual place
269 		if (!hPdb.getExperimentalTechniques().contains(ExperimentalTechnique.ELECTRON_CRYSTALLOGRAPHY) &&
270 				!hPdb.getExperimentalTechniques().contains(ExperimentalTechnique.ELECTRON_MICROSCOPY)) {			
271 			assertEquals("failed getResolution:",hPdb.getResolution(), hCif.getResolution(), DELTA_RESOLUTION);
272 		}
273 		
274 		// JRNL record is sometimes missing (e.g. 21bi) and thus is null, we can't test for nulls here in the general case 
275 		//assertNotNull("journal article null",hPdb.getJournalArticle());
276 		// TODO journal article not parsed in mmCIF parser
277 		// TODO when fixed in mmCIF parser, compare PDB to mmCIF values if not null
278 		//assertNotNull("journal article null",hCif.getJournalArticle());
279 		
280 		assertNotNull("title null in pdb",hPdb.getTitle());
281 		assertNotNull("title null in cif",hCif.getTitle());
282 		// for titles we strip spaces in case of ambiguities with spacing 
283 		assertEquals("failed for getTitle",
284 					hPdb.getTitle().toLowerCase().replaceAll(" ", ""),
285 					hCif.getTitle().toLowerCase().replaceAll(" ", ""));
286 		
287 		// tests specific to experimental techniques
288 		if (isNmr) {
289 			assertEquals("resolution is not the default value in NMR structure",
290 					PDBHeader.DEFAULT_RESOLUTION, hPdb.getResolution(), DELTA_RESOLUTION);
291 		}
292 		
293 		if (!isCrystallographic) {
294 			assertEquals("rfree is not the default value in non-crystallographic structure in pdb", 
295 					PDBHeader.DEFAULT_RFREE, DELTA_RFREE, hPdb.getRfree());
296 			assertEquals("rfree is not the default value in non-crystallographic structure in cif", 
297 					PDBHeader.DEFAULT_RFREE, DELTA_RFREE, hCif.getRfree());
298 		}
299 		
300 		if (isCrystallographic) {
301 			
302 			assertEquals("failed for Rfree:",hPdb.getRfree(), hCif.getRfree(), DELTA_RFREE);
303 			
304 			assertNotNull("getCrystallographicInfo is null in pdb",hPdb.getCrystallographicInfo());
305 			assertNotNull("getCrystallographicInfo is null in cif",hCif.getCrystallographicInfo());
306 			
307 			PDBCrystallographicInfo ciPdb = hPdb.getCrystallographicInfo();
308 			PDBCrystallographicInfo ciCif = hCif.getCrystallographicInfo();
309 			
310 			assertNotNull("space group null in pdb", ciPdb.getSpaceGroup());
311 			assertNotNull("space group null in cif", ciCif.getSpaceGroup());
312 			assertNotNull("crystal cell null in pdb",ciPdb.getCrystalCell());
313 			assertNotNull("crystal cell null in cif",ciCif.getCrystalCell());
314 			assertEquals("failed for space group short symbol pdb vs cif",
315 					ciPdb.getSpaceGroup().getShortSymbol(), ciCif.getSpaceGroup().getShortSymbol());
316 			
317 			CrystalCell ccPdb = ciPdb.getCrystalCell();
318 			CrystalCell ccCif = ciCif.getCrystalCell();
319 			
320 			assertEquals("failed for cell A:",ccPdb.getA(),ccCif.getA(),DELTA);
321 			assertEquals("failed for cell B:",ccPdb.getB(),ccCif.getB(),DELTA);
322 			assertEquals("failed for cell C:",ccPdb.getC(),ccCif.getC(),DELTA);
323 			assertEquals("failed for cell Alpha:",ccPdb.getAlpha(),ccCif.getAlpha(),DELTA);
324 			assertEquals("failed for cell Beta:",ccPdb.getBeta(),ccCif.getBeta(),DELTA);
325 			assertEquals("failed for cell Gamma:",ccPdb.getGamma(),ccCif.getGamma(),DELTA);
326 			
327 			
328 			if (ciPdb.getNcsOperators()==null) {
329 				assertTrue(ciCif.getNcsOperators()==null);
330 			} else {
331 				
332 				Matrix4d[] ncsOpersPdb = ciPdb.getNcsOperators();
333 				Matrix4d[] ncsOpersCif = ciCif.getNcsOperators();
334 								
335 				assertEquals("Number of NCS operators don't coincide", ncsOpersPdb.length, ncsOpersCif.length);
336 				
337 				for (int i=0;i<ncsOpersPdb.length;i++) {
338 					assertTrue("NCS operators "+i+" don't coincide",ncsOpersPdb[i].epsilonEquals(ncsOpersCif[i], 0.0001)); 
339 				}
340 			}
341 		}
342 		
343 		// biological assemblies
344 		// a) we don't test in non-crystallographic case because annotation is inconsistent between PDB and mmCIF,
345 		//    e.g. 2kli (NMR) has bioassembly annotation in mmCIF but not in PDB
346 		// b) we don't test virus entries (we check via looking at ncs operators==null): 
347 		//    they are inconsistent PDB vs mmCIF (e.g. 1pgw has no oligomeric size in PDB, and 120 in mmCIF) 
348 		if (isCrystallographic && hPdb.getCrystallographicInfo().getNcsOperators()==null
349 				// 1ruh, 2ms2, 2r06: virus proteins with data consistency issue: it's missing the MTRXn record (so it appears as ncs operators==null)
350 				&& (!sPdb.getPDBCode().equalsIgnoreCase("1ruh"))
351 				&& (!sPdb.getPDBCode().equalsIgnoreCase("2ms2")) 
352 				&& (!sPdb.getPDBCode().equalsIgnoreCase("2r06"))) {			
353 			
354 			assertEquals("Number of bioassemblies doesn't coincide",
355 					hPdb.getNrBioAssemblies(), hCif.getNrBioAssemblies());
356 
357 			Map<Integer,BioAssemblyInfo> batPdb = hPdb.getBioAssemblies();
358 			Map<Integer,BioAssemblyInfo> batCif = hCif.getBioAssemblies();		
359 
360 			assertEquals("Size of bioassemblies map doesn't coincide with nr of bioassemblies",
361 					hPdb.getNrBioAssemblies(),batPdb.size());
362 			assertEquals("Size of bioassemblies maps don't coincide",batPdb.size(), batCif.size());
363 			
364 			for (int id:batPdb.keySet()) {
365 				assertTrue("Bioassembly id is not contained in mmCIF",batCif.containsKey(id));
366 				// there's an inconsistency in 4amh pdb vs mmCIF in mmSize
367 				if (sPdb.getPDBCode().equalsIgnoreCase("4amh")) continue;
368 				
369 				assertEquals("Macromolecular size of assemblies doesn't coincide",
370 						batPdb.get(id).getMacromolecularSize(), batCif.get(id).getMacromolecularSize());
371 			}
372 		}
373 	}

391 	private void testSingleChain(Chain cPdb, Chain cCif) {
392 		assertNotNull(cPdb);
393 		assertNotNull(cCif);
394 		
395 		String chainId = cPdb.getChainID();
396 		
397 		assertEquals("failed for getChainID():",cPdb.getChainID(),cCif.getChainID());
398 		// TODO no internalChainID if parsed from PDB, should an ID be assigned following the same rules as in mmCIF?
399 		//assertEquals("failed for getInternalChainID():",cPdb.getInternalChainID(),cCif.getInternalChainID());
400 		assertNotNull("getInternalChainId is null",cCif.getInternalChainID());
401 		assertTrue("internalChainId used in mmCIF files must be at most 4 characters",cCif.getInternalChainID().length()<=4);
402 		assertEquals("chainID must be 1 character only, failed for pdb", 1, cPdb.getChainID().length());
403 		assertEquals("chainID must be 1 character only, failed for cif", 1, cCif.getChainID().length());
404 		
405 		// getCompound() is some times null for badly formatted PDB files (e.g. 4a10, all waters are in a separate chain F)
406 		if (isPolymer(cPdb)) {
407 			assertNotNull("getCompound is null in pdb (chain "+chainId+")",cPdb.getCompound());
408 			assertNotNull("getCompound is null in cif (chain "+chainId+")",cCif.getCompound());
409 			
410 			// for some badly formatted entries there are mismatches of mol_ids on pdb cs mmcif, e.g. 2efw
411 			// we thus count them and only warn at the end
412 			int molIdPdb = cPdb.getCompound().getMolId();
413 			int molIdCif = cCif.getCompound().getMolId();
414 			if (molIdPdb!=molIdCif) {
415 				logger.warn("Mismatching mol_id (entity_id) for {}. pdb: {}, mmCIF: {}",pdbId,molIdPdb,molIdCif);
416 				pdbIdsWithMismatchingMolIds.add(pdbId);
417 			}			
418 		}
419 
420 		
421 		assertNotNull("getParent is null in pdb (chain "+chainId+")",cPdb.getParent());
422 		assertNotNull("getParent is null in cif (chain "+chainId+")",cCif.getParent());
423 
424 			
425 		assertEquals("failed for getAtomLength (chain "+chainId+"):",cPdb.getAtomLength(),cCif.getAtomLength());
426 
427 		// note for getSeqResLength to work one needs the setAlignSeqRes option in the parsers
428 		
429 		assertEquals("failed for getSeqResLength pdb vs cif (chain "+chainId+"):",
430 						cPdb.getSeqResLength(),cCif.getSeqResLength());
431 		assertEquals("failed for getSeqResGroups().size pdb vs cif",
432 				cPdb.getSeqResGroups().size(), cCif.getSeqResGroups().size());
433 		assertEquals("getSeqResLength and getSeqResGroups.size should coincide in pdb:",
434 				cPdb.getSeqResLength(),cPdb.getSeqResGroups().size());
435 		assertEquals("getSeqResLength and getSeqResGroups.size should coincide in cif:",
436 				cCif.getSeqResLength(),cCif.getSeqResGroups().size());
437 
438 		
439 		assertEquals("failed for getAtomLength:",cPdb.getAtomLength(),cCif.getAtomLength());
440 		assertEquals("failed for getAtomGroups().size pdb vs cif",
441 				cPdb.getAtomGroups().size(), cCif.getAtomGroups().size());
442 		assertEquals("getAtomLength and getAtomGroups.size should coincide in pdb:",
443 				cPdb.getAtomLength(),cPdb.getAtomGroups().size());
444 		assertEquals("getAtomLength and getAtomGroups.size should coincide in cif:",
445 				cCif.getAtomLength(),cCif.getAtomGroups().size());
446 		
447 		assertEquals("failed for getAtomGroups(GroupType.AMINOACID) pdb vs cif:",
448 				cPdb.getAtomGroups(GroupType.AMINOACID).size(),cCif.getAtomGroups(GroupType.AMINOACID).size());
449 		assertEquals("failed for getAtomGroups(GroupType.HETATM) pdb vs cif:",
450 				cPdb.getAtomGroups(GroupType.HETATM).size(),cCif.getAtomGroups(GroupType.HETATM).size());
451 		assertEquals("failed for getAtomGroups(GroupType.NUCLEOTIDE) pdb vs cif:",
452 				cPdb.getAtomGroups(GroupType.NUCLEOTIDE).size(),cCif.getAtomGroups(GroupType.NUCLEOTIDE).size());
453 
454 		
455 		assertEquals("failed for getSeqResGroups(GroupType.AMINOACID) pdb vs cif:",
456 				cPdb.getSeqResGroups(GroupType.AMINOACID).size(),cCif.getSeqResGroups(GroupType.AMINOACID).size());
457 		assertEquals("failed for getAtomGroups(GroupType.HETATM) pdb vs cif:",
458 				cPdb.getSeqResGroups(GroupType.HETATM).size(),cCif.getSeqResGroups(GroupType.HETATM).size());
459 		assertEquals("failed for getAtomGroups(GroupType.NUCLEOTIDE) pdb vs cif:",
460 				cPdb.getSeqResGroups(GroupType.NUCLEOTIDE).size(),cCif.getSeqResGroups(GroupType.NUCLEOTIDE).size());
461 
462 
463 		
464 		assertTrue("getAtomLength must be at least 1 in length (chain "+chainId+")",cPdb.getAtomLength()>=1);		
465 		
466 		if (isPolymer(cPdb)) {
467 			// some badly formatted PDB files (e.g. 4a10, all waters are in a separate chain F) have 0 seqres length for some chains
468 			assertTrue("getSeqResLength must be at least 1 in length (chain "+chainId+")",cPdb.getSeqResLength()>=1);
469 		}
470 		
471 		// in the current implementation this is not a valid test, entries that have aminoacid residues in 
472 		// ligands, e.g. 3o6g won't pass this test
473 		//assertTrue("getSeqResLength ("+cPdb.getSeqResLength()+") must be >= than getAtomGroups(GroupType.AMINOACID).size() ("+
474 		//		cPdb.getAtomGroups(GroupType.AMINOACID).size()+") (chain "+chainId+")",
475 		//		cPdb.getSeqResLength()>=cPdb.getAtomGroups(GroupType.AMINOACID).size());
476 
477 		int allAtomGroupsSizePdb = 
478 				cPdb.getAtomGroups(GroupType.AMINOACID).size()+
479 				cPdb.getAtomGroups(GroupType.HETATM).size()+

528 	/**
529 	 * Reads a file containing a list of PDB codes.
530 	 * Lines starting with "#" will be treated as comments
531 	 * Will stop reading after finding an empty line, this is useful to quickly test a modified list. 
532 	 * @param testSetFile
533 	 * @return
534 	 * @throws IOException
535 	 */
536 	private List<String> readTestSetFile(String testSetFile) throws IOException {
537 		
538 		InputStream inStream = this.getClass().getResourceAsStream(testSetFile);
539 		BufferedReader br = new BufferedReader(new InputStreamReader(inStream));
540 		
541 		List<String> list = new ArrayList<String>();
542 		
543 		String line;
544 		while ((line=br.readLine())!=null) {
545 			if (line.startsWith("#")) continue;
546 			if (line.isEmpty()) break;
547 			
548 			if (!line.matches("\\d\\w\\w\\w")) 
549 				throw new IllegalArgumentException("The input test set "+testSetFile+" contains an invalid PDB code: "+line); 
550 			
551 			list.add(line);
552 		}
553 		br.close();
554 		
555 		return list;
556 	}

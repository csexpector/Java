23  package org.biojava.nbio.structure.test;
24  
25  import junit.framework.TestCase;
26  import org.biojava.nbio.structure.Structure;
27  import org.biojava.nbio.structure.align.StructurePairAligner;
28  import org.biojava.nbio.structure.align.pairwise.AlternativeAlignment;
29  import org.biojava.nbio.structure.io.PDBFileParser;
30  
31  import java.io.IOException;
32  import java.io.InputStream;
33  
34  public class StructurePairAlignerTest extends TestCase {
35  
36      Structure structure1;
37      Structure structure2;
38  
39      protected void setUp()
40      {
41          InputStream inStream = this.getClass().getResourceAsStream("/5pti.pdb");
42          assertNotNull(inStream);
43          InputStream inStream2 = this.getClass().getResourceAsStream("/1tap.pdb");
44          assertNotNull(inStream2);
45  
46          PDBFileParser pdbpars = new PDBFileParser();
47          try {
48              structure1 = pdbpars.parsePDBFile(inStream) ;
49              structure2 = pdbpars.parsePDBFile(inStream2);
50          } catch (IOException e) {
51              e.printStackTrace();
52          }
53  
54          assertNotNull(structure1);
55          assertNotNull(structure2);
56          assertEquals("structure does not contain one chain ", 1 ,structure1.size());
57  
58      }
59  
60  
61      public void testAlignStructureStructure() {
62         StructurePairAligner aligner = new StructurePairAligner();
63  
64         boolean allFine = true;
65         String msg = "";
66         try {
67             aligner.align(structure1,structure2);
68  
69             AlternativeAlignment[] aligs = aligner.getAlignments();
70             assertEquals("the number of obtained alternative alignments is not correct",20, aligs.length);
71             AlternativeAlignment a = aligs[0];
72             
73             assertNotNull(a);
74             
75             assertEquals("the expected nr of eq. residues is not correct.",47,a.getEqr());
76             
77             // they are v. close, but not identical
78             assertTrue(a.getRmsd() < 4);
79             assertTrue(a.getRmsd() > 3);
80             assertTrue(a.getPercId() > 9);
81             assertTrue(a.getScore() > 140);
82  
83         } catch (Exception e){
84             msg = e.getMessage();
85             allFine = false;
86             e.printStackTrace();
87         }
88         assertTrue(allFine);
89         assertEquals("an error occured","" ,msg);
90  
91  
92  
93      }
94  
95  
96  
97  }

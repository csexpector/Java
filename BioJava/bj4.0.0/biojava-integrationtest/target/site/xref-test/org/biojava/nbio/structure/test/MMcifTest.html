89  	protected void comparePDB2cif(String id, String chainId){
90  		String fileName = "/"+id+".cif";
91  		InputStream inStream = this.getClass().getResourceAsStream(fileName);
92  		assertNotNull("Could not find file " + fileName + ". Config problem?" , inStream);
93  
94  		MMcifParser parser = new SimpleMMcifParser();
95  
96  		SimpleMMcifConsumer consumer = new SimpleMMcifConsumer();
97  		FileParsingParameters params = new FileParsingParameters();
98  		params.setHeaderOnly(headerOnly);
99  		consumer.setFileParsingParameters(params);
100 
101 
102 		parser.addMMcifConsumer(consumer);
103 		try {
104 			parser.parse(new BufferedReader(new InputStreamReader(inStream)));
105 		} catch (IOException e){
106 			fail(e.getMessage());
107 		}
108 		// remove to avoid memory leaks
109 		parser.clearConsumers();
110 		Structure cifStructure = consumer.getStructure();
111 		assertNotNull(cifStructure);
112 
113 
114 		// load the PDB file via the PDB parser
115 		Structure pdbStructure = null;
116 		InputStream pinStream = this.getClass().getResourceAsStream("/"+id+".pdb");
117 		assertNotNull(inStream);
118 
119 		PDBFileParser pdbpars = new PDBFileParser();
120 		pdbpars.setFileParsingParameters(params);
121 
122 		try {
123 			pdbStructure = pdbpars.parsePDBFile(pinStream) ;

125 			e.printStackTrace();
126 		}
127 
128 		assertNotNull(pdbStructure);
129

226 	private void checkGroups(Group g1, Group g2){
227 
228 		//System.out.print("comparing " +g1 + " " + g2);
229 		String pdbId1 = g1.getChain().getParent().getPDBCode();
230 		String pdbId2 = g1.getChain().getParent().getPDBCode();
231 		assertEquals(pdbId1, pdbId2);
232 
233 		assertEquals(g1.getType(),g2.getType());
234 		assertEquals(g1.getResidueNumber().getSeqNum(),g2.getResidueNumber().getSeqNum());
235 		assertEquals(g1.getResidueNumber().getInsCode(),g2.getResidueNumber().getInsCode());
236 		assertEquals(g1.getPDBName(),g2.getPDBName());
237 		assertEquals(g1.has3D(),g2.has3D());
238 
239 		assertEquals(g1.hasAltLoc(), g2.hasAltLoc());
240 
241 		assertEquals(pdbId1 + ":" + g1 + " - " + pdbId2+":"+ g2,g1.getAltLocs().size(), g2.getAltLocs().size());
242 
243 		assertEquals(pdbId1 + ":" + g1 + " - " + pdbId2+":"+ g2 , g1.getAtoms().size(), g2.getAtoms().size());
244 		if ( g1.has3D()){
245 
246 			Atom a1 = g1.getAtom(0);
247 			Atom a2 = g2.getAtom(0);
248 			if ( a1 == null)
249 				fail("could not get atom for group " + g1);
250 			if (a2 == null)
251 				fail("could not get atom for group " + g2);
252 			assertEquals(a1.getX(),a2.getX());
253 			assertEquals(a1.getOccupancy(),a2.getOccupancy());
254 			assertEquals(a1.getTempFactor(),a2.getTempFactor());
255 			assertEquals(a1.getName(),a2.getName());
256 
257 
258 		}
259 		//System.out.println(" ... done");
260 
261 	}

263 	private void checkNMR(Structure s){
264 		assertTrue(s.isNmr());
265 
266 		int models = s.nrModels();
267 		assertTrue(models > 0);
268 
269 		List<Chain> model0 = s.getModel(0);
270 
271 		// compare with all others
272 		for (int i = 1 ; i < models; i++){
273 			List<Chain> modelX = s.getModel(i);
274 
275 			assertEquals(model0.size(),modelX.size());
276 
277 			// compare lengths:
278 			for (int j=0 ; j< model0.size();j++){
279 				Chain c1 = model0.get(j);
280 				Chain cx = modelX.get(j);
281 				assertEquals(c1.getAtomLength(),cx.getAtomLength());
282 				// can;t compare seq res, since this is only done for 1st...
283 				//assertEquals("c1.getSeqResLength(),cx.getSeqResLength());
284 				assertEquals(c1.getAtomSequence(),cx.getAtomSequence());
285 				assertEquals(c1.getAtomGroups(GroupType.AMINOACID).size(),cx.getAtomGroups(GroupType.AMINOACID).size());
286 				assertEquals(c1.getAtomGroups(GroupType.NUCLEOTIDE).size(),cx.getAtomGroups(GroupType.NUCLEOTIDE).size());
287 				assertEquals(c1.getAtomGroups(GroupType.HETATM).size(),cx.getAtomGroups(GroupType.HETATM).size());
288 			}
289 
290 
291 		}
292 	}

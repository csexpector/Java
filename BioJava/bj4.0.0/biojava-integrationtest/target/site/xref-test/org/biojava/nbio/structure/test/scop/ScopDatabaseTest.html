37  /**
38   * Generic tests for ScopDatabases. All implementing classes should pass these tests.
39   *
40   * This abstract class defines the tests for the interface. Each implementation
41   * should extend this class and implement the requirements for a parametic test
42   * (annotate the class with @RunWith, add an @Parameters static method,
43   * and provide a single constructor calling super)
44   * @author blivens
45   *
46   */
47  @RunWith(Parameterized.class)
48  public abstract class ScopDatabaseTest {
49      protected ScopDatabase scop;
50      String tag;
51  
52      /**
53       *
54       * @param tag A short string, displayed for failed asserts
55       * @param scop the database instance to test
56       */
57      public ScopDatabaseTest(String tag, ScopDatabase scop) {
58          if( tag != null) {
59              this.tag = "["+tag+"] "; // for messages
60          } else {
61              this.tag = "";
62          }
63  
64          this.scop = scop;
65      }

67      /**
68       * Traverse through the SCOP hierarchy
69       *
70       */
71      @Test
72      public void traverseHierarchy()
73      {
74          String pdbId = "4HHB";
75  
76          List<ScopDomain> domains = scop.getDomainsForPDB(pdbId);
77          assertEquals(tag+"Wrong number of domains",4,domains.size());
78  
79          // Check domains (order doesn't matter)
80          assertEquals(tag+"Wrong domain","d4hhba_",domains.get(0).getScopId());
81          assertEquals(tag+"Wrong domain","d4hhbb_",domains.get(2).getScopId());
82          assertEquals(tag+"Wrong domain","d4hhbc_",domains.get(1).getScopId());
83          assertEquals(tag+"Wrong domain","d4hhbd_",domains.get(3).getScopId());
84  
85          // Check the heirarchy
86          ScopNode node = scop.getScopNode(domains.get(0).getSunid());
87          ScopDescription desc = scop.getScopDescriptionBySunid(node.getSunid());
88          assertEquals(tag,15251,node.getSunid());
89          assertEquals(tag,"d4hhba_",desc.getName());
90          assertEquals(tag,"4hhb A:",desc.getDescription());
91          assertEquals(tag,"a.1.1.2",desc.getClassificationId());
92  
93          node = scop.getScopNode(node.getParentSunid());
94          desc = scop.getScopDescriptionBySunid(node.getSunid());
95          assertEquals(tag,46487,node.getSunid());
96          assertEquals(tag,"-",desc.getName());
97          assertTrue(tag,Pattern.matches("Human \\(Homo sapiens\\)( \\[TaxId: 9606\\])?",desc.getDescription()));
98          assertEquals(tag,"a.1.1.2",desc.getClassificationId());
99  
100         node = scop.getScopNode(node.getParentSunid());
101         desc = scop.getScopDescriptionBySunid(node.getSunid());
102         assertEquals(tag,46486,node.getSunid());
103         assertEquals(tag,"-",desc.getName());
104         assertEquals(tag,"Hemoglobin, alpha-chain",desc.getDescription());
105         assertEquals(tag,"a.1.1.2",desc.getClassificationId());
106 
107         node = scop.getScopNode(node.getParentSunid());
108         desc = scop.getScopDescriptionBySunid(node.getSunid());
109         assertEquals(tag,46463,node.getSunid());
110         assertEquals(tag,"-",desc.getName());
111         assertEquals(tag,"Globins",desc.getDescription());
112         assertEquals(tag,"a.1.1.2",desc.getClassificationId());
113 
114         node = scop.getScopNode(node.getParentSunid());
115         desc = scop.getScopDescriptionBySunid(node.getSunid());
116         assertEquals(tag,46458,node.getSunid());
117         assertEquals(tag,"-",desc.getName());
118         assertEquals(tag,"Globin-like",desc.getDescription());
119         assertEquals(tag,"a.1.1",desc.getClassificationId());
120 
121 
122         node = scop.getScopNode(node.getParentSunid());
123         desc = scop.getScopDescriptionBySunid(node.getSunid());
124         assertEquals(tag,46457,node.getSunid());
125         assertEquals(tag,"-",desc.getName());
126         assertEquals(tag,"Globin-like",desc.getDescription());
127         assertEquals(tag,"a.1",desc.getClassificationId());
128 
129         node = scop.getScopNode(node.getParentSunid());
130         desc = scop.getScopDescriptionBySunid(node.getSunid());
131         assertEquals(tag,46456,node.getSunid());
132         assertEquals(tag,"-",desc.getName());
133         assertEquals(tag,"All alpha proteins",desc.getDescription());
134         assertEquals(tag,"a",desc.getClassificationId());
135 
136         // root node
137         node = scop.getScopNode(node.getParentSunid());
138         desc = scop.getScopDescriptionBySunid(node.getSunid());
139         assertEquals(tag,0,node.getSunid());
140         assertNull(tag+"Root should not have a description", desc);
141     }

143     @Test
144     public void testNodes() {
145 
146         ScopNode node = scop.getScopNode(15251); //4hhb
147         assertEquals(tag,15251,node.getSunid());
148         assertEquals(tag,46487,node.getParentSunid());
149 
150         node = scop.getScopNode(46456); // all-alpha
151         assertEquals(tag,46456,node.getSunid());
152         assertEquals(tag,0,node.getParentSunid());
153 
154         node = scop.getScopNode(0); // root
155         assertEquals(tag,0,node.getSunid());
156         assertEquals(tag,-1,node.getParentSunid());
157         assertEquals(tag+"Wrong number of children",11,node.getChildren().size());
158 
159         node = scop.getScopNode(-1); // illegal
160         assertNull(tag,node);
161 
162         node = scop.getScopNode(Integer.MAX_VALUE); // unused
163         assertNull(tag,node);
164 
165     }

167     /** Get various categories
168      *
169      */
170     @Test
171     public void testCategories(){
172         List<ScopDescription> superfams = scop.getByCategory(ScopCategory.Superfamily);
173 
174         assertNotNull(tag,superfams);
175         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_75) == 0 ) {
176             assertEquals(tag,2223,superfams.size());
177         } else {
178             // defaults for other versions
179             assertFalse(tag,superfams.isEmpty());
180         }
181 
182         List<ScopDescription> folds = scop.getByCategory(ScopCategory.Fold);
183 
184         assertNotNull(tag,folds);
185         assertFalse(tag,folds.isEmpty());
186         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_75) == 0 ) {
187             assertEquals(tag,1393,folds.size());
188         }
189     }

191     @Test
192     public void testComments() {
193     	// Note that comments change often between SCOP versions.
194     	// This test is likely to need updating after each SCOPe release.
195         List<String> comments;
196 
197         // root node
198         comments = scop.getComments(0);
199         assertTrue(comments.isEmpty());
200 
201         //TODO add additional version checks, since comments change a lot
202 
203         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_71) >= 0 ) {
204             comments = scop.getComments(15016);
205             assertEquals(tag+"Wrong number of comments", 1, comments.size());
206             assertEquals(tag+"Wrong comment", "complexed with cmo, hem", comments.get(0).trim());
207             comments = scop.getComments(82738);
208             assertEquals(tag+"Wrong number of comments", 1, comments.size());
209             assertEquals(tag+"Wrong comment", "inverting reaction mechanism", comments.get(0).trim());
210         }
211         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_73) >= 0 &&
212         		scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_75) <= 0) {
213             // Note: only tested so far with 1.75, so may need some modification for earlier versions
214 
215             comments = scop.getComments(127355);
216             assertEquals(tag+"Wrong number of comments", 2, comments.size());
217             assertEquals(tag+"Wrong comment", "automatically matched to d2hbia_", comments.get(0).trim());
218             assertTrue(tag+"Wrong comment", Pattern.matches("complexed with hem(; mutant)?", comments.get(1).trim()));
219         }
220         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_75) == 0 ) {
221             comments = scop.getComments(160555);
222             assertEquals(tag+"Wrong number of comments", 1, comments.size());
223             assertEquals(tag+"Wrong comment", "<a href=\"http://pfam.sanger.ac.uk/family?acc=PF06262\">PF06262</a>; DUF1025; minimal zincin fold that retains 3-stranded mixed beta-sheet and contains HExxH motif in the C-terminal helix; no metal ion bound to this motif is observed in the first determined structures", comments.get(0));
224 
225 
226         }
227         if(scop.getScopVersion().compareToIgnoreCase( ScopFactory.VERSION_1_75B) >= 0 ) {
228             // The following were added or modified in 1.75B
229 
230             comments = scop.getComments(127355);
231             assertEquals(tag+"Wrong number of comments", 2, comments.size());
232             assertEquals(tag+"Wrong comment", "automated match to d2hbia_", comments.get(0).trim());
233             assertEquals(tag+"Wrong comment", "complexed with hem", comments.get(1).trim());
234             // d3ueea_ was added in 1.75B update
235             // domain
236             comments = scop.getComments(190700);
237             assertEquals(tag+"Wrong number of comments", 1, comments.size());
238             assertEquals(tag+"Wrong comment", "not a true protein", comments.get(0));
239 
240             // fold
241             comments = scop.getComments(57923);
242             assertEquals(tag+"Wrong number of comments", 1, comments.size());
243             assertEquals(tag+"Wrong comment", "metal(zinc)-bound alpha+beta fold", comments.get(0));
244         }
245     }

22  package org.biojava.nbio.ontology.utils;
23  
24  import java.util.Map;
25  
26  
27  
28  /**
29   * Annotation that is optimized for memory usage.  Access time
30   * is linear, so SmallAnnotations are not recommended when
31   * the number of entries is large.  However, they are fine for
32   * small numbers of keys.
33   *
34   * @author Thomas Down
35   * @author Matthew Pocock
36   * @since 1.2
37   *
38   * 
39   * A minimal-memory alternative to SimpleAnnotation
40   *
41   * 
42   * When creating a large number of small Annotation instances, it is worth
43   * instantiating SmallAnnotation. Small is anything up to at least 30 properties
44   * but will vary with the JavaVM and underlying platform.
45   */
46  
47  public class SmallAnnotation extends AbstractAnnotation {
48    private Map properties;
49    
50    protected final Map getProperties() {
51      if(!propertiesAllocated()) {
52        properties = new SmallMap();
53      }
54      return properties;
55    }
56    
57    protected final boolean propertiesAllocated() {
58      return properties != null;
59    }
60  
61    /**
62     * Return a new SmallAnnotation optimised for small sets of properties.
63     */
64    public SmallAnnotation() {
65      super();
66    }
67  
68    /**
69     * Return a new SmallAnnotation that copies all values from another annoation.
70     *
71     * @param ann  the Annoation to copy all values from
72     * @throws NullPointerException if ann is null
73     */
74    public SmallAnnotation(Annotation ann) {
75      super(ann);
76    }
77  
78    /**
79     * Return a new SmallAnnotation that copies all values from a Map.
80     *
81     * @param map  the Map to copy values from
82     */
83    public SmallAnnotation(Map map) {
84      super(map);
85    }
86  }

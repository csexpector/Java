21  package demo;
22  
23  import org.biojava.nbio.core.sequence.ProteinSequence;
24  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
25  import org.biojava.nbio.core.sequence.compound.AminoAcidCompoundSet;
26  import org.biojava.nbio.core.sequence.loader.UniprotProxySequenceReader;
27  import org.biojava.nbio.ronn.Jronn;
28  
29  import java.util.Arrays;
30  
31  public class PredictDisorder {
32    
33  	public static void main(String[] args) throws Exception{
34  
35  		String uniprotID = "O30642";
36  
37  		ProteinSequence seq = getUniprot(uniprotID);
38  		System.out.println("Protein Sequence: "+ seq.toString());
39  		AminoAcidCompoundSet compoundSet = AminoAcidCompoundSet.getAminoAcidCompoundSet();
40  
41  		if (!compoundSet.isValidSequence(seq) ) {
42  			System.err.println("Invalid sequence, exiting");
43  			System.exit(1);
44  		}
45  
46  		float[] values = Jronn.getDisorderScores(seq);
47  
48  		System.out.println("Disorder Scores: "+ Arrays.toString(values));
49  			
50  		
51  	}
52  
53  	/** Fetch a protein sequence from the UniProt web site
54  	 * 
55  	 * @param uniProtID
56  	 * @return a Protein Sequence
57  	 * @throws Exception
58  	 */
59  	private static ProteinSequence getUniprot(String uniProtID) throws Exception {
60  
61  		AminoAcidCompoundSet set = AminoAcidCompoundSet.getAminoAcidCompoundSet();
62  		UniprotProxySequenceReader<AminoAcidCompound> uniprotSequence = new UniprotProxySequenceReader<AminoAcidCompound>(uniProtID,set);
63  
64  		ProteinSequence seq = new ProteinSequence(uniprotSequence);
65  
66  		return seq;
67  	}
68  }

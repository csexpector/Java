50      public void testDNAAlignment() {
51  
52          try {
53              List<DNASequence> lst = getDNAFASTAFile();
54  
55              Profile<DNASequence, NucleotideCompound> profile = Alignments.getMultipleSequenceAlignment(lst);
56  
57              assertTrue(profile.getSize() == 10);
58  
59              assertTrue(profile.getAlignedSequence(1).getSequenceAsString().length() > 50);
60  
61  
62              // here how to print the MSA:
63  
64              //System.out.printf("MSA:%n%s%n", profile);
65          } catch (Exception e) {
66              e.printStackTrace();
67              fail(e.getMessage());
68          }
69          ConcurrencyTools.shutdown();
70      }

72      private static List<DNASequence> getDNAFASTAFile() throws Exception {
73  
74          InputStream inStream = TestDNAAlignment.class.getResourceAsStream(String.format("/dna-fasta.txt"));
75          LinkedHashMap<String, DNASequence> fastas = FastaReaderHelper.readFastaDNASequence(inStream);
76  
77          List<DNASequence> sequences = new ArrayList<DNASequence>();
78  
79          for (String key : fastas.keySet()) {
80              DNASequence seq = fastas.get(key);
81              sequences.add(seq);
82          }
83  
84          return sequences;
85      }

112         DNASequence target = new DNASequence("ACTGACGTGTAGCTGACTGA", DNACompoundSet.getDNACompoundSet());

115         SimpleGapPenalty gapP = new SimpleGapPenalty();
116         gapP.setOpenPenalty((short) 5);

124         	// expected exception

127     /**
128      * @author Daniel Cameron
129      */
130     public void testMixedCaseInputStringsMatchUnderlyingBases() throws CompoundNotFoundException {
131         DNASequence target = new DNASequence("AAAAAAAAGTC", DNACompoundSet.getDNACompoundSet());
132         DNASequence query = new DNASequence("aaaaaaaagtc", DNACompoundSet.getDNACompoundSet());
133         SubstitutionMatrix<NucleotideCompound> matrix = SubstitutionMatrixHelper.getNuc4_4();
134         SimpleGapPenalty gapP = new SimpleGapPenalty((short)5, (short)2);
135         // should be a full match with +5 per match
136         assertEquals(5.0 * query.getLength(), Alignments.getPairwiseAligner(query, target, PairwiseSequenceAlignerType.LOCAL, gapP, matrix).getScore(), PRECISION);
137     }

141     public void testNoAlignedBases() throws CompoundNotFoundException {
142         DNASequence target = new DNASequence("A", DNACompoundSet.getDNACompoundSet());
143         DNASequence query = new DNASequence("T", DNACompoundSet.getDNACompoundSet());
144         SubstitutionMatrix<NucleotideCompound> matrix = SubstitutionMatrixHelper.getNuc4_4();
145         SimpleGapPenalty gapP = new SimpleGapPenalty((short)0, (short)1);
146         PairwiseSequenceAligner<DNASequence, NucleotideCompound> aligner = Alignments.getPairwiseAligner(query, target, PairwiseSequenceAlignerType.GLOBAL, gapP, matrix);
147         assertEquals(2, aligner.getPair().getLength());
148     }

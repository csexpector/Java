24  package org.biojava.nbio.alignment.routines;
25  
26  import org.biojava.nbio.alignment.SimpleGapPenalty;
27  import org.biojava.nbio.alignment.SubstitutionMatrixHelper;
28  import org.biojava.nbio.alignment.template.GapPenalty;
29  import org.biojava.nbio.alignment.template.SubstitutionMatrix;
30  import org.biojava.nbio.core.exceptions.CompoundNotFoundException;
31  import org.biojava.nbio.core.sequence.DNASequence;
32  import org.biojava.nbio.core.sequence.ProteinSequence;
33  import org.biojava.nbio.core.sequence.compound.AmbiguityDNACompoundSet;
34  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
35  import org.biojava.nbio.core.sequence.compound.NucleotideCompound;
36  import org.junit.Before;
37  import org.junit.Test;
38  
39  import static org.junit.Assert.assertEquals;
40  import static org.junit.Assert.assertTrue;
41  
42  public class GuanUberbacherTest {
43  
44  	private static final double PRECISION = 0.00000001;
45  	
46      private ProteinSequence query, target;
47      private GapPenalty gaps;
48      private SubstitutionMatrix<AminoAcidCompound> blosum62;
49      private GuanUberbacher<ProteinSequence, AminoAcidCompound> alignment, self;
50  
51      @Before
52      public void setup() throws CompoundNotFoundException { 
53          query = new ProteinSequence("ARND");
54          target = new ProteinSequence("RDG");
55          gaps = new SimpleGapPenalty(10, 1);
56          blosum62 = SubstitutionMatrixHelper.getBlosum62();
57          alignment = new GuanUberbacher<ProteinSequence, AminoAcidCompound>(query, target, gaps, blosum62);
58          self = new GuanUberbacher<ProteinSequence, AminoAcidCompound>(query, query, gaps, blosum62);
59      }
60  
61      @Test
62      public void testGuanUberbacher() {
63          GuanUberbacher<ProteinSequence, AminoAcidCompound> gu =
64                  new GuanUberbacher<ProteinSequence, AminoAcidCompound>();
65          gu.setQuery(query);
66          gu.setTarget(target);
67          gu.setGapPenalty(gaps);
68          gu.setSubstitutionMatrix(blosum62);
69          assertEquals(gu.getScore(), alignment.getScore(), PRECISION);
70      }
71  
72      @Test
73      public void testGetComputationTime() {
74          assertTrue(alignment.getComputationTime() > 0);
75          assertTrue(self.getComputationTime() > 0);
76      }
77  
78      @Test
79      public void testGetProfile() {
80          assertEquals(String.format("ARND%n-RDG%n"), alignment.getProfile().toString());
81          assertEquals(String.format("ARND%nARND%n"), self.getProfile().toString());
82      }
83  
84      @Test
85      public void testGetMaxScore() {
86          assertEquals(21, alignment.getMaxScore(), PRECISION);
87          assertEquals(21, self.getMaxScore(), PRECISION);
88      }
89  
90      @Test
91      public void testGetMinScore() {
92          assertEquals(-27, alignment.getMinScore(), PRECISION);
93          assertEquals(-28, self.getMinScore(), PRECISION);
94      }
95  
96      @Test
97      public void testGetScore() {
98          assertEquals(-6, alignment.getScore(), PRECISION);
99          assertEquals(21, self.getScore(), PRECISION);
100     }
101 
102     @Test
103     public void testGetPair() {
104         assertEquals(String.format("ARND%n-RDG%n"), alignment.getPair().toString());
105         assertEquals(String.format("ARND%nARND%n"), self.getPair().toString());
106     }
107     /**
108      * @author Daniel Cameron 
109      */
110     @Test
111 	public void should_align_shorter_query() throws CompoundNotFoundException {
112     	DNASequence query = new DNASequence("A", AmbiguityDNACompoundSet.getDNACompoundSet());
113 		DNASequence target = new DNASequence("AT", AmbiguityDNACompoundSet.getDNACompoundSet());
114 		GuanUberbacher<DNASequence, NucleotideCompound> aligner = new GuanUberbacher<DNASequence, NucleotideCompound>(query, target, new SimpleGapPenalty((short)5, (short)2), SubstitutionMatrixHelper.getNuc4_4());
115 		assertEquals(String.format("A-%nAT%n"), aligner.getPair().toString());
116     }
117     /**
118      * @author Daniel Cameron 
119      */
120     @Test
121 	public void should_align_shorter_target() throws CompoundNotFoundException {
122     	DNASequence query = new DNASequence("AT", AmbiguityDNACompoundSet.getDNACompoundSet());
123 		DNASequence target = new DNASequence("A", AmbiguityDNACompoundSet.getDNACompoundSet());
124 		GuanUberbacher<DNASequence, NucleotideCompound> aligner = new GuanUberbacher<DNASequence, NucleotideCompound>(query, target, new SimpleGapPenalty((short)5, (short)2), SubstitutionMatrixHelper.getNuc4_4());
125 		assertEquals(String.format("AT%nA-%n"), aligner.getPair().toString());
126     }
127     /**
128      * @author Daniel Cameron 
129      */
130     @Test
131 	public void should_align_multiple_cuts() throws CompoundNotFoundException {
132     	DNASequence query = new DNASequence("AAT", AmbiguityDNACompoundSet.getDNACompoundSet());
133 		DNASequence target = new DNASequence("AATG", AmbiguityDNACompoundSet.getDNACompoundSet());
134 		GuanUberbacher<DNASequence, NucleotideCompound> aligner = new GuanUberbacher<DNASequence, NucleotideCompound>(query, target, new SimpleGapPenalty((short)0, (short)2), SubstitutionMatrixHelper.getNuc4_4());
135 		aligner.setCutsPerSection(2); // 3 bases with 2 cuts
136 		assertEquals(String.format("AAT-%nAATG%n"), aligner.getPair().toString());
137     }
138 }

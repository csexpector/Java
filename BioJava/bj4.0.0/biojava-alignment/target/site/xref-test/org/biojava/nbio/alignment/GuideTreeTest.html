24  package org.biojava.nbio.alignment;
25  
26  import org.biojava.nbio.alignment.Alignments.PairwiseSequenceScorerType;
27  import org.biojava.nbio.alignment.Alignments.ProfileProfileAlignerType;
28  import org.biojava.nbio.alignment.template.GapPenalty;
29  import org.biojava.nbio.alignment.template.GuideTreeNode;
30  import org.biojava.nbio.alignment.template.SubstitutionMatrix;
31  import org.biojava.nbio.core.exceptions.CompoundNotFoundException;
32  import org.biojava.nbio.core.sequence.ProteinSequence;
33  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
34  import org.junit.Before;
35  import org.junit.Test;
36  
37  import java.util.Arrays;
38  import java.util.List;
39  
40  import static org.junit.Assert.*;
41  
42  public class GuideTreeTest {
43  
44      private List<ProteinSequence> proteins;
45      private GapPenalty gaps;
46      private SubstitutionMatrix<AminoAcidCompound> blosum62;
47      private GuideTree<ProteinSequence, AminoAcidCompound> tree;
48  
49      @Before
50      public void setup() throws CompoundNotFoundException { 
51          proteins = Arrays.asList(new ProteinSequence[] {new ProteinSequence("ARND"), new ProteinSequence("ARND"),
52                  new ProteinSequence("HILK"), new ProteinSequence("ANDR")});
53          gaps = new SimpleGapPenalty((short) 2, (short) 1);
54          blosum62 = SubstitutionMatrixHelper.getBlosum62();
55          tree = new GuideTree<ProteinSequence, AminoAcidCompound>(proteins, Alignments.getAllPairsScorers(proteins,
56                  PairwiseSequenceScorerType.GLOBAL_IDENTITIES, gaps, blosum62));
57      }
58  
59      @Test
60      public void testGuideTree() {
61          assertNotNull(tree);
62      }
63  
64      @Test
65      public void testGetAllPairsScores() {
66          assertArrayEquals(tree.getAllPairsScores(), new double[] {4, 0, 3, 0, 3, 0}, 0.00001);
67      }
68  
69      @Test
70      public void testGetDistanceMatrix() {
71          assertArrayEquals(tree.getDistanceMatrix(), new double[][] {
72                  {0.0, 0.0, 1.0, 0.4},
73                  {0.0, 0.0, 1.0, 0.4},
74                  {1.0, 1.0, 0.0, 1.0},
75                  {0.4, 0.4, 1.0, 0.0}});
76      }
77  
78      @Test
79      public void testGetRoot() {
80          assertEquals(Alignments.getProgressiveAlignment(tree, ProfileProfileAlignerType.GLOBAL, gaps,
81                  blosum62).toString(), String.format("%s%n%s%n%s%n%s%n",
82                  "--ARND-",
83                  "--ARND-",
84                  "HILK---",
85                  "--A-NDR"));
86      }
87  
88      @Test
89      public void testGetScoreMatrix() {
90          assertArrayEquals(tree.getScoreMatrix(), new double[][] {
91                  {4, 4, 0, 3},
92                  {4, 6, 0, 3},
93                  {0, 0, 5, 0},
94                  {3, 3, 0, 6}});
95      }
96  
97      @Test
98      public void testGetSequences() {
99          List<ProteinSequence> list = tree.getSequences();
100         assertEquals(list.size(), 4);
101         assertEquals(list.get(0), proteins.get(0));
102         assertEquals(list.get(1), proteins.get(1));
103         assertEquals(list.get(2), proteins.get(2));
104         assertEquals(list.get(3), proteins.get(3));
105     }
106 
107     @Test
108     public void testIterator() {
109         int i = 0;
110         for (GuideTreeNode<ProteinSequence, AminoAcidCompound> n : tree) {
111             switch (i++) {
112             case 0: assertEquals(n.getName(), "1"); break;
113             case 1: assertEquals(n.getName(), "2"); break;
114             case 2: assertEquals(n.getName(), ""); break;
115             case 3: assertEquals(n.getName(), "3"); break;
116             case 4: assertEquals(n.getName(), ""); break;
117             case 5: assertEquals(n.getName(), "4"); break;
118             case 6: assertEquals(n.getName(), ""); break;
119             }
120         }
121     }
122 
123     @Test
124     public void testToString() {
125         assertEquals(tree.toString(),
126                 "(((1:0.0,2:0.0):-1.4,3:0.8999999999999999):-0.7,4:-0.7)");
127     }
128 
129 }

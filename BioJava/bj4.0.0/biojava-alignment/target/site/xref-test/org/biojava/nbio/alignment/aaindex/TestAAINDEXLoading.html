60  		String matrixName = "PRLA000101";
61  	
62  		SubstitutionMatrix<AminoAcidCompound> sdm = SubstitutionMatrixHelper.getMatrixFromAAINDEX(matrixName);
63  		
64  		int scale = 1;
65  		if ( sdm instanceof ScaledSubstitutionMatrix) {
66  			ScaledSubstitutionMatrix scaledSDM = (ScaledSubstitutionMatrix)sdm;
67  			scale = scaledSDM.getScale();
68  			assertEquals(100,scale);
69  		}
70  		
71  		
72  		AminoAcidCompoundSet aas = AminoAcidCompoundSet.getAminoAcidCompoundSet();
73  		
74  		AminoAcidCompound v = aas.getCompoundForString("V");
75  		AminoAcidCompound w = aas.getCompoundForString("W");
76  		AminoAcidCompound r = aas.getCompoundForString("R");
77  		AminoAcidCompound n = aas.getCompoundForString("N");
78  		
79  		short rn = sdm.getValue(r,n);		
80  		assertEquals(60,rn);
81  		
82  		short nr = sdm.getValue(n,r);
83  		assertEquals(rn,nr);
84  		
85  		
86  		short vv = sdm.getValue(v,v); 		
87  		assertEquals(205,vv);
88  		
89  		short vw = sdm.getValue(v,w);
90  		assertEquals( -109,vw);
91  		
92  		short wv = sdm.getValue(w,v);
93  		assertEquals(vw,wv);

152 		
153 		AminoAcidCompound j = aas.getCompoundForString("J");
154 		AminoAcidCompound y = aas.getCompoundForString("Y");

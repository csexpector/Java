24  package org.biojava.nbio.alignment;
25  
26  import org.biojava.nbio.alignment.template.AlignedSequence.Step;
27  import org.biojava.nbio.alignment.template.GapPenalty;
28  import org.biojava.nbio.alignment.template.Profile;
29  import org.biojava.nbio.alignment.template.ProfilePair;
30  import org.biojava.nbio.alignment.template.SubstitutionMatrix;
31  import org.biojava.nbio.core.exceptions.CompoundNotFoundException;
32  import org.biojava.nbio.core.sequence.ProteinSequence;
33  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
34  import org.junit.Before;
35  import org.junit.Test;
36  
37  import java.util.Arrays;
38  
39  import static org.junit.Assert.assertEquals;
40  
41  public class SimpleProfilePairTest {
42  
43      private ProteinSequence protein1, protein2, protein3, protein4;
44      private GapPenalty gaps;
45      private SubstitutionMatrix<AminoAcidCompound> blosum62;
46      private Profile<ProteinSequence, AminoAcidCompound> pair1, pair2;
47  
48      @Before
49      public void setup() throws CompoundNotFoundException { 
50          protein1 = new ProteinSequence("ARND");
51          protein2 = new ProteinSequence("ARND");
52          protein3 = new ProteinSequence("HILK");
53          protein4 = new ProteinSequence("ANDR");
54          gaps = new SimpleGapPenalty((short) 2, (short) 1);
55          blosum62 = SubstitutionMatrixHelper.getBlosum62();
56          pair1 = new NeedlemanWunsch<ProteinSequence, AminoAcidCompound>(protein1, protein2, gaps, blosum62).getPair();
57          pair2 = new NeedlemanWunsch<ProteinSequence, AminoAcidCompound>(protein3, protein4, gaps, blosum62).getPair();
58      }
59  
60      @Test
61      public void testSimpleProfilePair() {
62          ProfilePair<ProteinSequence, AminoAcidCompound> all =
63                  new SimpleProfilePair<ProteinSequence, AminoAcidCompound>(pair1, pair2, Arrays.asList(new Step[] {
64                  Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.GAP, Step.GAP}), Arrays.asList(
65                  new Step[] {Step.COMPOUND, Step.GAP, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND}));
66          assertEquals(all.toString(), String.format("ARND--%nARND--%n--HILK%nA-ND-R%n"));
67      }
68  
69      @Test
70      public void testGetQuery() {
71          ProfilePair<ProteinSequence, AminoAcidCompound> all =
72                  new SimpleProfilePair<ProteinSequence, AminoAcidCompound>(pair1, pair2, Arrays.asList(new Step[] {
73                  Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.GAP, Step.GAP}), Arrays.asList(
74                  new Step[] {Step.COMPOUND, Step.GAP, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND}));
75          assertEquals(all.getQuery(), pair1);
76      }
77  
78      @Test
79      public void testGetTarget() {
80          ProfilePair<ProteinSequence, AminoAcidCompound> all =
81                  new SimpleProfilePair<ProteinSequence, AminoAcidCompound>(pair1, pair2, Arrays.asList(new Step[] {
82                  Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.GAP, Step.GAP}), Arrays.asList(
83                  new Step[] {Step.COMPOUND, Step.GAP, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND, Step.COMPOUND}));
84          assertEquals(all.getTarget(), pair2);
85      }
86  
87  }

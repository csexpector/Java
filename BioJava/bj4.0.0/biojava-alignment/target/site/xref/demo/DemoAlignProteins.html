21  package demo;
22  
23  import org.biojava.nbio.alignment.Alignments;
24  import org.biojava.nbio.alignment.Alignments.PairwiseSequenceAlignerType;
25  import org.biojava.nbio.alignment.SimpleGapPenalty;
26  import org.biojava.nbio.alignment.SubstitutionMatrixHelper;
27  import org.biojava.nbio.alignment.template.GapPenalty;
28  import org.biojava.nbio.alignment.template.PairwiseSequenceAligner;
29  import org.biojava.nbio.alignment.template.SequencePair;
30  import org.biojava.nbio.alignment.template.SubstitutionMatrix;
31  import org.biojava.nbio.core.sequence.ProteinSequence;
32  import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
33  import org.biojava.nbio.core.sequence.io.FastaReaderHelper;
34  
35  import java.net.URL;
36  
37  public class DemoAlignProteins {
38  
39  	public static void main(String[] args) throws Exception {
40  
41  		String uniprotID1 = "P69905";
42  		String uniprotID2 = "P68871";
43  
44  		ProteinSequence s1 = getSequenceForId(uniprotID1);
45  		ProteinSequence s2 = getSequenceForId(uniprotID2);
46  
47  		SubstitutionMatrix<AminoAcidCompound> matrix = SubstitutionMatrixHelper.getBlosum65();
48  
49  		GapPenalty penalty = new SimpleGapPenalty();
50  
51  		int gop = 8;
52  		int extend = 1;
53  		penalty.setOpenPenalty(gop);
54  		penalty.setExtensionPenalty(extend);
55  
56  
57  		PairwiseSequenceAligner<ProteinSequence, AminoAcidCompound> smithWaterman =
58  				Alignments.getPairwiseAligner(s1, s2, PairwiseSequenceAlignerType.LOCAL, penalty, matrix);
59  
60  		SequencePair<ProteinSequence, AminoAcidCompound> pair = smithWaterman.getPair();
61  
62  
63  		System.out.println(pair.toString(60));
64  
65  		
66  	}
67  
68  	private static ProteinSequence getSequenceForId(String uniProtId) throws Exception {
69  		URL uniprotFasta = new URL(String.format("http://www.uniprot.org/uniprot/%s.fasta", uniProtId));
70  		ProteinSequence seq = FastaReaderHelper.readFastaProteinSequence(uniprotFasta.openStream()).get(uniProtId);
71  		System.out.printf("id : %s %s%s%s", uniProtId, seq, System.getProperty("line.separator"), seq.getOriginalHeader());
72  		System.out.println();
73  		
74  		return seq;
75  	}
76  }

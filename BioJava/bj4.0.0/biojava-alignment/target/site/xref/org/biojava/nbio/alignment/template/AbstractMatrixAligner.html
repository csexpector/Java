41  /**
42   * Implements common code for an {@link Aligner} which builds a score matrix during computation.
43   *
44   * @author Mark Chapman
45   * @author Daniel Cameron
46   * @param <S> each element of the alignment {@link Profile} is of type S
47   * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
48   */
49  public abstract class AbstractMatrixAligner<S extends Sequence<C>, C extends Compound> extends AbstractScorer
50          implements MatrixAligner<S, C> {
51  
52      // input fields
53      protected GapPenalty gapPenalty;
54      private SubstitutionMatrix<C> subMatrix;
55      private boolean local, storingScoreMatrix;
56      protected List<Anchor> anchors = new ArrayList<Anchor>();
57      protected int cutsPerSection;
58  
59      // output fields
60      protected Profile<S, C> profile;
61      /**
62       * Start position of the aligned sequence in the query and target respectively
63       */
64      protected int[] xyStart;
65      /**
66       * End position of the aligned sequence in the query and target respectively
67       */
68      protected int[] xyMax;
69      protected int max, min, score;
70      /**
71       * Dynamic programming score matrix
72       * The first dimension has the length of the first (query) sequence + 1
73       * The second has the length of the second (target) sequence + 1
74       * The third has length 1 for linear gap penalty and 3 for affine/constant gap
75       * (one each for match/substitution, deletion, insertion) 
76       */
77      protected int[][][] scores;
78      /**
79       * Friendly name of each copy of the scoring matrix.
80       * The number of elements must match the number of elements in third dimension of @see scores 
81       */
82      private String[] types;
83      protected long time = -1;

86       * Before running an alignment, data must be sent in via calls to {@link #setGapPenalty(GapPenalty)} and
87       * {@link #setSubstitutionMatrix(SubstitutionMatrix)}.
88       */

90      }

186     @Override
187     public int[][][] getScoreMatrix() {
188         boolean tempStoringScoreMatrix = storingScoreMatrix;
189         if (scores == null) {
190             storingScoreMatrix = true;
191             align();
192             if (scores == null) {
193                 return null;
194             }
195         }
196         int[][][] copy = scores;
197         if (tempStoringScoreMatrix) {
198             copy = new int[scores.length][scores[0].length][];
199             for (int i = 0; i < copy.length; i++) {
200                 for (int j = 0; j < copy[0].length; j++) {
201                     copy[i][j] = Arrays.copyOf(scores[i][j], scores[i][j].length);
202                 }
203             }
204         }
205         setStoringScoreMatrix(tempStoringScoreMatrix);
206         return copy;
207     }

209     @Override
210     public String getScoreMatrixAsString() {
211         int[][][] scores = getScoreMatrix();
212         return scoreMatrixToString(scores);
213     }
214 	private String scoreMatrixToString(int[][][] scores) {
215 		StringBuilder s = new StringBuilder();
216         CompoundSet<C> compoundSet = getCompoundSet();
217         int lengthCompound = compoundSet.getMaxSingleCompoundStringLength(), lengthRest =
218                 Math.max(Math.max(Integer.toString(min).length(), Integer.toString(max).length()), lengthCompound) + 1;
219         String padCompound = "%" + Integer.toString(lengthCompound) + "s",
220                 padRest = "%" + Integer.toString(lengthRest);
221         List<C> query = getCompoundsOfQuery(), target = getCompoundsOfTarget();
222         for (int type = 0; type < scores[0][0].length; type++) {
223             if (type > 0) {
224                 s.append(String.format("%n"));
225             }
226             if (types[type] != null) {
227                 s.append(String.format("%s%n", types[type]));
228             }
229             s.append(String.format(padCompound, ""));
230             s.append(String.format(padRest + "s", ""));
231             for (C col : target) {
232                 s.append(String.format(padRest + "s", compoundSet.getStringForCompound(col)));
233             }
234             s.append(String.format("%n"));
235             for (int x = 0; x < scores.length; x++) {
236                 s.append(String.format(padCompound, (x == 0) ? "" :
237                     compoundSet.getStringForCompound(query.get(x - 1))));
238                 for (int y = 0; y < scores[0].length; y++) {
239                     s.append(scores[x][y][type] >= min ? String.format(padRest + "d", scores[x][y][type]) :
240                             String.format(padRest + "s", "-\u221E"));
241                 }
242                 s.append(String.format("%n"));
243             }
244         }
245         return s.toString();
246 	}

293     /**
294      * Performs alignment
295      */
296     protected void align() {
297         if (!isReady()) {
298             return;
299         }
300 
301         long timeStart = System.nanoTime();
302 
303         int[] dim = getScoreMatrixDimensions();
304         if (storingScoreMatrix) {
305             scores = new int[dim[0]][dim[1]][dim[2]];
306         } else {
307             scores = new int[dim[0]][][];
308             scores[0] = new int[dim[1]][dim[2]];
309             scores[1] = new int[dim[1]][dim[2]];
310         }
311         boolean linear = (gapPenalty.getType() == GapPenalty.Type.LINEAR);
312         Last[][][] traceback = new Last[dim[0]][][];
313         List<Step> sx = new ArrayList<Step>(), sy = new ArrayList<Step>();
314 
315         if (!local) {
316         	xyMax = new int[] { dim[0] - 1, dim[1] - 1 };
317         	xyStart = new int[] { 0, 0 };
318             score = 0;
319             List<Subproblem> problems = Subproblem.getSubproblems(anchors, xyMax[0], xyMax[1]);
320             assert problems.size() == anchors.size() + 1;
321             for (int i = 0; i < problems.size(); i++) {
322             	Subproblem subproblem = problems.get(i);
323             	for (int x = subproblem.getQueryStartIndex(); x <= subproblem.getQueryEndIndex(); x++) {
324                 	
325             		traceback[x] = 
326                 		
327             		  linear ? 
328                 		setScoreVector(x, subproblem, gapPenalty.getExtensionPenalty(), getSubstitutionScoreVector(x, subproblem), storingScoreMatrix, scores) : 
329                 		
330                 		setScoreVector(x, subproblem, gapPenalty.getOpenPenalty(), gapPenalty.getExtensionPenalty(), getSubstitutionScoreVector(x, subproblem), storingScoreMatrix, scores);
331                 }
332             }
333             setSteps(traceback, scores, sx, sy);
334             score = Integer.MIN_VALUE;
335             int[] finalScore = scores[xyMax[0]][xyMax[1]];
336             for (int z = 0; z < finalScore.length; z++) {
337             	score = Math.max(score, finalScore[z]);
338             }
339         } else {
340             for (int x = 0; x < dim[0]; x++) {
341             	
342             	traceback[x] = 
343             			
344             	  linear ? 
345            			setScoreVector(x, gapPenalty.getExtensionPenalty(), getSubstitutionScoreVector(x), storingScoreMatrix, scores, xyMax, score) :
346                     
347            			setScoreVector(x, gapPenalty.getOpenPenalty(), gapPenalty.getExtensionPenalty(), getSubstitutionScoreVector(x), storingScoreMatrix, scores, xyMax, score);
348            				
349                 if (xyMax[0] == x) {
350                     score = scores[x][xyMax[1]][0];
351                 }
352             }
353             xyStart = local ? setSteps(traceback, xyMax, sx, sy) : setSteps(traceback, scores, sx, sy);
354         }
355 
356         setProfile(sx, sy);
357 
358         if (!storingScoreMatrix) {
359             scores = null;
360         }
361 
362         time = System.nanoTime() - timeStart;
363     }

365     /**
366      * Returns score for the alignment of the query column to all target columns
367      * @param queryColumn
368      * @return
369      */
370     protected int[] getSubstitutionScoreVector(int queryColumn) {
371         return getSubstitutionScoreVector(queryColumn, new Subproblem(0, 0, scores.length - 1, scores[0].length - 1));
372     }
373 
374     /**
375      * Returns score for the alignment of the query column to all target columns
376      * @param queryColumn
377      * @param subproblem
378      * @return
379      */
380     protected int[] getSubstitutionScoreVector(int queryColumn, Subproblem subproblem) {
381         int[] subs = new int[subproblem.getTargetEndIndex() + 1];
382         if (queryColumn > 0) {
383             for (int y = Math.max(1, subproblem.getTargetStartIndex()); y <= subproblem.getTargetEndIndex(); y++) {
384                 subs[y] = getSubstitutionScore(queryColumn, y);
385             }
386         }
387         return subs;
388     }

390     /**
391      * Resets output fields; should be overridden to set max and min
392      */
393     protected void reset() {
394         xyMax = new int[] {0, 0};
395         xyStart = new int[] {0, 0};
396         scores = null;
397         types = (gapPenalty == null || gapPenalty.getType() == GapPenalty.Type.LINEAR) ? new String[] { null } :
398                 new String[] { "Substitution", "Deletion", "Insertion" };
399         time = -1;
400         profile = null;
401     }

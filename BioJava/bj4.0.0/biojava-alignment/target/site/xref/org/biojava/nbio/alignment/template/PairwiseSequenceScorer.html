26  import org.biojava.nbio.core.sequence.template.Compound;
27  import org.biojava.nbio.core.sequence.template.Sequence;

29  /**
30   * Defines an algorithm which computes a score for a pair of sequences.
31   *
32   * @author Mark Chapman
33   * @param <S> each {@link Sequence} of the alignment pair is of type S
34   * @param <C> each element of a Sequence is a {@link Compound} of type C
35   */
36  public interface PairwiseSequenceScorer<S extends Sequence<C>, C extends Compound> extends Scorer {
37  
38      /**
39       * Returns the first sequence of the pair.
40       *
41       * @return the first sequence of the pair
42       */
43      S getQuery();
44  
45      /**
46       * Returns the second sequence of the pair.
47       *
48       * @return the second sequence of the pair
49       */
50      S getTarget();
51  
52  }

24  package org.biojava.nbio.alignment.template;
25  
26  /**
27   * Defines a data structure for the gap penalties used during a sequence alignment routine.
28   *
29   * @author Mark Chapman
30   */
31  public interface GapPenalty {
32  
33      /**
34       * Defines the possible types of gap penalties.  This is:
35       * <ul>
36       *  <li>CONSTANT, if static and the extension penalty is 0
37       *  <li>LINEAR, if static and the open penalty is 0
38       *  <li>AFFINE, if static but neither CONSTANT nor LINEAR
39       *  <li>DYNAMIC, if penalty values change during alignment
40       * </ul>
41       */
42      enum Type {CONSTANT, LINEAR, AFFINE, DYNAMIC};
43  
44      /**
45       * Returns penalty given when an already open gap elongates by a single element
46       *
47       * @return gap extension penalty
48       */
49      int getExtensionPenalty();
50  
51      /**
52       * Returns penalty given when a deletion or insertion gap first opens
53       *
54       * @return gap open penalty
55       */
56      int getOpenPenalty();
57  
58      /**
59       * Returns {@link GapPenalty.Type} stored.
60       *
61       * @return gap penalty type
62       */
63      Type getType();
64  
65      /**
66       * Sets penalty given when an already open gap elongates by a single element
67       *
68       * @param gep gap extension penalty
69       */
70      void setExtensionPenalty(int gep);
71  
72      /**
73       * Sets penalty given when a deletion or insertion gap first opens
74       *
75       * @param gop gap open penalty
76       */
77      void setOpenPenalty(int gop);
78  
79  }

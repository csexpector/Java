24  package org.biojava.nbio.alignment;
25  
26  import org.biojava.nbio.alignment.template.AlignedSequence;
27  import org.biojava.nbio.alignment.template.AlignedSequence.Step;
28  import org.biojava.nbio.alignment.template.Profile;
29  import org.biojava.nbio.alignment.template.ProfilePair;
30  import org.biojava.nbio.core.sequence.template.Compound;
31  import org.biojava.nbio.core.sequence.template.Sequence;
32  
33  import java.util.List;
34  
35  /**
36   * Implements a data structure for the results of the alignment of a pair of {@link Profile}s.
37   *
38   * @author Mark Chapman
39   * @param <S> each element of an alignment {@link Profile} is of type S
40   * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
41   */
42  public class SimpleProfilePair<S extends Sequence<C>, C extends Compound> extends SimpleProfile<S, C>
43          implements ProfilePair<S, C> {
44  
45      private Profile<S, C> query, target;
46  
47      /**
48       * Creates a pair profile for the given profiles.
49       *
50       * @param query the first profile of the pair
51       * @param target the second profile of the pair
52       * @param sx lists whether the query profile aligns a {@link Compound} or gap at each index of the alignment
53       * @param sy lists whether the target profile aligns a {@link Compound} or gap at each index of the alignment
54       * @throws IllegalArgumentException if alignments differ in size or given profiles do not fit in alignments
55       */
56      public SimpleProfilePair(Profile<S, C> query, Profile<S, C> target, List<Step> sx, List<Step> sy) {
57          super(query, target, sx, sy);
58          this.query = query;
59          this.target = target;
60      }
61  
62      @Override
63      public Profile<S, C> getQuery() {
64          return query;
65      }
66  
67      @Override
68      public Profile<S, C> getTarget() {
69          return target;
70      }
71  
72  }

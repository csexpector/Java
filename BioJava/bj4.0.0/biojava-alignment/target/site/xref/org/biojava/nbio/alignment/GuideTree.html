42  /**
43   * Implements a data structure for a guide tree used during progressive multiple sequence alignment.  Leaf
44   * {@link Node}s correspond to single {@link Sequence}s.  Internal {@link Node}s correspond to multiple sequence
45   * alignments.  The root {@link Node} corresponds to the full multiple sequence alignment.
46   *
47   * @author Mark Chapman
48   * @param <S> each {@link Sequence} in the tree is of type S
49   * @param <C> each element of a {@link Sequence} is a {@link Compound} of type C
50   */
51  public class GuideTree<S extends Sequence<C>, C extends Compound> implements Iterable<GuideTreeNode<S, C>> {
52  
53      private List<S> sequences;
54      private List<PairwiseSequenceScorer<S, C>> scorers;
55      private BasicSymmetricalDistanceMatrix distances;
56      private String newick;
57      private Node root;
58  
59      /**
60       * Creates a guide tree for use during progressive multiple sequence alignment.
61       *
62       * @param sequences the {@link List} of {@link Sequence}s to align
63       * @param scorers list of sequence pair scorers, one for each pair of sequences given
64       */
65      public GuideTree(List<S> sequences, List<PairwiseSequenceScorer<S, C>> scorers) {
66          this.sequences = Collections.unmodifiableList(sequences);
67          this.scorers = Collections.unmodifiableList(scorers);
68          distances = new BasicSymmetricalDistanceMatrix(sequences.size());
69          BasicSymmetricalDistanceMatrix distclone = new BasicSymmetricalDistanceMatrix(sequences.size());
70          for (int i = 0, n = 0; i < sequences.size(); i++) {
71              AccessionID id = sequences.get(i).getAccession();
72              String str = (id == null) ? Integer.toString(i + 1) : id.getID();
73              distances.setIdentifier(i, str);
74              distclone.setIdentifier(i, str);
75              for (int j = i+1; j < sequences.size(); j++) {
76                  double dist = scorers.get(n++).getDistance();
77                  distances.setValue(i, j, dist);
78                  distclone.setValue(i, j, dist);
79              }
80          }
81          // TODO UPGMA and other hierarchical clustering routines
82          Phylogeny phylogeny = NeighborJoining.createInstance().execute(distclone);
83          newick = phylogeny.toString();
84          root = new Node(phylogeny.getRoot(), null);
85      }
86  
87      /**
88       * Returns a sequence pair score for all {@link Sequence} pairs in the given {@link List}.
89       *
90       * @return list of sequence pair scores
91       */
92      public double[] getAllPairsScores() {
93      	double[] scores = new double[scorers.size()];
94          int n = 0;
95          for (PairwiseSequenceScorer<S, C> scorer : scorers) {
96              scores[n++] = scorer.getScore();
97          }
98          return scores;
99      }
100 
101     /**
102      * Returns the distance matrix used to construct this guide tree.  The scores have been normalized.
103      *
104      * @return the distance matrix used to construct this guide tree
105      */
106     public double[][] getDistanceMatrix() {
107         double[][] matrix = new double[distances.getSize()][distances.getSize()];
108         for (int i = 0; i < matrix.length; i++) {
109             for (int j = i+1; j < matrix.length; j++) {
110                 matrix[i][j] = matrix[j][i] = distances.getValue(i, j);
111             }
112         }
113         return matrix;
114     }
115 
116     /**
117      * Returns the root {@link Node} which corresponds to the full multiple sequence alignment.
118      *
119      * @return the root node
120      */
121     public Node getRoot() {
122         return root;
123     }
124 
125     /**
126      * Returns the similarity matrix used to construct this guide tree.  The scores have not been normalized.
127      *
128      * @return the similarity matrix used to construct this guide tree
129      */
130     public double[][] getScoreMatrix() {
131     	double[][] matrix = new double[sequences.size()][sequences.size()];
132         for (int i = 0, n = 0; i < matrix.length; i++) {
133             matrix[i][i] = scorers.get(i).getMaxScore();
134             for (int j = i+1; j < matrix.length; j++) {
135                 matrix[i][j] = matrix[j][i] = scorers.get(n++).getScore();
136             }
137         }
138         return matrix;
139     }

144      * @return the sequences which make up the leaves of this tree
145      */
146     public List<S> getSequences() {
147         return sequences;
148     }

167     /**
168      * Implements a data structure for the node in a guide tree used during progressive multiple sequence alignment.
169      */
170     public class Node implements GuideTreeNode<S, C> {
171 
172         private GuideTreeNode<S, C> parent, child1, child2;
173         private double distance;
174         private String name;
175         private boolean isLeaf, isVisited;
176         private Profile<S, C> profile;
177         private Future<ProfilePair<S, C>> profileFuture;
178 
179         private Node(PhylogenyNode node, Node parent) {
180             this.parent = parent;
181             distance = node.getDistanceToParent();
182             name = node.getName();
183             if(isLeaf = node.isExternal()) {
184                 profile = new SimpleProfile<S, C>(sequences.get(distances.getIndex(name)));
185             } else {
186                 child1 = new Node(node.getChildNode1(), this);
187                 child2 = new Node(node.getChildNode2(), this);
188             }
189         }
190 
191         // methods for GuideTreeNode
192 
193         @Override
194         public GuideTreeNode<S, C> getChild1() {
195             return child1;
196         }
197 
198         @Override
199         public GuideTreeNode<S, C> getChild2() {
200             return child2;
201         }
202 
203         @Override
204         public double getDistanceToParent() {
205             return distance;
206         }
207 
208         @Override
209         public String getName() {
210             return name;
211         }
212 
213         @Override
214         public Profile<S, C> getProfile() {
215             return profile;
216         }
217 
218         @Override
219         public Future<ProfilePair<S, C>> getProfileFuture() {
220             return profileFuture;
221         }
222 
223         @Override
224         public void setProfile(Profile<S, C> profile) {
225             this.profile = profile;
226             profileFuture = null;
227         }
228 
229         @Override
230         public void setProfileFuture(Future<ProfilePair<S, C>> profileFuture) {
231             this.profileFuture = profileFuture;
232             profile = null;
233         }
234 
235         // methods for TreeNode
236 
237         @Override
238         public Enumeration<GuideTreeNode<S, C>> children() {
239             Vector<GuideTreeNode<S, C>> children = new Vector<GuideTreeNode<S, C>>();
240             children.add(getChild1());
241             children.add(getChild2());
242             return children.elements();
243         }
244 
245         @Override
246         public boolean getAllowsChildren() {
247             return !isLeaf();
248         }
249 
250         @Override
251         public GuideTreeNode<S, C> getChildAt(int childIndex) {
252             if (childIndex == 1) {
253                 return getChild1();
254             } else if (childIndex == 2) {
255                 return getChild2();
256             }
257             throw new IndexOutOfBoundsException();
258         }
259 
260         @Override
261         public int getChildCount() {
262             return 2;
263         }
264 
265         @Override
266         public int getIndex(TreeNode child) {
267             return getChildAt(1) == child ? 1 : (getChildAt(2) == child ? 2 : -1);
268         }
269 
270         @Override
271         public GuideTreeNode<S, C> getParent() {
272             return parent;
273         }
274 
275         @Override
276         public boolean isLeaf() {
277             return isLeaf;
278         }
279 
280         // helper methods for iterator
281 
282         private void clearVisited() {
283             isVisited = false;
284             if (!isLeaf()) {
285                 ((Node) getChild1()).clearVisited();
286                 ((Node) getChild2()).clearVisited();
287             }
288         }
289 
290         private boolean isVisited() {
291             return isVisited;
292         }
293 
294         private void visit() {
295             isVisited = true;
296         }
297 
298     }

300     // helper class that defines the default post-order (leaves to root) traversal
301     private class PostOrderIterator implements Iterator<GuideTreeNode<S, C>> {
302 
303         private Stack<Node> nodes;
304 
305         private PostOrderIterator() {
306             getRoot().clearVisited();
307             nodes = new Stack<Node>();
308             nodes.push(getRoot());
309         }
310 
311         // methods for Iterator
312 
313         @Override
314         public boolean hasNext() {
315             return !nodes.isEmpty();
316         }
317 
318         @Override
319         public GuideTreeNode<S, C> next() {
320             while (hasNext()) {
321                 Node next = nodes.peek(), child1 = (Node) next.getChild1(), child2 = (Node) next.getChild2();
322                 if (child1 != null && !child1.isVisited()) {
323                     nodes.push(child1);
324                 } else if (child2 != null && !child2.isVisited()) {
325                     nodes.push(child2);
326                 } else {
327                     next.visit();
328                     return nodes.pop();
329                 }
330             }
331             return null;
332         }
333 
334         @Override
335         public void remove() {
336             throw new UnsupportedOperationException();
337         }
338 
339     }

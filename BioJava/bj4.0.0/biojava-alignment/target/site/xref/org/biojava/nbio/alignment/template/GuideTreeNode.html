24  package org.biojava.nbio.alignment.template;
25  
26  import org.biojava.nbio.core.sequence.template.Compound;
27  import org.biojava.nbio.core.sequence.template.Sequence;
28  
29  import javax.swing.tree.TreeNode;
30  import java.util.concurrent.Future;
31  
32  /**
33   * Defines a data structure for the node in a guide tree used during progressive multiple sequence alignment.
34   * 
35   * @author Mark Chapman
36   * @param <S> each {@link Sequence} in the tree is of type S
37   * @param <C> each element of a {@link Sequence} is a {@link Compound} of type C
38   */
39  public interface GuideTreeNode<S extends Sequence<C>, C extends Compound> extends TreeNode {
40  
41      /**
42       * Returns the first child node of this node.  For leaf nodes (sequences), this will be null.
43       *
44       * @return the first child node of this node
45       */
46      GuideTreeNode<S, C> getChild1();
47  
48      /**
49       * Returns the second child node of this node.  For leaf nodes (sequences), this will be null.
50       *
51       * @return the second child node of this node
52       */
53      GuideTreeNode<S, C> getChild2();
54  
55      /**
56       * Returns the difference in height of this node and it's parent node.  A likely meaning of this distance is half
57       * the percent difference between this node and it's sibling node.
58       *
59       * @return the difference in height of this node to it's parent node
60       */
61      double getDistanceToParent();
62  
63      /**
64       * Returns the name of this node.  For leaf nodes (sequences), this will likely be the accession ID.
65       *
66       * @return the name of this node
67       */
68      String getName();
69  
70      /**
71       * Returns the profile stored at this node.  If the node is a leaf, the profile is that of a single sequence.  If
72       * not, this returns null until {@link #setProfile(Profile)} has been called.
73       *
74       * @return the profile stored at this node
75       */
76      Profile<S, C> getProfile();
77  
78      /**
79       * Returns the profile future stored at this node, but does not force the calculation, yet.  This allows alignment
80       * tasks for the entire tree to be queued in a post-order traversal before concurrent execution.
81       *
82       * @return the profile future stored at this node
83       */
84      Future<ProfilePair<S, C>> getProfileFuture();
85  
86      /**
87       * Stores the given profile.
88       *
89       * @param profile new profile stored at this node
90       */
91      void setProfile(Profile<S, C> profile);
92  
93      /**
94       * Stores the given profile future.  This allows concurrent execution of alignment tasks.
95       *
96       * @param profileFuture new profile to be calculated and then stored at this node
97       */
98      void setProfileFuture(Future<ProfilePair<S, C>> profileFuture);
99  
100 }

24  package org.biojava.nbio.alignment.template;
25  
26  import org.biojava.nbio.core.sequence.template.Compound;
27  import org.biojava.nbio.core.sequence.template.Sequence;
28  
29  /**
30   * Defines a data structure for the results of pairwise sequence alignment.
31   *
32   * @author Mark Chapman
33   * @param <S> each element of the alignment {@link Profile} is of type S
34   * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
35   */
36  public interface SequencePair<S extends Sequence<C>, C extends Compound> extends Profile<S, C> {
37  
38      /**
39       * Returns the {@link Compound} in query sequence at given column index in alignment.
40       *
41       * @param alignmentIndex column index in alignment
42       * @return the query sequence element
43       * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
44       */
45      C getCompoundInQueryAt(int alignmentIndex);
46  
47      /**
48       * Returns the {@link Compound} in target sequence at given column index in alignment.
49       *
50       * @param alignmentIndex column index in alignment
51       * @return the target sequence element
52       * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
53       */
54      C getCompoundInTargetAt(int alignmentIndex);
55  
56      /**
57       * Returns the query index corresponding to a given alignment column.
58       *
59       * @param alignmentIndex column index in alignment
60       * @return index in query {@link Sequence}
61       * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
62       */
63      int getIndexInQueryAt(int alignmentIndex);
64  
65      /**
66       * Returns the query index corresponding to a given target index.
67       *
68       * @param targetIndex index in target {@link Sequence}
69       * @return index in query {@link Sequence}
70       * @throws IndexOutOfBoundsException if targetIndex < 1 or targetIndex > {@link #getTarget()}.getLength()
71       */
72      int getIndexInQueryForTargetAt(int targetIndex);
73  
74      /**
75       * Returns the target index corresponding to a given alignment column.
76       *
77       * @param alignmentIndex column index in alignment
78       * @return index in target {@link Sequence}
79       * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
80       */
81      int getIndexInTargetAt(int alignmentIndex);
82  
83      /**
84       * Returns the target index corresponding to a given query index.
85       *
86       * @param queryIndex index in query {@link Sequence}
87       * @return index in target {@link Sequence}
88       * @throws IndexOutOfBoundsException if queryIndex < 1 or queryIndex > {@link #getQuery()}.getLength()
89       */
90      int getIndexInTargetForQueryAt(int queryIndex);
91  
92      /**
93       * Returns the number of indices for which both the query and target sequences have an identical {@link Compound}.
94       *
95       * @return the number of identical indices
96       */
97      int getNumIdenticals();
98  
99      /**
100      * Returns the number of indices for which both the query and target sequences have a similar {@link Compound}.
101      *
102      * @return the number of similar indices
103      */
104     int getNumSimilars();
105 
106     /**
107      * Returns the first {@link AlignedSequence} of the pair.
108      *
109      * @return the first {@link AlignedSequence} of the pair
110      */
111     AlignedSequence<S, C> getQuery();
112 
113     /**
114      * Returns the second {@link AlignedSequence} of the pair.
115      *
116      * @return the second {@link AlignedSequence} of the pair
117      */
118     AlignedSequence<S, C> getTarget();
119 
120 }

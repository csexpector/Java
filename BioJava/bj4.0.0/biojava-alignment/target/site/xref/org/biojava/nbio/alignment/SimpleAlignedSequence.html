39  /**
40   * Implements a data structure for a {@link Sequence} within an alignment.
41   *
42   * @author Mark Chapman
43   * @param <C> each element of the {@link Sequence} is a {@link Compound} of type C
44   */
45  public class SimpleAlignedSequence<S extends Sequence<C>, C extends Compound> implements AlignedSequence<S, C> {
46  
47      private static final String gap = "-";
48  
49      // always stored
50      private AlignedSequence<S, C> prev;
51      private S original;
52      private int length, numBefore, numAfter;
53      private Location location;
54  
55      // cached (lazily initialized)
56      private int numGaps = -1;
57      private int[] alignmentFromSequence, sequenceFromAlignment;
58  
59      /**
60       * Creates an {@link AlignedSequence} for the given {@link Sequence} in a global alignment.
61       *
62       * @param original the original {@link Sequence} before alignment
63       * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
64       * @throws IllegalArgumentException if given sequence does not fit in alignment
65       */
66      public SimpleAlignedSequence(S original, List<Step> steps) {
67          this(original, steps, 0, 0);
68      }
69  
70      /**
71       * Creates an {@link AlignedSequence} for the given {@link Sequence} in a local alignment.
72       *
73       * @param original the original {@link Sequence} before alignment
74       * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
75       * @param numBefore number of {@link Compound}s before a local alignment
76       * @param numAfter number of {@link Compound}s after a local alignment
77       * @throws IllegalArgumentException if given sequence does not fit in alignment
78       */
79      public SimpleAlignedSequence(S original, List<Step> steps, int numBefore, int numAfter) {
80          this.original = original;
81          this.numBefore = numBefore;
82          this.numAfter = numAfter;
83          length = steps.size();
84          setLocation(steps);
85      }
86  
87      /**
88       * Creates a new {@link AlignedSequence} for the given {@link AlignedSequence} in a global alignment.
89       *
90       * @param prev the previous {@link AlignedSequence} before this alignment
91       * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
92       * @throws IllegalArgumentException if given sequence does not fit in alignment
93       */
94      public SimpleAlignedSequence(AlignedSequence<S, C> prev, List<Step> steps) {
95          this(prev, steps, 0, 0);
96      }
97  
98      /**
99       * Creates a new {@link AlignedSequence} for the given {@link AlignedSequence} in a local alignment.
100      *
101      * @param prev the previous {@link AlignedSequence} before this alignment
102      * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
103      * @param numBefore number of {@link Compound}s before a local alignment
104      * @param numAfter number of {@link Compound}s after a local alignment
105      * @throws IllegalArgumentException if given sequence does not fit in alignment
106      */
107     public SimpleAlignedSequence(AlignedSequence<S, C> prev, List<Step> steps, int numBefore, int numAfter) {
108         this.prev = prev;
109         this.original = prev.getOriginalSequence();
110         this.numBefore = numBefore;
111         this.numAfter = numAfter;
112         if (prev instanceof SimpleAlignedSequence<?, ?>) {
113             SimpleAlignedSequence<?, ?> p = (SimpleAlignedSequence<?, ?>) prev;
114             this.numBefore += p.numBefore;
115             this.numAfter += p.numAfter;
116         }
117         length = steps.size();
118         setLocation(steps);
119     }

123     @Override
124     public void clearCache() {
125         alignmentFromSequence = null;
126         sequenceFromAlignment = null;
127     }

129     @Override
130     public int getAlignmentIndexAt(int sequenceIndex) {
131         if (alignmentFromSequence == null) {
132             alignmentFromSequence = new int[original.getLength()];
133             int s = 1, a = 1;
134             for (int i = 0; i < numBefore; i++, s++) {
135                 alignmentFromSequence[s - 1] = a;
136             }
137             for (; s <= alignmentFromSequence.length && a <= length; s++, a++) {
138                 while (a <= length && isGap(a)) {
139                     a++;
140                 }
141                 alignmentFromSequence[s - 1] = a;
142             }
143             a--;
144             for (int i = 0; i < numAfter; i++, s++) {
145                 alignmentFromSequence[s - 1] = a;
146             }
147         }
148         return alignmentFromSequence[sequenceIndex - 1];
149     }

161     @Override
162     public int getNumGaps() {
163         if (numGaps == -1) {
164             numGaps = 0;
165             C cGap = getCompoundSet().getCompoundForString(gap);
166             boolean inGap = false;
167             for (C compound : getAsList()) {
168                 if (compound == null || compound.equalsIgnoreCase(cGap)) {
169                     if (!inGap) {
170                         numGaps++;
171                         inGap = true;
172                     }
173                 } else {
174                     inGap = false;
175                 }
176             }
177         }
178         return numGaps;
179     }

192     @Override
193     public int getSequenceIndexAt(int alignmentIndex) {
194         if (sequenceFromAlignment == null) {
195             sequenceFromAlignment = new int[length];
196             int a = 1, s = numBefore + 1;
197             for (int i = 0; i < getStart().getPosition(); i++, a++) {
198                 sequenceFromAlignment[a - 1] = s;
199             }
200             for (; a <= length; a++) {
201                 if (!isGap(a)) {
202                     s++;
203                 }
204                 sequenceFromAlignment[a - 1] = s;
205             }
206         }
207         return sequenceFromAlignment[alignmentIndex - 1];
208     }

220     @Override
221     public boolean isGap(int alignmentIndex) {
222         if (getStart().getPosition() <= alignmentIndex && alignmentIndex <= getEnd().getPosition()) {
223             if (!location.isComplex()) {
224                 return false;
225             }
226             for (Location sublocation : location) {
227                 if (sublocation.getStart().getPosition() <= alignmentIndex &&
228                         alignmentIndex <= sublocation.getEnd().getPosition()) {
229                     return false;
230                 }
231             }
232         }
233         return true;
234     }

238     @Override
239     public int countCompounds(C... compounds) {
240         int count = 0;
241         List<C> search = Arrays.asList(compounds);
242         for (C compound : getAsList()) {
243             if (search.contains(compound)) {
244                 count++;
245             }
246         }
247         return count;
248     }

257         List<C> compounds = new ArrayList<C>();
258         for (int i = 1; i <= length; i++) {
259             compounds.add(getCompoundAt(i));

261         return compounds;

264     @Override
265     public C getCompoundAt(int alignmentIndex) {
266         return alignmentIndex >= 1 && alignmentIndex <= length && isGap(alignmentIndex) ?
267                 getCompoundSet().getCompoundForString(gap) :
268                 original.getCompoundAt(getSequenceIndexAt(alignmentIndex));
269     }

276     @Override
277     public int getIndexOf(C compound) {
278         for (int i = 1; i <= length; i++) {
279             if (compound.equals(getCompoundAt(i))) {
280                 return i;
281             }
282         }
283         return -1;
284     }

286     @Override
287     public int getLastIndexOf(C compound) {
288         for (int i = length; i >= 1; i--) {
289             if (compound.equals(getCompoundAt(i))) {
290                 return i;
291             }
292         }
293         return -1;
294     }

328     // helper method to initialize the location
329     private void setLocation(List<Step> steps) {
330         List<Location> sublocations = new ArrayList<Location>();
331         int start = 0, step = 0, oStep = numBefore+numAfter, oMax = this.original.getLength(), pStep = 0, pMax =
332                 (prev == null) ? 0 : prev.getLength();
333         boolean inGap = true;
334 
335         // build sublocations: pieces of sequence separated by gaps
336         for (; step < length; step++) {
337             boolean isGapStep = (steps.get(step) == Step.GAP),
338                     isGapPrev = (pStep < pMax && prev.isGap(pStep + 1));
339             if (!isGapStep && !isGapPrev) {
340                 oStep++;
341                 if (inGap) {
342                     inGap = false;
343                     start = step + 1;
344                 }
345             } else if (!inGap) {
346                 inGap = true;
347                 sublocations.add(new SimpleLocation(start, step, Strand.UNDEFINED));
348             }
349             if (prev != null && !isGapStep) {
350                 pStep++;
351             }
352         }
353         if (!inGap) {
354             sublocations.add(new SimpleLocation(start, step, Strand.UNDEFINED));
355         }
356 
357         // combine sublocations into 1 Location
358         if (sublocations.size() == 0) {
359         	location = null;
360         } else if (sublocations.size() == 1) {
361         	location = sublocations.get(0);
362         } else {
363         	location = new SimpleLocation(sublocations.get(0).getStart(), sublocations.get(sublocations.size() - 1).getEnd(),
364         			Strand.UNDEFINED,
365                     false, sublocations);
366         }
367         // TODO handle circular alignments
368 
369         // check that alignment has correct number of compounds in it to fit original sequence
370         if (step != length || oStep != oMax || pStep != pMax) {
371             throw new IllegalArgumentException("Given sequence does not fit in alignment.");
372         }
373     }

49  	@Test
50  	public void readAdvancedXml() throws JAXBException, IOException{
51  		ElementTable iTable = new ElementTable();
52  		// Get a JAXB Context for the object we created above
53  		JAXBContext jc = JAXBContext.newInstance(iTable.getClass());
54  		Unmarshaller u = jc.createUnmarshaller();
55  		iTable = (ElementTable)u.unmarshal(new FileInputStream("./src/main/resources/AdvancedElementMass.xml" ) );
56  		iTable.populateMaps();
57  		
58  		AminoAcidCompositionTable aTable = new AminoAcidCompositionTable();
59  		// Get a JAXB Context for the object we created above
60  		JAXBContext jc2 = JAXBContext.newInstance(aTable.getClass());
61  		Unmarshaller u2 = jc2.createUnmarshaller();
62  		
63  		aTable = (AminoAcidCompositionTable)u2.unmarshal(new FileInputStream("./src/main/resources/AdvancedAminoAcidComposition.xml" ) );
64  		aTable.computeMolecularWeight(iTable);
65  		//Assert the weight of the radioactives
66  		String sequence = "00000";
67  		assertEquals(398.558744445, PeptideProperties.getMolecularWeightBasedOnXML(sequence, aTable), delta);
68  		sequence = "1111";
69  		assertEquals(702.335483556, PeptideProperties.getMolecularWeightBasedOnXML(sequence, aTable), delta);
70  		sequence = "JJJJ";
71  		assertEquals(0.0, PeptideProperties.getMolecularWeightBasedOnXML(sequence, aTable), delta);
72  	}

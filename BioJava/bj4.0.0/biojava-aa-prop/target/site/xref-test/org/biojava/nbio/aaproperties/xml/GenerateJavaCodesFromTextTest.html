36  	
37  	private final static Logger logger = LoggerFactory.getLogger(GenerateJavaCodesFromTextTest.class);

39  	/*
40  	 * Generate java codes from two text files; Symbol2Name.txt and Symbol2Weight.txt
41  	 */
42  	@Test
43  	public void generateCodes() throws IOException{
44  		BufferedReader input = new BufferedReader(new FileReader("./src/test/resources/Symbol2Name.txt"));
45  		Map<String, String> symbol2Name = new HashMap<String, String>();
46  		String line = input.readLine();//Header line is not required
47  		while((line = input.readLine()) != null){
48  			String[] sA = line.split("\t");
49  			symbol2Name.put(sA[1].trim(), sA[2].trim());
50  		}
51  		input.close();
52  		
53  		input = new BufferedReader(new FileReader("./src/test/resources/Symbol2Weight.txt"));
54  		List<String> elementNameList = new ArrayList<String>();
55  		String elementName = null;
56  		String elementNameLower = null;
57  		Double elementMass = null;
58  		while((line = input.readLine()) != null){
59  			String[] sA = line.split("\t");
60  			if(sA[0].length() > 0){
61  				//Elements
62  				logger.info("{}.setIsotopes(iList);", elementNameLower);
63  				//int decimalPlace = getDecimalPlace(elementMass + "");
64  				//System.out.println("assertEquals(" + elementMass + ", Utils.roundToDecimals(" + elementNameLower + 
65  					//	".getMass(), " + decimalPlace + "));");
66  				
67  				String symbol = sA[1].trim();
68  				elementName = symbol2Name.get(symbol);
69  				elementNameLower = elementName.toLowerCase() ;
70  				int protonNumber = Integer.parseInt(sA[0].trim());
71  				elementMass = cleanNumber(sA[5]); 
72  				if(protonNumber > 82) break;
73  				elementNameList.add(elementNameLower);
74  
75  				logger.info("iList = new ArrayList<Isotope>();");
76  				logger.info("Element {} = new Element(\"{}\", \"{}\", {}, null, {});",
77  						elementNameLower, elementName, symbol, protonNumber, elementMass);
78  			}
79  			int neutronNumber = Integer.parseInt(sA[2]);
80  			double weight = cleanNumber(sA[3]);
81  			//if(sA.length > 4 && sA[4].length() > 0) abundance = cleanNumber(sA[4]);
82  			logger.info("iList.add(new Isotope(\"{}-{}\", {}, {}));",
83  					elementName, neutronNumber, neutronNumber, weight);
84  		}
85  		input.close();
86  		
87  		logger.info("List<Element> eList = new ArrayList<Element>();");
88  		for(String e:elementNameList) logger.info("eList.add({});", e);
89  	}
90  	
91  	private double cleanNumber(String s){
92  		int index = s.indexOf("(");
93  		if(index != -1) s = s.substring(0, index);
94  		return Double.parseDouble(s.replace(" ", "").replace("[", "").replace("]", ""));
95  	}

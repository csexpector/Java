320 	@Test
321 	public void testNetCharge(){
322 		/*
323 		 * Test for Innovagen
324 		 */
325 		//http://www.innovagen.se/custom-peptide-synthesis/peptide-property-calculator/peptide-property-calculator.asp
326 		assertEquals(2.9, PeptideProperties.getNetCharge(sequence, false), 0.1);
327 		assertEquals(0.0, PeptideProperties.getNetCharge(fullInvalidSequence, false), 0.1);
328 
329 		assertEquals(-3.2, PeptideProperties.getNetCharge("ACCACAAADADADACA", false), 0.1);
330 		/*
331 		 * Did not test for Expasy because in their website, net charge is not given.
332 		 * However, since Isoelectric point is given which rely on getNetCharge values therefore, 
333 		 * 	we infer that if getIsoelectricPoint is correct, getNetCharge would be correct for Expasy.
334 		 */
335 		
336 		
337 		/*
338 		 * Provided by Steve Darnell to compare the difference between Innovagen and Expasy
339 		 */
340 		String[] alpha = {"A",/*"B",*/"C","D","E","F","G","H","I",/*"J",*/
341 				"K","L","M","N",/*"O",*/"P","Q","R","S","T",
342 				/*"U",*/"V","W",/*"X",*/"Y"/*,"Z"*/};
343 		for (String aa : alpha) {
344 			String p = String.format("AA%sAA", aa);
345 			logger.info(p);
346 			logger.info("pH\tInnovagen\tExpasy\tdiff");
347 			for ( int i = 1; i < 15; i++) {
348 				double phPoint = (new Double(i)).doubleValue();
349 				double chrgInnovagen = PeptideProperties.getNetCharge(p,false,phPoint);
350 				double chrgExpasy = PeptideProperties.getNetCharge(p,true,phPoint);
351 				logger.info(String.format("%2.1f\t%2.2f\t%2.2f\t%2.2f", phPoint, chrgInnovagen, 
352 						chrgExpasy, chrgInnovagen - chrgExpasy));
353 			}
354 		}
355 	}

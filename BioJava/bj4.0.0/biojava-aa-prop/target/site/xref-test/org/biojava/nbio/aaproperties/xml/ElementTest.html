45  		File outputFile = new File(System.getProperty("java.io.tmpdir"),"ElementMass.xsd");
46  		outputFile.deleteOnExit();
47  		JAXBContext context = JAXBContext.newInstance(ElementTable.class);
48  		context.generateSchema(new SchemaGenerator(outputFile.toString()));
49  	}

51  	@Test
52  	public void readXml() throws JAXBException, IOException{
53  		ElementTable iTable = new ElementTable();
54  		// Get a JAXB Context for the object we created above
55  		JAXBContext jc = JAXBContext.newInstance(iTable.getClass());
56  		Unmarshaller u = jc.createUnmarshaller();
57  		iTable = (ElementTable)u.unmarshal(new FileInputStream("./src/main/resources/ElementMass.xml" ) );
58  		for(Element e:iTable.getElement()){
59  			logger.info("Element: {}", e);
60  		}
61  	}

21  package org.biojava.nbio.aaproperties.profeat.convertor;
22  
23  public class Convert2Charge extends Convertor{
24  	/**
25  	 * Class for the conversion of protein sequence into charge 
26  	 */
27  	
28  	@Override
29  	public char convert(char c){
30  		switch(c){
31  		case 'K':case 'R':
32  			return group1;//Positive
33  		case 'A':case 'N':case 'C':case 'Q':case 'G':case 'H':case 'I':case 'L':case 'M':case 'F':case 'P':case 'S':case 'T':case 'W':case 'Y':case 'V':
34  			return group2;//Neutral
35  		case 'D':case 'E':
36  			return group3;//Negative
37  		default:
38  			return unknownGroup;//Non-standard AA
39  		}
40  	}
41  	
42  	private static String[] subCategory = {"Positive", "Neutral", "Negative"};
43  	@Override
44  	public String[] getGrouping(){return Convert2Charge.subCategory;}
45  	@Override
46  	public String getAttribute(){return "Charge";}
47  }

21  package org.biojava.nbio.aaproperties.profeat.convertor;
22  
23  import org.biojava.nbio.core.sequence.ProteinSequence;
24  
25  public abstract class Convertor {
26  	/**
27  	 * Based on Table 2 of http://nar.oxfordjournals.org/content/34/suppl_2/W32.full.pdf<br/>;
28  	 * An abstract class to convert a protein sequence into representation of different attribute with each attribute having 3 groups.<br/>
29  	 * The seven different attributes are<p/>
30  	 * Hydrophobicity (Polar, Neutral, Hydrophobicity)<br/>
31  	 * Normalized van der Waals volume (Range 0 - 2.78, 2.95 - 4.0, 4.03 - 8.08)<br/>
32  	 * Polarity (Value 4.9 - 6.2, 8.0 - 9.2, 10.4 - 13.0)<br/>
33  	 * Polarizability (Value 0 - 1.08, 0.128 - 0.186, 0.219 - 0.409)<br/>
34  	 * Charge (Positive, Neutral, Negative)<br/>
35  	 * Secondary structure (Helix, Strand, Coil)<br/>
36  	 * Solvent accessibility (Buried, Exposed, Intermediate)<br/>
37  	 * 
38  	 * @author kohchuanhock
39  	 * @version 2011.06.09
40  	 */
41  	public final static char group1 = '1';
42  	public final static char group2 = '2';
43  	public final static char group3 = '3';
44  	public final static char unknownGroup = '0';
45  	
46   	/**
47   	 * Returns the grouping of the amino acid character.
48  	 * The aminoAcid argument is preferably of non-ambiguous characters.
49  	 * Standard amino acids will be converted to '1', '2' or '3' depending on its grouping
50  	 * Non-standard amino acids are simply converted to '0'.
51  	 * 
52  	 * @param aminoAcid 
53  	 * 		an amino acid character preferably of non-ambiguous characters
54  	 * @return its grouping
55   	 */
56  	public abstract char convert(char aminoAcid);
57  	
58  	/**
59  	 * Returns the groupings of the attribute
60  	 * @return the groupings of the attribute
61  	 */
62  	public abstract String[] getGrouping();
63  	
64  	/**
65  	 * Return the attribute of the grouping
66  	 * @return the attribute of the grouping
67  	 */
68  	public abstract String getAttribute();
69  	
70  	/**
71  	 * Returns the converted sequence.
72  	 * The sequence argument must be a protein sequence consisting of preferably non-ambiguous characters only.
73  	 * Standard amino acids will be converted to '1', '2' or '3' depending on its grouping
74  	 * Non-standard amino acids are simply converted to '0'.
75  	 * 
76  	 * @param sequence 
77  	 * 		a protein sequence consisting of preferably non-ambiguous characters only
78  	 * @return the converted sequence
79  	 */
80  	public String convert(ProteinSequence sequence){
81  		String convertedSequence = "";
82  		String uppercaseSequence = sequence.getSequenceAsString().toUpperCase();
83  		for(int x = 0; x < uppercaseSequence.length(); x++){
84  			convertedSequence += convert(uppercaseSequence.charAt(x));
85  		}
86  		return convertedSequence;
87  	}
88  	
89  }

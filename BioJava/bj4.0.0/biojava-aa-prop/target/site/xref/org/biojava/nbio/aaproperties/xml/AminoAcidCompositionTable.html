21  package org.biojava.nbio.aaproperties.xml;
22  
23  import javax.xml.bind.annotation.XmlAccessType;
24  import javax.xml.bind.annotation.XmlAccessorType;
25  import javax.xml.bind.annotation.XmlElement;
26  import javax.xml.bind.annotation.XmlRootElement;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  import java.util.Set;
31  
32  
33  @XmlRootElement(name="compoundtable", namespace="http://biojava.org")
34  @XmlAccessorType(XmlAccessType.NONE)
35  public class AminoAcidCompositionTable {
36  	
37  	/**
38  	 * Contains the list of amino acid composition 
39  	 */
40  	@XmlElement(name = "compound", required = true)
41  	private List<AminoAcidComposition> aminoacid;
42  	
43  	/**
44  	 * Defines the amino acid compound set unique to this table
45  	 */
46  	private ModifiedAminoAcidCompoundSet modifiedAminoAcidCompoundSet;
47  	
48  	/**
49  	 * Stores the mapping of amino acid symbol to its molecular weight
50  	 */
51  	private Map<Character, Double> aaSymbol2MolecularWeight;
52  	
53  	public AminoAcidCompositionTable(){}
54  	
55  	public AminoAcidCompositionTable(List<AminoAcidComposition> aaList){
56  		this.setAminoacid(aaList);
57  	}
58  
59  	public ModifiedAminoAcidCompoundSet getAminoAcidCompoundSet(){
60  		return this.modifiedAminoAcidCompoundSet;
61  	}
62  	
63  	public List<AminoAcidComposition> getAminoacid() {
64  		return aminoacid;
65  	}
66  
67  	public void setAminoacid(List<AminoAcidComposition> aminoacid) {
68  		this.aminoacid = aminoacid;
69  	}
70  	
71  	public Set<Character> getSymbolSet(){
72  		return this.aaSymbol2MolecularWeight.keySet();
73  	}
74  	
75  	private void generatesAminoAcidCompoundSet(){
76  		this.modifiedAminoAcidCompoundSet = new ModifiedAminoAcidCompoundSet(this.aminoacid, this.aaSymbol2MolecularWeight);
77  	}
78  	
79  	/**
80  	 * Computes and store the molecular weight of each amino acid by its symbol in aaSymbol2MolecularWeight.
81  	 * 
82  	 * @param eTable
83  	 * 		Stores the mass of elements and isotopes
84  	 */
85  	public void computeMolecularWeight(ElementTable eTable){
86  		this.aaSymbol2MolecularWeight = new HashMap<Character, Double>();
87  		for(AminoAcidComposition a:aminoacid){
88  			//Check to ensure that the symbol is of single character
89  			if(a.getSymbol().length() != 1){
90  				throw new Error(a.getSymbol() + " is not allowed. Symbols must be single character.\r\nPlease check AminoAcidComposition XML file");
91  			}
92  			//Check to ensure that the symbols are not repeated
93  			char c = a.getSymbol().charAt(0);
94  			if(this.aaSymbol2MolecularWeight.keySet().contains(c)){
95  				throw new Error("Symbol " + c + " is repeated.\r\n" +
96  						"Please check AminoAcidComposition XML file to ensure there are no repeated symbols. Note that this is case-insensitive.\r\n" +
97  						"This means that having 'A' and 'a' would be repeating.");
98  			}
99  			double total = 0.0;
100 			if(a.getElementList() != null){
101 				for(Name2Count element:a.getElementList()){
102 					element.getName();
103 					if(eTable.getElement(element.getName()) == null){
104 						throw new Error("Element " + element.getName() + " could not be found. " +
105 								"\r\nPlease ensure that its name is correct in AminoAcidComposition.xml and is defined in ElementMass.xml.");
106 					}
107 					eTable.getElement(element.getName()).getMass();
108 					total += eTable.getElement(element.getName()).getMass() * element.getCount();
109 				}
110 			}
111 			if(a.getIsotopeList() != null){
112 				for(Name2Count isotope:a.getIsotopeList()){
113 					isotope.getName();
114 					if(eTable.getIsotope(isotope.getName()) == null){
115 						throw new Error("Isotope " + isotope.getName() + " could not be found. " +
116 							"\r\nPlease ensure that its name is correct in AminoAcidComposition.xml and is defined in ElementMass.xml.");
117 					}
118 					eTable.getIsotope(isotope.getName()).getMass();
119 					total += eTable.getIsotope(isotope.getName()).getMass() * isotope.getCount();
120 				}
121 			}
122 			c = a.getSymbol().charAt(0);
123 			this.aaSymbol2MolecularWeight.put(c, total);
124 		}
125 		generatesAminoAcidCompoundSet();
126 	}
127 	
128 	/**
129 	 * @param aaSymbol
130 	 * 	Standard symbol of Amino Acid
131 	 * @return the molecular weight given its symbol
132 	 * @throws NullPointerException 
133 	 * 	thrown if AminoAcidCompositionTable.computeMolecularWeight(ElementTable) is not called before this method
134 	 */
135 	public double getMolecularWeight(Character aaSymbol) throws NullPointerException{
136 		if(this.aaSymbol2MolecularWeight == null){
137 			throw new NullPointerException("Please call AminoAcidCompositionTable.computeMolecularWeight(ElementTable) before this method");
138 		}
139 		Double d = this.aaSymbol2MolecularWeight.get(aaSymbol);
140 		if(d == null)
141 			return 0;
142 		else
143 			return d; 
144 	}
145 }

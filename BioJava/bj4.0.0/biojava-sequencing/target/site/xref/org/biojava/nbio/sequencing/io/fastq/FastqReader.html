21  package org.biojava.nbio.sequencing.io.fastq;
22  
23  import java.io.File;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.net.URL;
27  
28  /**
29   * Reader for FASTQ formatted sequences.
30   *
31   * @since 3.0.3
32   */
33  public interface FastqReader
34  {
35  
36      /**
37       * Parse the specified readable.
38       *
39       * @param readable readable, must not be null
40       * @param listener low-level event based parser callback, must not be null
41       * @throws IOException if an I/O error occurs
42       */
43      void parse(Readable readable, ParseListener listener) throws IOException;
44  
45      /**
46       * Stream the specified readable.
47       *
48       * @param readable readable, must not be null
49       * @param listener event based reader callback, must not be null
50       * @throws IOException if an I/O error occurs
51       */
52      void stream(Readable readable, StreamListener listener) throws IOException;
53  
54      /**
55       * Read zero or more FASTQ formatted sequences from the specified file.
56       *
57       * @param file file to read from, must not be null
58       * @return zero or more FASTQ formatted sequences read from the specified file
59       * @throws IOException if an I/O error occurs
60       */
61      Iterable<Fastq> read(File file) throws IOException;
62  
63      /**
64       * Read zero or more FASTQ formatted sequences from the specified url.
65       *
66       * @param url URL to read from, must not be null
67       * @return zero or more FASTQ formatted sequences read from the specified url
68       * @throws IOException if an I/O error occurs
69       */
70      Iterable<Fastq> read(URL url) throws IOException;
71  
72      /**
73       * Read zero or more FASTQ formatted sequences from the specified input stream.
74       *
75       * @param inputStream input stream to read from, must not be null
76       * @return zero or more FASTQ formatted sequences read from the specified input stream
77       * @throws IOException if an I/O error occurs
78       */
79      Iterable<Fastq> read(InputStream inputStream) throws IOException;
80  }
